module UVMVS.Core.Data.Pair.IEq where

open import UVMVS.Core.Init
open import UVMVS.Core.Classes

module _ {ℓ₁ ℓ₂ ℓᴿ₁ ℓᴿ₂} {A : Set ℓ₁} {B : Set ℓ₂} {{_ : has[≍][ ℓᴿ₁ ] A}} {{_ : has[≍][ ℓᴿ₂ ] B}} where

  data _≍⸢∧⸣_ : (x y : A ∧ B) → Set (ℓᴿ₁ ⩛ ℓᴿ₂) where
    ⟨_,_⟩ : ∀ {x₁ x₂ y₁ y₂} (ε₁ : x₁ ≍ y₁) (ε₂ : x₂ ≍ y₂) → ⟨ x₁ , x₂ ⟩ ≍⸢∧⸣ ⟨ y₁ , y₂ ⟩

  data _≭⸢∧⸣_ : (x y : A ∧ B) → Set (ℓᴿ₁ ⩛ ℓᴿ₂) where
    Inl : ∀ {x₁ x₂ y₁ y₂} (ε : x₁ ≭ y₁) → ⟨ x₁ , x₂ ⟩ ≭⸢∧⸣ ⟨ y₁ , y₂ ⟩
    Inr : ∀ {x₁ x₂ y₁ y₂} (ε : x₂ ≭ y₂) → ⟨ x₁ , x₂ ⟩ ≭⸢∧⸣ ⟨ y₁ , y₂ ⟩
  
  instance
    has[≍][∧] : has[≍][ ℓᴿ₁ ⩛ ℓᴿ₂ ] (A ∧ B)
    has[≍][∧] = record { _≍_ = _≍⸢∧⸣_ ; _≭_ = _≭⸢∧⸣_ }

  module _ {{_ : cor[≍] A}} {{_ : cor[≍] B}} where

    abstract
      sound[≍⸢∧⸣]<_,_> : ∀ (x y : A ∧ B) → x ≍⸢∧⸣ y → x ≡ y
      sound[≍⸢∧⸣]< ⟨ x₁ , x₂ ⟩ , ⟨ y₁ , y₂ ⟩ > ⟨ ε₁ , ε₂ ⟩ rewrite sound[≍]< x₁ , y₁ > ε₁ | sound[≍]< x₂ , y₂ > ε₂ = ↯
      
      xRx[≍⸢∧⸣]<_> : ∀ (x : A ∧ B) → x ≍⸢∧⸣ x
      xRx[≍⸢∧⸣]< ⟨ x₁ , x₂ ⟩ > = ⟨ xRx , xRx ⟩
      
      ¬xRx[≭⸢∧⸣] : ∀ {x : A ∧ B} → ¬ (x ≭⸢∧⸣ x)
      ¬xRx[≭⸢∧⸣] (Inl ε) = ¬xRx ε
      ¬xRx[≭⸢∧⸣] (Inr ε) = ¬xRx ε
      
      ◇[≭⸢∧⸣] : ∀ {x y} → x ≭⸢∧⸣ y → y ≭⸢∧⸣ x
      ◇[≭⸢∧⸣] (Inl ε) = Inl (◇ ε)
      ◇[≭⸢∧⸣] (Inr ε) = Inr (◇ ε)

    open make[≍] (A ∧ B) sound[≍⸢∧⸣]<_,_> xRx[≍⸢∧⸣]<_> ¬xRx[≭⸢∧⸣] ◇[≭⸢∧⸣]

    instance
      reflexive[≍⸢∧⸣] = reflexive[≍]†
      symmetric[≍⸢∧⸣] = symmetric[≍]†
      transitive[≍⸢∧⸣] = transitive[≍]†
      irreflexive[≭⸢∧⸣] = irreflexive[≭]†
      symmetric[≭⸢∧⸣] = symmetric[≭]†
      cor[≍][∧] = cor[≍]†
    
    _ = ⌁ AT reflexive           _≍⸢∧⸣_
    _ = ⌁ AT symmetric           _≍⸢∧⸣_
    _ = ⌁ AT transitive          _≍⸢∧⸣_
    _ = ⌁ AT irreflexive         _≭⸢∧⸣_
    _ = ⌁ AT symmetric           _≭⸢∧⸣_
    _ = ⌁ AT cor[≍]              (A ∧ B)
  _ = ⌁ AT has[≍][ ℓᴿ₁ ⩛ ℓᴿ₂ ] (A ∧ B)
