module UVMVS.Lib.Data.DVec where

open import UVMVS.Core

data ⟬_⟭[[_]] {ℓ₁ ℓ₂} {A : Set ℓ₁} (B : A → Set ℓ₂) : ∀ {N} → ⟬ A ⟭[ N ] → Set (ℓ₁ ⩛ ℓ₂) where
  [] : ⟬ B ⟭[[ [] ]]
  _∷_ : ∀ {N} {x : A} {xs : ⟬ A ⟭[ N ]} → B x → ⟬ B ⟭[[ xs ]] → ⟬ B ⟭[[ x ∷ xs ]]


_#[[_]] : ∀ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : A → Set ℓ₂} {N} {xs : ⟬ A ⟭[ N ]} (ys : ⟬ B ⟭[[ xs ]]) {ι : idx N} {x : A} (ε : ⟨ ι ↦ x ⟩∈ xs) → B x
(y ∷ ys) #[[ ᴢ ]] = y
(y ∷ ys) #[[ ꜱ ι ]] = ys #[[ ι ]]

_#[[_↦_]] : ∀ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : A → Set ℓ₂} {N} {xs : ⟬ A ⟭[ N ]} (ys : ⟬ B ⟭[[ xs ]]) {ι : idx N} {x : A} (ε : ⟨ ι ↦ x ⟩∈ xs) (y : B x) → ⟬ B ⟭[[ xs ]]
(y ∷ ys) #[[ ᴢ ↦ y′ ]] = y′ ∷ ys
(y ∷ ys) #[[ ꜱ ι ↦ y′ ]] = y ∷ (ys #[[ ι ↦ y′ ]])
