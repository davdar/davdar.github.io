module _ {ℓ} {A : Set ℓ} {{_ : has[≡?] A}} where

  _≡?ᴼ_ : ∀ (x y : ⦉ A ⦊) → ≡!
  • ≡?ᴼ • = [≡]
  • ≡?ᴼ ⟨ y ⟩ = [≢]
  ⟨ x ⟩ ≡?ᴼ • = [≢]
  ⟨ x ⟩ ≡?ᴼ ⟨ y ⟩ = x ≡? y

  instance
    has[≡?][option] : has[≡?] ⦉ A ⦊
    has[≡?][option] = record { _≡?_ = _≡?ᴼ_ }

  abstract
    corr[≡?ᴼ]<_,_> : ∀ (x y : ⦉ A ⦊) → (x ≡? y) ≡ (ιᴼ x ≡? ιᴼ y)
    corr[≡?ᴼ]< • , • > = ↯
    corr[≡?ᴼ]< • , ⟨ y ⟩ > = ↯
    corr[≡?ᴼ]< ⟨ x ⟩ , • > = ↯
    corr[≡?ᴼ]< ⟨ x ⟩ , ⟨ y ⟩ > = ↯

  module _ {{_ : cor[≡?] A}} where
    open derive[cor[≡?]] ⦉ A ⦊ (𝟙 ∨ A) ιᴼ corr[≡?ᴼ]<_,_>
  
    instance
      cor[≡?][option] = cor[≡?]†
    
    _ = ⌁ AT cor[≡?] ⦉ A ⦊
  _ = ⌁ AT has[≡?] ⦉ A ⦊
