module UVMVS.Core.Data.List.IEq where

open import UVMVS.Core.Init
open import UVMVS.Core.Classes

module _ {ℓ ℓᴿ} {A : Set ℓ} {{_ : has[≍][ ℓᴿ ] A}} where
  data _≍ᴸ_ : (xs ys : ⟬ A ⟭) → Set ℓᴿ where
    [] : [] ≍ᴸ []
    _∷_ : ∀ {x y xs ys} (ε : x ≍ y) (εs : xs ≍ᴸ ys) → (x ∷ xs) ≍ᴸ (y ∷ ys)

  data _≭ᴸ_ : (xs ys : ⟬ A ⟭) → Set ℓᴿ where
    []≭[∷] : ∀ {y ys} → [] ≭ᴸ (y ∷ ys)
    [∷]≭[] : ∀ {x xs} → (x ∷ xs) ≭ᴸ []
    _∷• : ∀ {x y xs ys} (ε : x ≭ y) → (x ∷ xs) ≭ᴸ (y ∷ ys)
    •∷_ : ∀ {x y xs ys} (εs : xs ≭ᴸ ys) → (x ∷ xs) ≭ᴸ (y ∷ ys)

  instance
    has[≍][list] : has[≍][ ℓᴿ ] ⟬ A ⟭
    has[≍][list] = record { _≍_ = _≍ᴸ_ ; _≭_ = _≭ᴸ_ }
  
  module _ {{_ : cor[≍] A}} where
    abstract
      sound[≍ᴸ]<_,_> : ∀ (xs ys : ⟬ A ⟭) → xs ≍ᴸ ys → xs ≡ ys
      sound[≍ᴸ]< [] , [] > [] = ↯
      sound[≍ᴸ]< x ∷ xs , y ∷ ys > (ε ∷ εs) rewrite sound[≍]< x , y > ε | sound[≍ᴸ]< xs , ys > εs = ↯
      
      xRx[≍ᴸ]<_> : ∀ (xs : ⟬ A ⟭) → xs ≍ᴸ xs
      xRx[≍ᴸ]< [] > = []
      xRx[≍ᴸ]< x ∷ xs > = xRx ∷ xRx[≍ᴸ]< xs >
      
      ¬xRx[≭ᴸ] : ∀ {xs : ⟬ A ⟭} → ¬ (xs ≭ᴸ xs)
      ¬xRx[≭ᴸ] (ε ∷•) = ¬xRx ε
      ¬xRx[≭ᴸ] (•∷ ε) = ¬xRx[≭ᴸ] ε
      
      ◇[≭ᴸ] : ∀ {xs ys} → xs ≭ᴸ ys → ys ≭ᴸ xs
      ◇[≭ᴸ] []≭[∷] = [∷]≭[]
      ◇[≭ᴸ] [∷]≭[] = []≭[∷]
      ◇[≭ᴸ] (ε ∷•) = ◇ ε ∷•
      ◇[≭ᴸ] (•∷ ε) = •∷ ◇[≭ᴸ] ε
  
    open make[≍] ⟬ A ⟭ sound[≍ᴸ]<_,_> xRx[≍ᴸ]<_> ¬xRx[≭ᴸ] ◇[≭ᴸ]

    instance
      reflexive[≍ᴸ] = reflexive[≍]†
      symmetric[≍ᴸ] = symmetric[≍]†
      transitive[≍ᴸ] = transitive[≍]†
      irreflexive[≭ᴸ] = irreflexive[≭]†
      symmetric[≭ᴸ] = symmetric[≭]†
      cor[≍][list] = cor[≍]†
    
    _ = ⌁ AT reflexive    _≍ᴸ_
    _ = ⌁ AT symmetric    _≍ᴸ_
    _ = ⌁ AT transitive   _≍ᴸ_
    _ = ⌁ AT irreflexive  _≭ᴸ_
    _ = ⌁ AT symmetric    _≭ᴸ_
    _ = ⌁ AT cor[≍]       ⟬ A ⟭
  _ = ⌁ AT has[≍][ ℓᴿ ] ⟬ A ⟭
