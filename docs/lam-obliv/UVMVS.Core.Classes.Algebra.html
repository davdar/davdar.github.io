record split[+] {ℓ} (A : Set ℓ) {{_ : has[+] A}} {{_ : cor[+] A}} : Set ℓ where
  field
    split[+]<_,_> : ∀ (x y : A) (ε : x + y ≡ zero) → x ≡ zero ∧ y ≡ zero

  ¬lzero[+]<_,_> : ∀ (x y : A) (ε : x ≢ zero) → x + y ≢ zero
  ¬lzero[+]< x , y > ε = ⟨ (λ ε′ → contra[≡] ε (π₁ (split[+]< x , y > ε′))) ⟩

  ¬rzero[+]<_,_> : ∀ (x y : A) (ε : y ≢ zero) → x + y ≢ zero
  ¬rzero[+]< x , y > ε = ⟨ (λ ε′ → contra[≡] ε (π₂ (split[+]< x , y > ε′))) ⟩
open split[+] {{…}} public

{-# DISPLAY split[+].split[+]<_,_> _ = split[+]<_,_> #-}

record inj[+] {ℓ} (A : Set ℓ) {{_ : has[+] A}} {{_ : cor[+] A}} : Set ℓ where
  field
    linj[+]<_,_,_> : ∀ (x y z : A) (ε : x + y ≡ x + z) → y ≡ z
    rinj[+]<_,_,_> : ∀ (x y z : A) (ε : x + z ≡ y + z) → x ≡ y
open inj[+] {{…}} public

{-# DISPLAY inj[+].linj[+]<_,_,_> _ = linj[+]<_,_,_> #-}
{-# DISPLAY inj[+].rinj[+]<_,_,_> _ = rinj[+]<_,_,_> #-}

record has[×] {ℓ} (A : Set ℓ) {{_ : has[+] A}} : Set ℓ where

{-# DISPLAY has[×].one _ = one #-}

record cor[×] {ℓ} (A : Set ℓ) {{_ : has[+] A}} {{_ : has[×] A}} : Set ℓ where
  field
    lzero[×]<_> : ∀ (x : A) → zero × x ≡ zero
    rzero[×]<_> : ∀ (x : A) → x × zero ≡ zero
    lunit[×]<_> : ∀ (x : A) → one × x ≡ x
    runit[×]<_> : ∀ (x : A) → x × one ≡ x
    ldist[×]<_,_,_> : ∀ (x y z : A) → x × (y + z) ≡ (x × y) + (x × z)
    rdist[×]<_,_,_> : ∀ (x y z : A) → (x + y) × z ≡ (x × z) + (y × z)
    assoc[×]<_,_,_> : ∀ (x y z : A) → x × (y × z) ≡ (x × y) × z
    commu[×]<_,_> : ∀ (x y : A) → x × y ≡ y × x
    discr[×]<_,_> : ∀ (x y : A) (ε : x × y ≡ zero) → x ≡ zero ∨ y ≡ zero
  ¬zero[×]<_,_> : ∀ (x y : A) (ε₁ : x ≢ zero) (ε₂ : y ≢ zero) → x × y ≢ zero
  ¬zero[×]< x , y > ε₁ ε₂ = ⟨ (λ ε′ → CASE discr[×]< x , y > ε′ OF λ where (ʟ ε₁′) → contra[≡] ε₁ ε₁′ ; (ʀ ε₂′) → contra[≡] ε₂ ε₂′) ⟩
open cor[×] {{…}} public

{-# DISPLAY cor[×].lzero[×]<_> _ = lzero[×]<_> #-}
{-# DISPLAY cor[×].rzero[×]<_> _ = rzero[×]<_> #-}
{-# DISPLAY cor[×].lunit[×]<_> _ = lunit[×]<_> #-}
{-# DISPLAY cor[×].runit[×]<_> _ = runit[×]<_> #-}
{-# DISPLAY cor[×].ldist[×]<_,_,_> _ = ldist[×]<_,_,_> #-}
{-# DISPLAY cor[×].rdist[×]<_,_,_> _ = rdist[×]<_,_,_> #-}
{-# DISPLAY cor[×].assoc[×]<_,_,_> _ = assoc[×]<_,_,_> #-}
{-# DISPLAY cor[×].commu[×]<_,_> _ = commu[×]<_,_> #-}
{-# DISPLAY cor[×].discr[×]<_,_> _ = discr[×]<_,_> #-}

record has[⊡] {ℓ} (𝑓 : Set ℓ → Set ℓ) : Set (↟ ℓ) where
  infixl 16 _⊠_
  infixl 30 _⊡_
  infix  14 _᚜_᚛_
  field
    pure : ∀ {A} (x : A) → 𝑓 A
    _⊡_  : ∀ {A B} (f : 𝑓 (A → B)) (x : 𝑓 A) → 𝑓 B
  _⊠_ : ∀ {A B} (x : 𝑓 A) (y : 𝑓 B) → 𝑓 (A ∧ B)
  x ⊠ y = pure ⟨_,_⟩ ⊡ x ⊡ y
  _᚜_᚛_ : ∀ {A B C} (x : 𝑓 A) (f : A → B → C) (y : 𝑓 B) → 𝑓 C
  x ᚜ f ᚛ y = pure f ⊡ x ⊡ y
open has[⊡] {{…}} public

{-# DISPLAY has[⊡].pure _ = pure #-}
{-# DISPLAY has[⊡]._⊡_ _ = _⊡_ #-}

record has[≫=] {ℓ} (𝑓 : Set ℓ → Set ℓ) : Set (↟ ℓ) where
  infixl 2 _≫=_
  field
    return : ∀ {A} → A → 𝑓 A
    _≫=_ : ∀ {A B} → 𝑓 A → (A → 𝑓 B) → 𝑓 B
  _≫_ : ∀ {A B} → 𝑓 A → 𝑓 B → 𝑓 B
  xM ≫ yM = xM ≫= λ _ → yM
  _>>=_ = _≫=_
  _>>_ = _≫_
  mmap : ∀ {A B} → (A → B) → 𝑓 A → 𝑓 B
  mmap f xM = do
    x ← xM
    return $ f x
open has[≫=] {{…}} public

{-# DISPLAY has[≫=].return _ = return #-}
{-# DISPLAY has[≫=]._≫=_ _ = _≫=_ #-}

infixr 1 for-in-each
syntax for-in-each (λ x → uM) xM = FOR x IN xM EACH uM
for-in-each : ∀ {A : Set} {𝑓} {{_ : has[≫=] 𝑓}} → (A → 𝑓 𝟙) → ⟬ A ⟭ → 𝑓 𝟙
for-in-each f [] = return •
for-in-each f (x ∷ xs) = f x ≫ for-in-each f xs

record cor[≫=] {ℓ} (𝑓 : Set ℓ → Set ℓ) {{_ : has[≫=] 𝑓}} : Set (↟ ℓ) where
  field
    lunit[≫=]<_,_> : ∀ {A B} (x : A) (f : A → 𝑓 B) → (return x ≫= f) ≡ f x
    runit[≫=]<_> : ∀ {A} (xM : 𝑓 A) → (xM ≫= return) ≡ xM
    assoc[≫=]<_,_,_> : ∀ {A B C} (xM : 𝑓 A) (f : A → 𝑓 B) (g : B → 𝑓 C) → (xM ≫= f ≫= g) ≡ (xM ≫= (λ x → f x ≫= g))
open cor[≫=] {{…}} public

{-# DISPLAY cor[≫=].lunit[≫=]<_,_> _ = lunit[≫=]<_,_> #-}
{-# DISPLAY cor[≫=].runit[≫=]<_> _ = runit[≫=]<_> #-}
{-# DISPLAY cor[≫=].assoc[≫=]<_,_,_> _ = assoc[≫=]<_,_,_> #-}

record category {ℓ ℓᴿ} {A : Set ℓ} (_≼_ : A → A → Set ℓᴿ) {{_ : reflexive _≼_}} {{_ : transitive _≼_}} : Set (ℓ ⩛ ℓᴿ) where

open category {{…}} public
