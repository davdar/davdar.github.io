module LamObliv.Sampling.Semantics where

open import UVMVS

open import LamObliv.Common
open import LamObliv.Sampling.Syntax
open import LamObliv.Sampling.Weakening

data is-val {R I} {Σ : tsto R I} {τ} : Σ , [] , [] ⊢ᴱ τ , [] → Set where
  IsVal : ∀
    (v : Σ ⊢ⱽ τ)
    → is-val (Val v)

infix 10 [_]_[ς]⊢_
data [_]_[ς]⊢_ G {R} {I} (Σ : tsto R I)  (τ : type R) : Set where
  [_]⟨_⟩ : ∀ {G′} {I′}
    (Σ′ : tsto R I′)
    (ςs : 𝐼[ G′ + G ] ((Σ ⇧ⱽ Σ′ , [] , [] ⊢ᴱ τ , []) ∧ vsto (Σ ⇧ⱽ Σ′)))
    → [ G ] Σ [ς]⊢ τ

step : ∀ {R} {I} {Σ : tsto R I} → ∀ {τ} (G : ℕ) (σ : vsto Σ) (e : Σ , [] , [] ⊢ᴱ τ , []) → is-val e ∨ [ G ] Σ [ς]⊢ τ
step G σ (Var () ε₂ ε₃)
step G σ (Val v) = ʟ (IsVal v)
step G σ (Bit b r ℓ) = ʀ [ [] ]⟨ [ G ]⌊ ⟨ Val (Bit b ℓ) , σ ⟩ ⌋[𝐼] ⟩
step G σ (Flip ρ r) = ʀ [ [] ]⟨ ⟨ [ G ]⌊ ⟨ Val (Flip ᴏ) , σ ⟩ ⌋[𝐼] , [ G ]⌊ ⟨ Val (Flip ɪ) , σ ⟩ ⌋[𝐼] ⟩ ⟩
step G σ (CastS () ε₂ ε₃)
step G σ (CastS-V (Flip b)) = ʀ [ [] ]⟨ [ G ]⌊ ⟨ Val (Bit b ɪ) , σ ⟩ ⌋[𝐼] ⟩
step G σ (CastP e) with step G σ e
… | ʀ [ Σ′ ]⟨ es ⟩ = ʀ [ Σ′ ]⟨ mapOn es (λ where ⟨ e′ , σ′ ⟩ → ⟨ CastP e′ , σ′ ⟩) ⟩
… | ʟ (IsVal (Flip b)) = ʀ [ [] ]⟨ [ G ]⌊ ⟨ Val (Bit b ᴏ) , σ ⟩ ⌋[𝐼] ⟩
step G σ (MuxBit {A′ = []} {A″ = []} e₁ e₂ e₃) with step G σ e₁
… | ʀ [ Σ′ ]⟨ es ⟩ = ʀ [ Σ′ ]⟨ mapOn es (λ where ⟨ e₁′ , σ′ ⟩ → ⟨ MuxBit e₁′ (e₂ ⇧[Σ]ᴱ Σ′) (e₃ ⇧[Σ]ᴱ Σ′) , σ′ ⟩ ) ⟩
… | ʟ (IsVal (Bit b₁ ℓ₁)) with step G σ e₂
… | ʀ [ Σ′ ]⟨ es ⟩ = ʀ [ Σ′ ]⟨ mapOn es (λ where ⟨ e₂′ , σ′ ⟩ → ⟨ MuxBit (Val (Bit b₁ ℓ₁)) e₂′ (e₃ ⇧[Σ]ᴱ Σ′) , σ′ ⟩ ) ⟩
… | ʟ (IsVal (Bit b₂ ℓ₂)) with step G σ e₃
… | ʀ [ Σ′ ]⟨ es ⟩ = ʀ [ Σ′ ]⟨ mapOn es (λ where ⟨ e₃′ , σ′ ⟩ → ⟨ MuxBit (Val (Bit b₁ ℓ₁)) (Val (Bit b₂ ℓ₂)) e₃′ , σ′ ⟩ ) ⟩
… | ʟ (IsVal (Bit b₃ ℓ₃)) = ʀ [ [] ]⟨ [ G ]⌊ ⟨ Val (Pair (Bit (cond b₁ b₂ b₃) (ℓ₁ ⊔ ℓ₂ ⊔ ℓ₃)) (Bit (cond b₁ b₃ b₂) (ℓ₁ ⊔ ℓ₂ ⊔ ℓ₃))) , σ ⟩ ⌋[𝐼] ⟩
step G σ (MuxFlip {A′ = []} {A″ = []} e₁ e₂ e₃ ε₁ ε₂) with step G σ e₁
… | ʀ [ Σ′ ]⟨ es ⟩ = ʀ [ Σ′ ]⟨ mapOn es (λ where ⟨ e₁′ , σ′ ⟩ → ⟨ MuxFlip e₁′ (e₂ ⇧[Σ]ᴱ Σ′) (e₃ ⇧[Σ]ᴱ Σ′) ε₁ ε₂ , σ′ ⟩ ) ⟩
… | ʟ (IsVal (Bit b₁ ℓ₁)) with step G σ e₂
… | ʀ [ Σ′ ]⟨ es ⟩ = ʀ [ Σ′ ]⟨ mapOn es (λ where ⟨ e₂′ , σ′ ⟩ → ⟨ MuxFlip (Val (Bit b₁ ℓ₁)) e₂′ (e₃ ⇧[Σ]ᴱ Σ′) ε₁ ε₂ , σ′ ⟩ ) ⟩
… | ʟ (IsVal (Flip b₂)) with step G σ e₃
… | ʀ [ Σ′ ]⟨ es ⟩ = ʀ [ Σ′ ]⟨ mapOn es (λ where ⟨ e₃′ , σ′ ⟩ → ⟨ MuxFlip (Val (Bit b₁ ℓ₁)) (Val (Flip b₂)) e₃′ ε₁ ε₂ , σ′ ⟩ ) ⟩
… | ʟ (IsVal (Flip b₃)) = ʀ [ [] ]⟨ [ G ]⌊ ⟨ Val (Pair (Flip (cond b₁ b₂ b₃)) (Flip (cond b₁ b₃ b₂))) , σ ⟩ ⌋[𝐼] ⟩
step G σ (If {A′ = []} {A″₁ = []} {A″₂ = []} e₁ e₂ e₃ ε) with step G σ e₁
… | ʀ [ Σ′ ]⟨ es ⟩ = ʀ [ Σ′ ]⟨ mapOn es (λ where ⟨ e₁′ , σ′ ⟩ → ⟨ If e₁′ (e₂ ⇧[Σ]ᴱ Σ′) (e₃ ⇧[Σ]ᴱ Σ′) ε , σ′ ⟩ ) ⟩
… | ʟ (IsVal (Bit b₁ ℓ₁)) = CASE b₁ OF λ where
  ɪ → ʀ [ [] ]⟨ [ G ]⌊ ⟨ e₂ , σ ⟩ ⌋[𝐼] ⟩
  ᴏ → ʀ [ [] ]⟨ [ G ]⌊ ⟨ e₃ , σ ⟩ ⌋[𝐼] ⟩
step G σ (Pair {A′ = []} e₁ e₂) with step G σ e₁
… | ʀ [ Σ′ ]⟨ es ⟩ = ʀ [ Σ′ ]⟨ mapOn es (λ where ⟨ e₁′ , σ′ ⟩ → ⟨ Pair e₁′ (e₂ ⇧[Σ]ᴱ Σ′) , σ′ ⟩ ) ⟩
… | ʟ (IsVal v₁) with step G σ e₂
… | ʀ [ Σ′ ]⟨ es ⟩ = ʀ [ Σ′ ]⟨ mapOn es (λ where ⟨ e₂′ , σ′ ⟩ → ⟨ Pair (Val (v₁ ⇧[Σ]ⱽ Σ′)) e₂′ , σ′ ⟩ ) ⟩
… | ʟ (IsVal v₂) = ʀ [ [] ]⟨ [ G ]⌊ ⟨ Val (Pair v₁ v₂) , σ ⟩ ⌋[𝐼] ⟩
step G σ (Let {A′ = []} e₁ e₂) with step G σ e₁
… | ʀ [ Σ′ ]⟨ es ⟩ = ʀ [ Σ′ ]⟨ mapOn es (λ where ⟨ e₁′ , σ′ ⟩ → ⟨ Let e₁′ (e₂ ⇧[Σ]ᴱ Σ′) , σ′ ⟩ ) ⟩
… | ʟ (IsVal v₁) = ʀ [ [] ]⟨ [ G ]⌊ ⟨ s[⊢]< ᴢ > v₁ e₂ , σ ⟩ ⌋[𝐼] ⟩
step G σ (LetPair {A′ = []} e₁ e₂) with step G σ e₁
… | ʀ [ Σ′ ]⟨ es ⟩ = ʀ [ Σ′ ]⟨ mapOn es (λ where ⟨ e₁′ , σ′ ⟩ → ⟨ LetPair e₁′ (e₂ ⇧[Σ]ᴱ Σ′) , σ′ ⟩ ) ⟩
… | ʟ (IsVal (Pair v₁ v₂)) = ʀ [ [] ]⟨ [ G ]⌊ ⟨ s[⊢]< ᴢ > v₂ (s[⊢]< ᴢ > v₁ e₂) , σ ⟩ ⌋[𝐼] ⟩
step G σ (Ref e) with step G σ e
… | ʀ [ Σ′ ]⟨ es ⟩ = ʀ [ Σ′ ]⟨ mapOn es (λ where ⟨ e′ , σ′ ⟩ → ⟨ Ref e′ , σ′ ⟩ ) ⟩
… | ʟ (IsVal v) = ʀ [ [ _ ] ]⟨ [ G ]⌊ ⟨ Val (Addr ᴢ) , (v ⇧[Σ]ⱽ [ _ ]) ∷ (σ ⇧[Σ]ˢ [ _ ]) ⟩ ⌋[𝐼] ⟩
step G σ (Read/Write {A′ = []} e₁ e₂) with step G σ e₁
… | ʀ [ Σ′ ]⟨ es ⟩ = ʀ [ Σ′ ]⟨ mapOn es (λ where ⟨ e₁′ , σ′ ⟩ → ⟨ Read/Write e₁′ (e₂ ⇧[Σ]ᴱ Σ′) , σ′ ⟩ ) ⟩
… | ʟ (IsVal v) with step G σ e₂
… | ʀ [ Σ′ ]⟨ es ⟩ = ʀ [ Σ′ ]⟨ mapOn es (λ where ⟨ e₂′ , σ′ ⟩ → ⟨ Read/Write (Val (v ⇧[Σ]ⱽ Σ′)) e₂′ , σ′ ⟩ ) ⟩
… | ʟ (IsVal (Addr ι)) = ʀ [ [] ]⟨ [ G ]⌊ ⟨ Val (σ #[[ ι ]]) , σ #[[ ι ↦ v ]] ⟩ ⌋[𝐼] ⟩
