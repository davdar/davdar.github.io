_⊔ᴮ_ : (b₁ b₂ : 𝔹) → 𝔹
ᴏ ⊔ᴮ b = b
ɪ ⊔ᴮ b = ɪ

instance
  has[⊔][𝔹] : has[⊔] 𝔹
  has[⊔][𝔹] = record { _⊔_ = _⊔ᴮ_ }

abstract
  lbound[⊔ᴮ]<_,_> : ∀ (b₁ b₂ : 𝔹) → b₁ ⊑ (b₁ ⊔ᴮ b₂)
  lbound[⊔ᴮ]< ᴏ , b > = ᴏ
  lbound[⊔ᴮ]< ɪ , b > = ɪ

  rbound[⊔ᴮ]<_,_> : ∀ (b₁ b₂ : 𝔹) → b₂ ⊑ (b₁ ⊔ᴮ b₂)
  rbound[⊔ᴮ]< b₁ , ᴏ > = ᴏ
  rbound[⊔ᴮ]< ᴏ , ɪ > = ɪ
  rbound[⊔ᴮ]< ɪ , ɪ > = ɪ

  introd[⊔ᴮ]<_,_,_> : ∀ (b₁ b₂ b₃ : 𝔹) → b₁ ⊑ b₃ → b₂ ⊑ b₃ → (b₁ ⊔ᴮ b₂) ⊑ b₃ 
  introd[⊔ᴮ]< ᴏ , b₂ , b₃ > ε₁ ε₂ = ε₂
  introd[⊔ᴮ]< ɪ , b₂ , b₃ > ε₁ ε₂ = ε₁

instance
  cor[⊔][𝔹] : cor[⊔] 𝔹
  cor[⊔][𝔹] = record
    { lbound[⊔]<_,_> = lbound[⊔ᴮ]<_,_>
    ; rbound[⊔]<_,_> = rbound[⊔ᴮ]<_,_>
    ; introd[⊔]<_,_,_> = introd[⊔ᴮ]<_,_,_>
    }

_ = ⌁ AT cor[⊔] 𝔹

_⊓ᴮ_ : ∀ (b₁ b₂ : 𝔹) → 𝔹
ᴏ ⊓ᴮ b₂ = ᴏ
ɪ ⊓ᴮ b₂ = b₂

instance
  has[⊓][𝔹] : has[⊓] 𝔹
  has[⊓][𝔹] = record { _⊓_ = _⊓ᴮ_ }

abstract
  lbound[⊓ᴮ]<_,_> : ∀ (b₁ b₂ : 𝔹) → (b₁ ⊓ᴮ b₂) ⊑ b₁
  lbound[⊓ᴮ]< ᴏ , b₂ > = ᴏ
  lbound[⊓ᴮ]< ɪ , ᴏ > = ᴏ
  lbound[⊓ᴮ]< ɪ , ɪ > = ɪ

  rbound[⊓ᴮ]<_,_> : ∀ (b₁ b₂ : 𝔹) → (b₁ ⊓ᴮ b₂) ⊑ b₂ 
  rbound[⊓ᴮ]< ᴏ , b₂ > = ᴏ
  rbound[⊓ᴮ]< ɪ , ᴏ > = ᴏ
  rbound[⊓ᴮ]< ɪ , ɪ > = ɪ

  introd[⊓ᴮ]<_,_,_> : ∀ (b₁ b₂ b₃ : 𝔹) → b₁ ⊑ b₂ → b₁ ⊑ b₃ → b₁ ⊑ (b₂ ⊓ᴮ b₃)
  introd[⊓ᴮ]< ᴏ , b₂ , b₃ > ε₁ ε₂ = ᴏ
  introd[⊓ᴮ]< ɪ , .ɪ , .ɪ > ɪ ɪ = ɪ

instance
  cor[⊓][𝔹] : cor[⊓] 𝔹
  cor[⊓][𝔹] = record
    { lbound[⊓]<_,_> = lbound[⊓ᴮ]<_,_>
    ; rbound[⊓]<_,_> = rbound[⊓ᴮ]<_,_>
    ; introd[⊓]<_,_,_> = introd[⊓ᴮ]<_,_,_>
    }

_ = ⌁ AT cor[⊓] 𝔹

⊥ᴮ : 𝔹
