module UVMVS.Core.Data.Idx.IEq where

open import UVMVS.Core.Init
open import UVMVS.Core.Classes
open import UVMVS.Core.Data.Nat

open import UVMVS.Core.Data.Idx.Base
open import UVMVS.Core.Data.Idx.Injection

data _≍ᴵ_ : ∀ {N} (n m : idx N) → Set where
  ᴢ : ∀ {N} → (ᴢ AT idx (ꜱ N)) ≍ᴵ ᴢ
  ꜱ : ∀ {N} {n m : idx N} (ε : n ≍ᴵ m) → ꜱ n ≍ᴵ ꜱ m

data _≭ᴵ_ : ∀ {N} (n m : idx N) → Set where
  ᴢꜱ : ∀ {N} {n : idx N} → ᴢ ≭ᴵ ꜱ n
  ꜱᴢ : ∀ {N} {n : idx N} → ꜱ n ≭ᴵ ᴢ
  ꜱ : ∀ {N} {n m : idx N} (ε : n ≭ᴵ m) → ꜱ n ≭ᴵ ꜱ m

instance
  has[≍][idx] : ∀ {N} → has[≍][ L0 ] (idx N)
  has[≍][idx] = record { _≍_ = _≍ᴵ_ ; _≭_ = _≭ᴵ_ }

abstract
  isoto[≍ᴵ] : ∀ {N} {n m : idx N} → n ≍ᴵ m → ιᴵ n ≍ ιᴵ m
  isoto[≍ᴵ] ᴢ = ᴢ
  isoto[≍ᴵ] (ꜱ ε) = ꜱ (isoto[≍ᴵ] ε)
  
  isofr[≍ᴵ] : ∀ {N} {n m : idx N} → ιᴵ n ≍ ιᴵ m → n ≍ᴵ m
  isofr[≍ᴵ] {ꜱ N} {ᴢ} {ᴢ} ᴢ = ᴢ
  isofr[≍ᴵ] {ꜱ N} {ᴢ} {ꜱ m} ()
  isofr[≍ᴵ] {ꜱ N} {ꜱ n} {ᴢ} ()
  isofr[≍ᴵ] {ꜱ N} {ꜱ n} {ꜱ m} (ꜱ ε) = ꜱ (isofr[≍ᴵ] ε)
  
  isoto[≭ᴵ] : ∀ {N} {n m : idx N} → n ≭ᴵ m → ιᴵ n ≭ ιᴵ m
  isoto[≭ᴵ] ᴢꜱ = ᴢꜱ
  isoto[≭ᴵ] ꜱᴢ = ꜱᴢ
  isoto[≭ᴵ] (ꜱ ε) = ꜱ (isoto[≭ᴵ] ε)
  
  isofr[≭ᴵ] : ∀ {N} {n m : idx N} → ιᴵ n ≭ ιᴵ m → n ≭ᴵ m
  isofr[≭ᴵ] {ꜱ N} {ᴢ} {ᴢ} ()
  isofr[≭ᴵ] {ꜱ N} {ᴢ} {ꜱ m} ᴢꜱ = ᴢꜱ
  isofr[≭ᴵ] {ꜱ N} {ꜱ n} {ᴢ} ꜱᴢ = ꜱᴢ
  isofr[≭ᴵ] {ꜱ N} {ꜱ n} {ꜱ m} (ꜱ ε) = ꜱ (isofr[≭ᴵ] ε)

module _ {N} where
  open derive[cor[≍]] (idx N) ℕ ιᴵ isoto[≍ᴵ] isofr[≍ᴵ] isoto[≭ᴵ] isofr[≭ᴵ]
  
  instance
    reflexive[≍ᴵ] = reflexive[≍]†
    symmetric[≍ᴵ] = symmetric[≍]†
    transitive[≍ᴵ] = transitive[≍]†
    irreflexive[≭ᴵ] = irreflexive[≭]†
    symmetric[≭ᴵ] = symmetric[≭]†
    cor[≍][idx] = cor[≍]†

  _ = ⌁ AT reflexive    (_≍ᴵ_ {N})
  _ = ⌁ AT symmetric    (_≍ᴵ_ {N})
  _ = ⌁ AT transitive   (_≍ᴵ_ {N})
  _ = ⌁ AT irreflexive  (_≭ᴵ_ {N})
  _ = ⌁ AT symmetric    (_≭ᴵ_ {N})
  _ = ⌁ AT cor[≍]       (idx N)   
  _ = ⌁ AT has[≍][ L0 ] (idx N)
