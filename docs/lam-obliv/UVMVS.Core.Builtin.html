-- Names --

postulate 𝔯name : Set
{-# BUILTIN QNAME 𝔯name #-}

primitive
  primQNameEquality : 𝔯name → 𝔯name → 𝔹
  primQNameLess     : 𝔯name → 𝔯name → 𝔹

-- Fixity --

data 𝔯associativity : Set where
  LeftAssoc  : 𝔯associativity
  RightAssoc : 𝔯associativity
  NonAssoc   : 𝔯associativity

data 𝔯precedence : Set where
  Related   : ℤ → 𝔯precedence
  Unrelated : 𝔯precedence

data 𝔯fixity : Set where
  Fixity : 𝔯associativity → 𝔯precedence → 𝔯fixity

{-# BUILTIN ASSOC      𝔯associativity #-}
{-# BUILTIN ASSOCLEFT  LeftAssoc    #-}
{-# BUILTIN ASSOCRIGHT RightAssoc   #-}
{-# BUILTIN ASSOCNON   NonAssoc     #-}

{-# BUILTIN PRECEDENCE    𝔯precedence #-}
{-# BUILTIN PRECRELATED   Related    #-}
{-# BUILTIN PRECUNRELATED Unrelated  #-}

{-# BUILTIN FIXITY       𝔯fixity #-}
{-# BUILTIN FIXITYFIXITY Fixity #-}

primitive
  primQNameFixity : 𝔯name → 𝔯fixity

-- Metavariables --

postulate 𝔯meta : Set
{-# BUILTIN AGDAMETA 𝔯meta #-}

primitive
  primMetaEquality : 𝔯meta → 𝔯meta → 𝔹
  primMetaLess     : 𝔯meta → 𝔯meta → 𝔹
  primShowMeta     : 𝔯meta → 𝕊

-- Arguments --

-- Arguments can be (visible), {hidden}, or {{instance}}.
data 𝔯visibility : Set where
  Visible : 𝔯visibility
  Hidden : 𝔯visibility
  Instance : 𝔯visibility

{-# BUILTIN HIDING   𝔯visibility #-}
{-# BUILTIN VISIBLE  Visible    #-}
{-# BUILTIN HIDDEN   Hidden     #-}
{-# BUILTIN INSTANCE Instance  #-}

-- Arguments can be relevant or irrelevant.
data 𝔯relevance : Set where
  Relevant : 𝔯relevance
  Irrelevant : 𝔯relevance

{-# BUILTIN RELEVANCE  𝔯relevance  #-}
{-# BUILTIN RELEVANT   Relevant   #-}
{-# BUILTIN IRRELEVANT Irrelevant #-}

data 𝔯arg-info : Set where
  ArgInfo : 𝔯visibility → 𝔯relevance → 𝔯arg-info

data 𝔯arg (A : Set) : Set where
  Arg : 𝔯arg-info → A → 𝔯arg A

{-# BUILTIN ARGINFO    𝔯arg-info  #-}
{-# BUILTIN ARGARGINFO ArgInfo #-}
{-# BUILTIN ARG        𝔯arg      #-}
{-# BUILTIN ARGARG     Arg      #-}

-- Name abstraction --

data 𝔯abs (A : Set) : Set where
  Abs : 𝕊 → A → 𝔯abs A

{-# BUILTIN ABS    𝔯abs #-}
{-# BUILTIN ABSABS Abs #-}

-- Literals --

data 𝔯literal : Set where
  Nat    : (n : ℕ)    → 𝔯literal
  Word64 : (n : ℕ64)  → 𝔯literal
  Float  : (x : 𝔽)    → 𝔯literal
  Char   : (c : ℂ)    → 𝔯literal
  String : (s : 𝕊)    → 𝔯literal
  Name   : (x : 𝔯name) → 𝔯literal
  Meta   : (x : 𝔯meta) → 𝔯literal

{-# BUILTIN AGDALITERAL   𝔯literal #-}
{-# BUILTIN AGDALITNAT    Nat     #-}
{-# BUILTIN AGDALITWORD64 Word64  #-}
{-# BUILTIN AGDALITFLOAT  Float   #-}
{-# BUILTIN AGDALITCHAR   Char    #-}
{-# BUILTIN AGDALITSTRING String  #-}
{-# BUILTIN AGDALITQNAME  Name    #-}
{-# BUILTIN AGDALITMETA   Meta    #-}

-- Patterns --

data 𝔯pat : Set where
  Con    : (c : 𝔯name) (ps : ⟬ 𝔯arg 𝔯pat ⟭) → 𝔯pat
  Dot    : 𝔯pat
  Var    : (s : 𝕊)  → 𝔯pat
  Lit    : (l : 𝔯literal) → 𝔯pat
  Proj   : (f : 𝔯name)    → 𝔯pat
  Absurd : 𝔯pat

{-# BUILTIN AGDAPATTERN   𝔯pat #-}
{-# BUILTIN AGDAPATCON    Con     #-}
{-# BUILTIN AGDAPATDOT    Dot     #-}
{-# BUILTIN AGDAPATVAR    Var     #-}
{-# BUILTIN AGDAPATLIT    Lit     #-}
{-# BUILTIN AGDAPATPROJ   Proj    #-}
{-# BUILTIN AGDAPATABSURD Absurd  #-}

-- Terms --

data 𝔯sort   : Set
data 𝔯clause : Set
data 𝔯term   : Set
𝔯type = 𝔯term

data 𝔯term where
  Var       : (x : ℕ) (args : ⟬ 𝔯arg 𝔯term ⟭) → 𝔯term
  Con       : (c : 𝔯name) (args : ⟬ 𝔯arg 𝔯term ⟭) → 𝔯term
  Def       : (f : 𝔯name) (args : ⟬ 𝔯arg 𝔯term ⟭) → 𝔯term
  Lam       : (v : 𝔯visibility) (t : 𝔯abs 𝔯term) → 𝔯term
  PatLam   : (cs : ⟬ 𝔯clause ⟭) (args : ⟬ 𝔯arg 𝔯term ⟭) → 𝔯term
  Pi        : (a : 𝔯arg 𝔯type) (b : 𝔯abs 𝔯type) → 𝔯term
  AgdaSort : (s : 𝔯sort) → 𝔯term
  Lit       : (l : 𝔯literal) → 𝔯term
  Meta      : (x : 𝔯meta) → ⟬ 𝔯arg 𝔯term ⟭ → 𝔯term
  Unknown   : 𝔯term

data 𝔯sort where
  SSet    : (t : 𝔯term) → 𝔯sort
  Lit     : (n : ℕ) → 𝔯sort
  Unknown : 𝔯sort

data 𝔯clause where
  Clause        : (ps : ⟬ 𝔯arg 𝔯pat ⟭) (t : 𝔯term) → 𝔯clause
  AbsurdClause : (ps : ⟬ 𝔯arg 𝔯pat ⟭) → 𝔯clause

{-# BUILTIN AGDASORT    𝔯sort   #-}
{-# BUILTIN AGDATERM    𝔯term   #-}
{-# BUILTIN AGDACLAUSE  𝔯clause #-}

{-# BUILTIN AGDATERMVAR         Var       #-}
{-# BUILTIN AGDATERMCON         Con       #-}
{-# BUILTIN AGDATERMDEF         Def       #-}
{-# BUILTIN AGDATERMMETA        Meta      #-}
{-# BUILTIN AGDATERMLAM         Lam       #-}
{-# BUILTIN AGDATERMEXTLAM      PatLam   #-}
{-# BUILTIN AGDATERMPI          Pi        #-}
{-# BUILTIN AGDATERMSORT        AgdaSort #-}
{-# BUILTIN AGDATERMLIT         Lit       #-}
{-# BUILTIN AGDATERMUNSUPPORTED Unknown   #-}

{-# BUILTIN AGDASORTSET         SSet     #-}
{-# BUILTIN AGDASORTLIT         Lit     #-}
{-# BUILTIN AGDASORTUNSUPPORTED Unknown #-}

{-# BUILTIN AGDACLAUSECLAUSE Clause        #-}
{-# BUILTIN AGDACLAUSEABSURD AbsurdClause #-}

-- Definitions --

data 𝔯definition : Set where
  Function   : ⟬ 𝔯clause ⟭ → 𝔯definition
  DataType   : ℕ → ⟬ 𝔯name ⟭ → 𝔯definition
  RecordType : 𝔯name → ⟬ 𝔯arg 𝔯name ⟭ → 𝔯definition
  DataCons   : 𝔯name → 𝔯definition
  Axiom      : 𝔯definition
  PrimFun    : 𝔯definition

{-# BUILTIN AGDADEFINITION                𝔯definition  #-}
{-# BUILTIN AGDADEFINITIONFUNDEF          Function    #-}
{-# BUILTIN AGDADEFINITIONDATADEF         DataType   #-}
{-# BUILTIN AGDADEFINITIONRECORDDEF       RecordType #-}
{-# BUILTIN AGDADEFINITIONDATACONSTRUCTOR DataCons   #-}
{-# BUILTIN AGDADEFINITIONPOSTULATE       Axiom       #-}
{-# BUILTIN AGDADEFINITIONPRIMITIVE       PrimFun    #-}

-- Errors --

data 𝔯error-part : Set where
  StrErr  : 𝕊 → 𝔯error-part
  TermErr : 𝔯term → 𝔯error-part
  NameErr : 𝔯name → 𝔯error-part

{-# BUILTIN AGDAERRORPART       𝔯error-part #-}
{-# BUILTIN AGDAERRORPARTSTRING StrErr    #-}
{-# BUILTIN AGDAERRORPARTTERM   TermErr   #-}
{-# BUILTIN AGDAERRORPARTNAME   NameErr   #-}

-- TC monad --

postulate
  TC            : ∀ {a} → Set a → Set a
  returnTC      : ∀ {a} {A : Set a} → A → TC A
  bindTC        : ∀ {a b} {A : Set a} {B : Set b} → TC A → (A → TC B) → TC B
  unify         : 𝔯term → 𝔯term → TC 𝟙
  typeError     : ∀ {a} {A : Set a} → ⟬ 𝔯error-part ⟭ → TC A
  inferType     : 𝔯term → TC 𝔯type
  checkType     : 𝔯term → 𝔯type → TC 𝔯term
  normalise     : 𝔯term → TC 𝔯term
  reduce        : 𝔯term → TC 𝔯term
  catchTC       : ∀ {a} {A : Set a} → TC A → TC A → TC A
  quoteTC       : ∀ {a} {A : Set a} → A → TC 𝔯term
  unquoteTC     : ∀ {a} {A : Set a} → 𝔯term → TC A
  getContext    : TC ⟬ 𝔯arg 𝔯type ⟭
  extendContext : ∀ {a} {A : Set a} → 𝔯arg 𝔯type → TC A → TC A
  inContext     : ∀ {a} {A : Set a} → ⟬ 𝔯arg 𝔯type ⟭ → TC A → TC A
  freshName     : 𝕊 → TC 𝔯name
  declareDef    : 𝔯arg 𝔯name → 𝔯type → TC 𝟙
  defineFun     : 𝔯name → ⟬ 𝔯clause ⟭ → TC 𝟙
  getType       : 𝔯name → TC 𝔯type
  getDefinition : 𝔯name → TC 𝔯definition
  blockOnMeta   : ∀ {a} {A : Set a} → 𝔯meta → TC A
  commitTC      : TC 𝟙
  isMacro       : 𝔯name → TC 𝔹

  -- If the argument is 'true' makes the following primitives also normalise
  -- their results: inferType, checkType, quoteTC, getType, and getContext
  withNormalisation : ∀ {a} {A : Set a} → 𝔹 → TC A → TC A

  debugPrint : 𝕊 → ℕ → ⟬ 𝔯error-part ⟭ → TC 𝟙

{-# BUILTIN AGDATCM                  TC                #-}
{-# BUILTIN AGDATCMRETURN            returnTC          #-}
{-# BUILTIN AGDATCMBIND              bindTC            #-}
{-# BUILTIN AGDATCMUNIFY             unify             #-}
{-# BUILTIN AGDATCMTYPEERROR         typeError         #-}
{-# BUILTIN AGDATCMINFERTYPE         inferType         #-}
{-# BUILTIN AGDATCMCHECKTYPE         checkType         #-}
{-# BUILTIN AGDATCMNORMALISE         normalise         #-}
{-# BUILTIN AGDATCMREDUCE            reduce            #-}
{-# BUILTIN AGDATCMCATCHERROR        catchTC           #-}
{-# BUILTIN AGDATCMQUOTETERM         quoteTC           #-}
{-# BUILTIN AGDATCMUNQUOTETERM       unquoteTC         #-}
{-# BUILTIN AGDATCMGETCONTEXT        getContext        #-}
{-# BUILTIN AGDATCMEXTENDCONTEXT     extendContext     #-}
{-# BUILTIN AGDATCMINCONTEXT         inContext         #-}
{-# BUILTIN AGDATCMFRESHNAME         freshName         #-}
{-# BUILTIN AGDATCMDECLAREDEF        declareDef        #-}
{-# BUILTIN AGDATCMDEFINEFUN         defineFun         #-}
{-# BUILTIN AGDATCMGETTYPE           getType           #-}
{-# BUILTIN AGDATCMGETDEFINITION     getDefinition     #-}
{-# BUILTIN AGDATCMBLOCKONMETA       blockOnMeta       #-}
{-# BUILTIN AGDATCMCOMMIT            commitTC          #-}
{-# BUILTIN AGDATCMISMACRO           isMacro           #-}
{-# BUILTIN AGDATCMWITHNORMALISATION withNormalisation #-}
{-# BUILTIN AGDATCMDEBUGPRINT        debugPrint        #-}
