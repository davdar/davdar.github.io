module UVMVS.Core.Data.Idx.Order where

open import UVMVS.Core.Init
open import UVMVS.Core.Classes
open import UVMVS.Core.Data.Nat

open import UVMVS.Core.Data.Idx.Base
open import UVMVS.Core.Data.Idx.Injection

data _<ᴵ_ : ∀ {N} (x y : idx N) → Set where
  ᴢ : ∀ {N} {x : idx N} → ᴢ <ᴵ ꜱ x
  ꜱ : ∀ {N} {x y : idx N} (ε : x <ᴵ y) → ꜱ x <ᴵ ꜱ y

instance
  has[<][idx] : ∀ {N} → has[<][ L0 ] (idx N)
  has[<][idx] = record { _<_ = _<ᴵ_ }

abstract
  isoto[<ᴵ] : ∀ {N} {x y : idx N} → x <ᴵ y → ιᴵ x < ιᴵ y
  isoto[<ᴵ] ᴢ = ᴢ
  isoto[<ᴵ] (ꜱ ε) = ꜱ (isoto[<ᴵ] ε)
  
  isofr[<ᴵ] : ∀ {N} {x y : idx N} → ιᴵ x < ιᴵ y → x <ᴵ y
  isofr[<ᴵ] {ꜱ N} {ᴢ} {ᴢ} ()
  isofr[<ᴵ] {ꜱ N} {ᴢ} {ꜱ y} ᴢ = ᴢ
  isofr[<ᴵ] {ꜱ N} {ꜱ x} {ᴢ} ()
  isofr[<ᴵ] {ꜱ N} {ꜱ x} {ꜱ y} (ꜱ ε) = ꜱ (isofr[<ᴵ] ε)

_⋚?ᴵ_ : ∀ {N} (x y : idx N) → ⋚!
ᴢ ⋚?ᴵ ᴢ = [≡]
ᴢ ⋚?ᴵ ꜱ y = [<]
ꜱ x ⋚?ᴵ ᴢ = [>]
ꜱ x ⋚?ᴵ ꜱ y = x ⋚?ᴵ y

instance
  has[⋚?][idx] : ∀ {N} → has[⋚?] (idx N)
  has[⋚?][idx] = record { _⋚?_ = _⋚?ᴵ_ }

abstract
  corr[⋚?ᴵ]<_,_> : ∀ {N} (x y : idx N) → (x ⋚?ᴵ y) ≡ (ιᴵ x ⋚? ιᴵ y)
  corr[⋚?ᴵ]< ᴢ , ᴢ > = ↯
  corr[⋚?ᴵ]< ᴢ , ꜱ y > = ↯
  corr[⋚?ᴵ]< ꜱ x , ᴢ > = ↯
  corr[⋚?ᴵ]< ꜱ x , ꜱ y > = corr[⋚?ᴵ]< x , y >
  
module _ {N : ℕ} where
  open derive[cor[<]] (idx N) ℕ ιᴵ isoto[<ᴵ] isofr[<ᴵ]
  
  instance
    irreflexive[<ᴵ] = irreflexive[<]†
    asymmetric[<ᴵ] = asymmetric[<]†
    transitive[<ᴵ] = transitive[<]†
    cor[<][idx] = cor[<]†
  
  open derive[cor[⋚?]] (idx N) ℕ ιᴵ isoto[<ᴵ] isofr[<ᴵ] corr[⋚?ᴵ]<_,_>
  
  instance
    cor[⋚?][idx] = cor[⋚?]†
  
  _ = ⌁ AT irreflexive    (_<ᴵ_ {N})
  _ = ⌁ AT asymmetric     (_<ᴵ_ {N})
  _ = ⌁ AT transitive     (_<ᴵ_ {N})
  _ = ⌁ AT has[<][ L0 ]   (idx N)
  _ = ⌁ AT cor[<]         (idx N)
  _ = ⌁ AT has[⋚?]        (idx N)
  _ = ⌁ AT cor[⋚?]        (idx N)
