data idx : (N : ℕ) → Set where
  ᴢ : ∀ {N} → idx (ꜱ N)
  ꜱ : ∀ {N} (x : idx N) → idx (ꜱ N)

⌊_⌋ : ∀ {N} (n : ℕ) {{_ : (n ⋚? N) ≡ [<]}} → idx N
⌊_⌋ {ᴢ} ᴢ {{()}}
⌊_⌋ {ꜱ N} ᴢ {{ε}} = ᴢ
⌊_⌋ {ᴢ} (ꜱ n) {{()}}
⌊_⌋ {ꜱ N} (ꜱ n) {{ε}} = ꜱ (⌊_⌋ {N} n {{ε}})
