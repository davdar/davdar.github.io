open import UVMVS.Core.Classes

open import UVMVS.Core.Data.Vec.Base

module _ {ℓ} {A : Set ℓ} where
  ιⱽ : ∀ {N} → ⟬ A ⟭[ N ] → ⟬ A ⟭
  ιⱽ [] = []
  ιⱽ (x ∷ xs) = x ∷ ιⱽ xs
  
  abstract
    inj[ιⱽ]<_,_> : ∀ {N} (xs ys : ⟬ A ⟭[ N ]) → ιⱽ xs ≡ ιⱽ ys → xs ≡ ys
    inj[ιⱽ]< [] , [] > ↯ = ↯
    inj[ιⱽ]< x ∷ xs , y ∷ ys > ε with ιⱽ xs | ιⱽ ys | inj[ιⱽ]< xs , ys >
    inj[ιⱽ]< x ∷ xs , y ∷ ys > ↯ | fxs | .fxs | IH rewrite IH ↯ = ↯
  
  module _ {N : ℕ} where
    instance
      injective[ιⱽ] : injective (ιⱽ {N})
      injective[ιⱽ] = record { inj<_,_> = inj[ιⱽ]<_,_> }
    
    _ = ⌁ AT injective (ιⱽ {N})
