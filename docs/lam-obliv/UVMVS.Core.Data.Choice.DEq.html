module UVMVS.Core.Data.Choice.DEq where

open import UVMVS.Core.Init
open import UVMVS.Core.Classes

module _ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : Set ℓ₂} {{_ : has[≡?] A}} {{_ : has[≡?] B}} where

  _≡?⸢∨⸣_ : ∀ (x y : A ∨ B) → ≡!
  ʟ x ≡?⸢∨⸣ ʟ y = x ≡? y
  ʟ x ≡?⸢∨⸣ ʀ y = [≢]
  ʀ x ≡?⸢∨⸣ ʟ y = [≢]
  ʀ x ≡?⸢∨⸣ ʀ y = x ≡? y

  instance
    has[≡?][∨] : has[≡?] (A ∨ B)
    has[≡?][∨] = record { _≡?_ = _≡?⸢∨⸣_ }
  
  module _ {{_ : cor[≡?] A}} {{_ : cor[≡?] B}} where
    abstract
      _≡⋆⸢∨⸣_ : ∀ (x y : A ∨ B) → x ≡⋆! y
      ʟ x ≡⋆⸢∨⸣ ʟ y with x ≡⋆ y
      … | [≡] ε = [≡] (change[ (λ X → ʟ X ≡ ʟ y) ] ε ↯)
      … | [≢] ε = [≢] ⟨ (λ where ↯ → contra[≡] ε ↯) ⟩
      ʟ x ≡⋆⸢∨⸣ ʀ y = [≢] ⟨ (λ ()) ⟩
      ʀ x ≡⋆⸢∨⸣ ʟ y = [≢] ⟨ (λ ()) ⟩
      ʀ x ≡⋆⸢∨⸣ ʀ y with x ≡⋆ y
      … | [≡] ε = [≡] (change[ (λ X → ʀ X ≡ ʀ y) ] ε ↯)
      … | [≢] ε = [≢] ⟨ (λ where ↯ → contra[≡] ε ↯) ⟩
      
      _≡∿⸢∨⸣_ : ∀ (x y : A ∨ B) → x ≡∿! y [ x ≡?⸢∨⸣ y , x ≡⋆⸢∨⸣ y ]
      ʟ x ≡∿⸢∨⸣ ʟ y with x ≡? y | x ≡⋆ y | x ≡∿ y
      … | [≡] | [≡] ε | [≡] = [≡]
      … | [≢] | [≢] ε | [≢] = [≢]
      ʟ x ≡∿⸢∨⸣ ʀ y = [≢]
      ʀ x ≡∿⸢∨⸣ ʟ y = [≢]
      ʀ x ≡∿⸢∨⸣ ʀ y with x ≡? y | x ≡⋆ y | x ≡∿ y
      … | [≡] | [≡] ε | [≡] = [≡]
      … | [≢] | [≢] ε | [≢] = [≢]

    instance
      cor[≡?][∨] : cor[≡?] (A ∨ B)
      cor[≡?][∨] = record { _≡⋆_ = _≡⋆⸢∨⸣_ ; _≡∿_ = _≡∿⸢∨⸣_ }

    _ = ⌁ AT cor[≡?] (A ∨ B)
  _ = ⌁ AT has[≡?] (A ∨ B)
