module _ {ℓ ℓᴿ} {A : Set ℓ} {{_ : has[<][ ℓᴿ ] A}} where

  data _<ⱽ_ : ∀ {N} (xs ys : ⟬ A ⟭[ N ]) → Set (ℓ ⩛ ℓᴿ) where
    [<] : ∀ {N} {x y} {xs ys : ⟬ A ⟭[ N ]} → x < y → (x ∷ xs) <ⱽ (y ∷ ys)
    [≡] : ∀ {N} {x y} {xs ys : ⟬ A ⟭[ N ]} → x ≡ y → xs <ⱽ ys → (x ∷ xs) <ⱽ (y ∷ ys)

  instance
    has[<][vec] : ∀ {N} → has[<][ ℓ ⩛ ℓᴿ ] ⟬ A ⟭[ N ]
    has[<][vec] = record { _<_ = _<ⱽ_ }

  abstract
    isoto[<ⱽ] : ∀ {N} {xs ys : ⟬ A ⟭[ N ]} → xs <ⱽ ys → ιⱽ xs < ιⱽ ys
    isoto[<ⱽ] {ᴢ} {[]} {[]} ()
    isoto[<ⱽ] {ꜱ N} {x ∷ xs} {y ∷ ys} ([<] ε) = [<] ε
    isoto[<ⱽ] {ꜱ N} {x ∷ xs} {y ∷ ys} ([≡] ↯ ε) = [≡] ↯ (isoto[<ⱽ] ε)

    isofr[<ⱽ] : ∀ {N} {xs ys : ⟬ A ⟭[ N ]} → ιⱽ xs < ιⱽ ys → xs <ⱽ ys
    isofr[<ⱽ] {ᴢ} {[]} {[]} ()
    isofr[<ⱽ] {ꜱ N} {x ∷ xs} {y ∷ ys} ([<] ε) = [<] ε
    isofr[<ⱽ] {ꜱ N} {x ∷ xs} {y ∷ ys} ([≡] ↯ ε) = [≡] ↯ (isofr[<ⱽ] ε)

module _ {ℓ} {A : Set ℓ} {{_ : has[⋚?] A}} where

  _⋚?ⱽ_ : ∀ {N} (xs ys : ⟬ A ⟭[ N ]) → ⋚!
  [] ⋚?ⱽ [] = [≡]
  (x ∷ xs) ⋚?ⱽ (y ∷ ys) with x ⋚? y
  … | [<] = [<]
  … | [>] = [>]
  … | [≡] = xs ⋚?ⱽ ys

  instance
    has[⋚?][vec] : ∀ {N} → has[⋚?] ⟬ A ⟭[ N ]
    has[⋚?][vec] = record { _⋚?_ = _⋚?ⱽ_ }

  abstract
    corr[⋚?ⱽ]<_,_> : ∀ {N} (xs ys : ⟬ A ⟭[ N ]) → (xs ⋚?ⱽ ys) ≡ (ιⱽ xs ⋚? ιⱽ ys)
    corr[⋚?ⱽ]< [] , [] > = ↯
    corr[⋚?ⱽ]< x ∷ xs , y ∷ ys > with x ⋚? y
    … | [<] = ↯
    … | [>] = ↯
    … | [≡] = corr[⋚?ⱽ]< xs , ys >

module _ {ℓ ℓᴿ} {A : Set ℓ} {{_ : has[<][ ℓᴿ ] A}} {{_ : cor[<] A}} {N : ℕ} where
  open derive[cor[<]] ⟬ A ⟭[ N ] ⟬ A ⟭ ιⱽ isoto[<ⱽ] isofr[<ⱽ]
  
  instance
    irreflexive[<ⱽ] = irreflexive[<]†
    asymmetric[<ⱽ] = asymmetric[<]†
    transitive[<ⱽ] = transitive[<]†
    cor[<][vec] = cor[<]†
  
  module _ {{_ : has[⋚?] A}} {{_ : cor[⋚?] A}} where
    open derive[cor[⋚?]] ⟬ A ⟭[ N ] ⟬ A ⟭ ιⱽ isoto[<ⱽ] isofr[<ⱽ] corr[⋚?ⱽ]<_,_>
  
    instance
      cor[⋚?][vec] = cor[⋚?]†
  
module _ {ℓ ℓᴿ} {A : Set ℓ} {{_ : has[<][ ℓᴿ ] A}} {N : ℕ} where
  module _ {{_ : cor[<] A}} where
    _ = ⌁ AT irreflexive    (_<ⱽ_ AT (⟬ A ⟭[ N ] → ⟬ A ⟭[ N ] → Set (ℓ ⩛ ℓᴿ)))
    _ = ⌁ AT asymmetric     (_<ⱽ_ AT (⟬ A ⟭[ N ] → ⟬ A ⟭[ N ] → Set (ℓ ⩛ ℓᴿ)))
    _ = ⌁ AT transitive     (_<ⱽ_ AT (⟬ A ⟭[ N ] → ⟬ A ⟭[ N ] → Set (ℓ ⩛ ℓᴿ)))
    _ = ⌁ AT cor[<]         ⟬ A ⟭[ N ]
  _ = ⌁ AT has[<][ ℓ ⩛ ℓᴿ ] ⟬ A ⟭[ N ]

module _ {ℓ} {A : Set ℓ} {{_ : has[⋚?] A}} {N : ℕ} where
  module _ {ℓᴿ} {{_ : has[<][ ℓᴿ ] A}} {{_ : cor[<] A}} {{_ : cor[⋚?] A}} where
    _ = ⌁ AT cor[⋚?]  ⟬ A ⟭[ N ]
  _ = ⌁ AT has[⋚?]    ⟬ A ⟭[ N ]
