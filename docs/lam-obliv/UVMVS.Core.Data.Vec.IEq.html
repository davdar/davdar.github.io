module _ {ℓ ℓᴿ} {A : Set ℓ} {{_ : has[≍][ ℓᴿ ] A}} where
  data _≍ⱽ_ : ∀ {N} (xs ys : ⟬ A ⟭[ N ]) → Set ℓᴿ where
    [] : [] ≍ⱽ []
    _∷_ : ∀ {N} {x y : A} {xs ys : ⟬ A ⟭[ N ]} (ε : x ≍ y) (εs : xs ≍ⱽ ys) → (x ∷ xs) ≍ⱽ (y ∷ ys)
  
  data _≭ⱽ_ : ∀ {N} (xs ys : ⟬ A ⟭[ N ]) → Set ℓᴿ where
    _∷• : ∀ {N} {x y} {xs ys : ⟬ A ⟭[ N ]} (ε : x ≭ y) → (x ∷ xs) ≭ⱽ (y ∷ ys)
    •∷_ : ∀ {N} {x y} {xs ys : ⟬ A ⟭[ N ]} (εs : xs ≭ⱽ ys) → (x ∷ xs) ≭ⱽ (y ∷ ys)
  
  instance
    has[≍][vec] : ∀ {N} → has[≍][ ℓᴿ ] ⟬ A ⟭[ N ]
    has[≍][vec] = record { _≍_ = _≍ⱽ_ ; _≭_ = _≭ⱽ_ }

  abstract
    isoto[≍ⱽ] : ∀ {N} {xs ys : ⟬ A ⟭[ N ]} → xs ≍ⱽ ys → ιⱽ xs ≍ ιⱽ ys
    isoto[≍ⱽ] [] = []
    isoto[≍ⱽ] (ε ∷ εs) = ε ∷ isoto[≍ⱽ] εs
    
    isofr[≍ⱽ] : ∀ {N} {xs ys : ⟬ A ⟭[ N ]} → ιⱽ xs ≍ ιⱽ ys → xs ≍ⱽ ys
    isofr[≍ⱽ] {ᴢ} {[]} {[]} [] = []
    isofr[≍ⱽ] {ꜱ N} {x ∷ xs} {y ∷ ys} (ε ∷ εs) = ε ∷ isofr[≍ⱽ] εs
    
    isoto[≭ⱽ] : ∀ {N} {xs ys : ⟬ A ⟭[ N ]} → xs ≭ⱽ ys → ιⱽ xs ≭ ιⱽ ys
    isoto[≭ⱽ] (ε ∷•) = ε ∷•
    isoto[≭ⱽ] (•∷ ε) = •∷ isoto[≭ⱽ] ε
    
    isofr[≭ⱽ] : ∀ {N} {xs ys : ⟬ A ⟭[ N ]} → ιⱽ xs ≭ ιⱽ ys → xs ≭ⱽ ys
    isofr[≭ⱽ] {ᴢ} {[]} {[]} ()
    isofr[≭ⱽ] {ꜱ N} {x ∷ xs} {y ∷ ys} (ε ∷•) = ε ∷•
    isofr[≭ⱽ] {ꜱ N} {x ∷ xs} {y ∷ ys} (•∷ ε) = •∷ isofr[≭ⱽ] ε
  
  module _ {N} where
    module _ {{_ : cor[≍] A}} where
      open derive[cor[≍]] ⟬ A ⟭[ N ] ⟬ A ⟭ ιⱽ isoto[≍ⱽ] isofr[≍ⱽ] isoto[≭ⱽ] isofr[≭ⱽ]
    
      instance
        reflexive[≍ⱽ] = reflexive[≍]†
        symmetric[≍ⱽ] = symmetric[≍]†
        transitive[≍ⱽ] = transitive[≍]†
        irreflexive[≭ⱽ] = irreflexive[≭]†
        symmetric[≭ⱽ] = symmetric[≭]†
        cor[≍][vec] = cor[≍]†
  
      _ = ⌁ AT reflexive    (_≍ⱽ_ {N})
      _ = ⌁ AT symmetric    (_≍ⱽ_ {N})
      _ = ⌁ AT transitive   (_≍ⱽ_ {N})
      _ = ⌁ AT irreflexive  (_≭ⱽ_ {N})
      _ = ⌁ AT symmetric    (_≭ⱽ_ {N})
      _ = ⌁ AT cor[≍]       ⟬ A ⟭[ N ]   
    _ = ⌁ AT has[≍][ ℓᴿ ] ⟬ A ⟭[ N ]
