module _ {ℓ ℓᴿ} {A : Set ℓ} {{_ : has[<][ ℓᴿ ] A}} where

  data _<ᴸ_ : ∀ (xs ys : ⟬ A ⟭) → Set (ℓ ⩛ ℓᴿ) where
    [] : ∀ {x xs} → [] <ᴸ (x ∷ xs)
    [<] : ∀ {x y xs ys} → x < y → (x ∷ xs) <ᴸ (y ∷ ys)
    [≡] : ∀ {x y xs ys} → x ≡ y → xs <ᴸ ys → (x ∷ xs) <ᴸ (y ∷ ys)

  instance
    has[<][list] : has[<][ ℓ ⩛ ℓᴿ ] ⟬ A ⟭
    has[<][list] = record { _<_ = _<ᴸ_ }

  module _ {{_ : cor[<] A}} where

    abstract
      ¬xRx[<ᴸ] : ∀ {xs : ⟬ A ⟭} → ¬ (xs <ᴸ xs)
      ¬xRx[<ᴸ] ([<] ε) = ¬xRx ε
      ¬xRx[<ᴸ] ([≡] ε₁ ε₂) = ¬xRx[<ᴸ] ε₂
    
      _¬◇[<ᴸ]_ : ∀ {xs ys : ⟬ A ⟭} (ε₁ : xs <ᴸ ys) (ε₂ : ys <ᴸ xs) → 𝟘
      [] ¬◇[<ᴸ] ()
      [<] ε₁ ¬◇[<ᴸ] [<] ε₂ = ε₁ ¬◇ ε₂
      [<] ε₁ ¬◇[<ᴸ] [≡] ε₂₁ ε₂₂ rewrite ε₂₁ = ¬xRx ε₁
      [≡] ε₁₁ ε₁₂ ¬◇[<ᴸ] [<] ε₂ rewrite ε₁₁ = ¬xRx ε₂
      [≡] ε₁₁ ε₁₂ ¬◇[<ᴸ] [≡] ε₂₁ ε₂₂ = ε₁₂ ¬◇[<ᴸ] ε₂₂
    
      _⊚[<ᴸ]_ : ∀ {xs ys zs : ⟬ A ⟭} (ε₁ : ys <ᴸ zs) (ε₂ : xs <ᴸ ys) → xs <ᴸ zs
      [] ⊚[<ᴸ] ()
      [<] _ ⊚[<ᴸ] [] = []
      [≡] _ _ ⊚[<ᴸ] [] = []
      [<] ε₁ ⊚[<ᴸ] [<] ε₂ = [<] (ε₁ ⊚ ε₂)
      [<] ε₁ ⊚[<ᴸ] [≡] ε₂₁ ε₂₂ rewrite ε₂₁ = [<] ε₁
      [≡] ε₁₁ ε₁₂ ⊚[<ᴸ] [<] ε₂ rewrite ε₁₁ = [<] ε₂
      [≡] ε₁₁ ε₁₂ ⊚[<ᴸ] [≡] ε₂₁ ε₂₂ rewrite ε₁₁ | ε₂₁ = [≡] ↯ (ε₁₂ ⊚[<ᴸ] ε₂₂)

    instance
      irreflexive[<ᴸ] : irreflexive _<ᴸ_
      irreflexive[<ᴸ] = record { ¬xRx = ¬xRx[<ᴸ] }

      asymmetric[<ᴸ] : asymmetric _<ᴸ_
      asymmetric[<ᴸ] = record { _¬◇_ = _¬◇[<ᴸ]_ }

      transitive[<ᴸ] : transitive _<ᴸ_
      transitive[<ᴸ] = record { _⊚_ = _⊚[<ᴸ]_ }

      cor[<][list] : cor[<] ⟬ A ⟭
      cor[<][list] = record {}

module _ {ℓ} {A : Set ℓ} {{_ : has[⋚?] A}} where
  _⋚?ᴸ_ : ∀ (xs ys : ⟬ A ⟭) → ⋚!
  [] ⋚?ᴸ [] = [≡]
  [] ⋚?ᴸ (y ∷ ys) = [<]
  (x ∷ xs) ⋚?ᴸ [] = [>]
  (x ∷ xs) ⋚?ᴸ (y ∷ ys) with x ⋚? y
  … | [<] = [<]
  … | [>] = [>]
  … | [≡] = xs ⋚?ᴸ ys

  instance
    has[⋚?][list] : has[⋚?] ⟬ A ⟭
    has[⋚?][list] = record { _⋚?_ = _⋚?ᴸ_ }

  module _ {ℓᴿ} {{_ : has[<][ ℓᴿ ] A}} {{_ : cor[<] A}} {{_ : cor[⋚?] A}} where
    abstract
      _⋚⋆ᴸ_ : (xs ys : ⟬ A ⟭) → xs ⋚⋆! ys
      [] ⋚⋆ᴸ [] = [≡] ↯
      [] ⋚⋆ᴸ (y ∷ ys) = [<] []
      (x ∷ xs) ⋚⋆ᴸ [] = [>] []
      (x ∷ xs) ⋚⋆ᴸ (y ∷ ys) with x ⋚⋆ y
      … | [<] ε = [<] ([<] ε)
      … | [>] ε = [>] ([<] ε)
      … | [≡] ε with xs ⋚⋆ᴸ ys
      … | [<] ε′ = [<] ([≡] ε ε′)
      … | [>] ε′ = [>] ([≡] (◇ ε) ε′)
      … | [≡] ε′ = [≡] (change2[ (λ X XS → X ∷ XS ≡ y ∷ ys) ] ε ε′ ↯)
    
      _⋚∿ᴸ_ : ∀ (xs ys : ⟬ A ⟭)→ xs ⋚∿! ys [ xs ⋚?ᴸ ys , xs ⋚⋆ᴸ ys ]
      [] ⋚∿ᴸ [] = [≡]
      [] ⋚∿ᴸ (y ∷ ys) = [<]
      (x ∷ xs) ⋚∿ᴸ [] = [>]
      (x ∷ xs) ⋚∿ᴸ (y ∷ ys) with x ⋚? y | x ⋚⋆ y | x ⋚∿ y
      … | [<] | [<] ε | [<] = [<]
      … | [>] | [>] ε | [>] = [>]
      … | [≡] | [≡] ε | [≡] with xs ⋚?ᴸ ys | xs ⋚⋆ᴸ ys | xs ⋚∿ᴸ ys
      … | [<] | [<] ε′ | [<] = [<]
      … | [≡] | [≡] ε′ | [≡] = [≡]
      … | [>] | [>] ε′ | [>] = [>]
    
    instance
      cor[⋚?][list] : cor[⋚?] ⟬ A ⟭
      cor[⋚?][list] = record { _⋚⋆_ = _⋚⋆ᴸ_ ; _⋚∿_ = _⋚∿ᴸ_ }

module _ {ℓ ℓᴿ} {A : Set ℓ} {{_ : has[<][ ℓᴿ ] A}} where

  module _ {{_ : cor[<] A}} where

    _ = ⌁ AT irreflexive (_<ᴸ_ AT (⟬ A ⟭ → ⟬ A ⟭ → Set _))
    _ = ⌁ AT asymmetric  (_<ᴸ_ AT (⟬ A ⟭ → ⟬ A ⟭ → Set _))
    _ = ⌁ AT transitive  (_<ᴸ_ AT (⟬ A ⟭ → ⟬ A ⟭ → Set _))
    _ = ⌁ AT cor[<]      ⟬ A ⟭

  _ = ⌁ AT has[<][ _ ] ⟬ A ⟭

module _ {ℓ} {A : Set ℓ} {{_ : has[⋚?] A}} where
  module _ {ℓᴿ} {{_ : has[<][ ℓᴿ ] A}} {{_ : cor[<] A}}  {{_ : cor[⋚?] A}} where
    _ = ⌁ AT cor[⋚?] ⟬ A ⟭
  _ = ⌁ AT has[⋚?] ⟬ A ⟭
