syntax 𝐼[] A G = 𝐼[ G ] A
data 𝐼[] {ℓ} (A : Set ℓ) : ∀ (G : ℕ) → Set ℓ where
  ⟨_⟩ : ∀ (x : A) → 𝐼[ ᴢ ] A
  ⟨_,_⟩ : ∀ {G} (𝓍 : 𝐼[ G ] A) (𝓎 : 𝐼[ G ] A) → 𝐼[ ꜱ G ] A

module _ {ℓ} {A : Set ℓ} where
  [_]⌊_⌋[𝐼] : ∀ (G : ℕ) (x : A) → 𝐼[ G ] A
  [ ᴢ ]⌊ x ⌋[𝐼] = ⟨ x ⟩
  [ ꜱ G ]⌊ x ⌋[𝐼] = ⟨ [ G ]⌊ x ⌋[𝐼] , [ G ]⌊ x ⌋[𝐼] ⟩

  ⌊_⌋[𝐼] : ∀ {G : ℕ} (x : A) → 𝐼[ G ] A
  ⌊ x ⌋[𝐼] = [ _ ]⌊ x ⌋[𝐼]

  ∣_∣[𝐼] : ∀ {G : ℕ} (𝓍 : 𝐼[ G ] A) → ℕ
  ∣_∣[𝐼] {G} _ = G

  _⇧[𝐼]_ : ∀ {G} (𝓍 : 𝐼[ G ] A) (G′ : ℕ) → 𝐼[ G′ + G ] A
  𝓍 ⇧[𝐼] ᴢ = 𝓍
  𝓍 ⇧[𝐼] ꜱ G′ = ⟨ 𝓍 ⇧[𝐼] G′ , 𝓍 ⇧[𝐼] G′ ⟩

module _ {ℓ} {A : Set ℓ} {{_ : has[≡?] A}} where
  _⩮_ : ∀ {G} (𝓍 : 𝐼[ G ] A) (x : A) → 𝐼[ G ] 𝔹
  ⟨ x′ ⟩ ⩮ x with x′ ≡? x
  … | [≡] = ⟨ ɪ ⟩
  … | [≢] = ⟨ ᴏ ⟩
  ⟨ 𝓍₁ , 𝓍₂ ⟩ ⩮ x = ⟨ 𝓍₁ ⩮ x , 𝓍₂ ⩮ x ⟩

module _ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : Set ℓ₂} where
  map[𝐼] : ∀ {G} → (A → B) → 𝐼[ G ] A → 𝐼[ G ] B
  map[𝐼] f ⟨ x ⟩ = ⟨ f x ⟩
  map[𝐼] f ⟨ xs₁ , xs₂ ⟩ = ⟨ map[𝐼] f xs₁ , map[𝐼] f xs₂ ⟩

  _⊡[𝐼]_ : ∀ {G} (f : 𝐼[ G ] (A → B)) (𝓍 : 𝐼[ G ] A) → 𝐼[ G ] B
  ⟨ f ⟩ ⊡[𝐼] ⟨ x ⟩ = ⟨ f x ⟩
  ⟨ f₁ , f₂ ⟩ ⊡[𝐼] ⟨ 𝓍₁ , 𝓍₂ ⟩ = ⟨ f₁ ⊡[𝐼] 𝓍₁ , f₂ ⊡[𝐼] 𝓍₂ ⟩

module _ {ℓ} where
  instance
    has[map][𝐼] : ∀ {G} → has[map] (λ (A : Set ℓ) → 𝐼[ G ] A)
    has[map][𝐼] = record { map = map[𝐼] }

    has[⊡][𝐼] : ∀ {G} → has[⊡] (λ (A : Set ℓ) → 𝐼[ G ] A)
    has[⊡][𝐼] = record { pure = ⌊_⌋[𝐼] ; _⊡_ = _⊡[𝐼]_ }

𝔠[_] : ∀ {G} → 𝐼[ G ] 𝔹 → ℕ
𝔠[ ⟨ ᴏ ⟩ ] = 0
𝔠[ ⟨ ɪ ⟩ ] = 1
𝔠[ ⟨ 𝓍₁ , 𝓍₂ ⟩ ] = 𝔠[ 𝓍₁ ] + 𝔠[ 𝓍₂ ]

𝔰[_] : ∀ (G : ℕ) → ℕ
𝔰[ ᴢ ] = 1
𝔰[ ꜱ G ] = 𝔰[ G ] + 𝔰[ G ]

𝔰≢0<_> : ∀ (G : ℕ) → 𝔰[ G ] ≢ 0
𝔰≢0< ᴢ > = ⟨ (λ ()) ⟩
𝔰≢0< ꜱ G > = ¬lzero[+]< 𝔰[ G ] , 𝔰[ G ] > 𝔰≢0< G >

𝔭[_] : ∀ {G} → 𝐼[ G ] 𝔹 → ℚ⁺
𝔭[ 𝒷 ] = [ 𝔠[ 𝒷 ] / 𝔰[ ∣ 𝒷 ∣[𝐼] ] ː 𝔰≢0< ∣ 𝒷 ∣[𝐼] > ]

𝔭[_¦_ː_] : ∀ {G} (𝒷₁ 𝒷₂ : 𝐼[ G ] 𝔹) (ε : 𝔠[ 𝒷₂ ] ≢ 0) → ℚ⁺
𝔭[ 𝒷₁ ¦ 𝒷₂ ː ε ] = [ 𝔠[ 𝒷₁ ᚜ _⊓_ ᚛ 𝒷₂ ] / 𝔠[ 𝒷₂ ] ː ε ]

𝔠[⇧]≢0<_,_> : ∀ {G} (Φ : 𝐼[ G ] 𝔹) (G′ : ℕ) → 𝔠[ Φ ] ≢ 0 → 𝔠[ Φ ⇧[𝐼] G′ ] ≢ 0
𝔠[⇧]≢0< Φ , ᴢ > ε = ε
𝔠[⇧]≢0< Φ , ꜱ G′ > ε = ¬lzero[+]< 𝔠[ Φ ⇧[𝐼] G′ ] , 𝔠[ Φ ⇧[𝐼] G′ ] > (𝔠[⇧]≢0< Φ , G′ > ε)

module _ {ℓ} {A : Set ℓ} {{_ : has[≡?] A}} where
  data 𝔲 {G} (𝓍 : 𝐼[ G ] A) : Set ℓ where
    𝔘 : ∀ (q : ℚ⁺)
      → (∀ {x} → 𝔭[ 𝓍 ⩮ x ] ≡ q)
      → 𝔲 𝓍
  data 𝔲[_¦_ː_] {G} (𝓍 : 𝐼[ G ] A) (Φ : 𝐼[ G ] 𝔹) (E : 𝔠[ Φ ] ≢ 0) : Set ℓ where
    𝔘 : ∀ (q : ℚ⁺)
      → (∀ {x} → 𝔭[ 𝓍 ⩮ x ¦ Φ ː E ] ≡ q)
      → 𝔲[ 𝓍 ¦ Φ ː E ]
postulate
  _⇧[𝔲¦]_ :
    ∀ {ℓ} {A : Set ℓ} {{_ : has[≡?] A}} {G} {𝓍 : 𝐼[ G ] A} {Φ : 𝐼[ G ] 𝔹} {E : 𝔠[ Φ ] ≢ 0}
    → 𝔲[ 𝓍 ¦ Φ ː E ]
    → (N : ℕ)
    → 𝔲[ 𝓍 ⇧[𝐼] N ¦ Φ ⇧[𝐼] N ː 𝔠[⇧]≢0< Φ , N > E ]

coin-flip : ∀ G → 𝐼[ ꜱ G ] 𝔹
coin-flip G = ⟨ ⌊ ᴏ ⌋[𝐼] , ⌊ ɪ ⌋[𝐼] ⟩

postulate
  𝔲-coin-flip : ∀ {G} (Φ : 𝐼[ G ] 𝔹) (E : 𝔠[ Φ ] ≢ 0) → 𝔲[ coin-flip G ¦ Φ ⇧[𝐼] 1 ː 𝔠[⇧]≢0< Φ , 1 > E ]

postulate
  𝔠[Φ⊓𝒷] : ∀ {G} (Φ : 𝐼[ G ] 𝔹) (𝒷 : 𝐼[ G ] 𝔹) (E : 𝔠[ Φ ] ≢ 0) → 𝔠[ Φ ᚜ _⊓_ ᚛ 𝒷 ] ≢ 0
