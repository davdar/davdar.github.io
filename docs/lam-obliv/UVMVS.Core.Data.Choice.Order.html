module _ {ℓ₁ ℓ₂ ℓᴿ₁ ℓᴿ₂} {A : Set ℓ₁} {B : Set ℓ₂} {{_ : has[<][ ℓᴿ₁ ] A}} {{_ : has[<][ ℓᴿ₂ ] B}} where

  data _<⸢∨⸣_ : (x y : A ∨ B) → Set (ℓᴿ₁ ⩛ ℓᴿ₂) where
    ʟ : ∀ {x y} (ε : x < y) → ʟ x <⸢∨⸣ ʟ y
    ʀ : ∀ {x y} (ε : x < y) → ʀ x <⸢∨⸣ ʀ y
    ʟʀ : ∀ {x y} → ʟ x <⸢∨⸣ ʀ y
 
  instance
    has[<][∨] : has[<][ ℓᴿ₁ ⩛ ℓᴿ₂ ] (A ∨ B)
    has[<][∨] = record { _<_ = _<⸢∨⸣_ }

  module _ {{_ : cor[<] A}} {{_ : cor[<] B}} where

    abstract
      ¬xRx[<⸢∨⸣] : ∀ {x : A ∨ B} → ¬ (x <⸢∨⸣ x)
      ¬xRx[<⸢∨⸣] (ʟ ε) = ¬xRx ε
      ¬xRx[<⸢∨⸣] (ʀ ε) = ¬xRx ε
      
      _¬◇[<⸢∨⸣]_ : ∀ {x y : A ∨ B} (ε₁ : x <⸢∨⸣ y) (ε₂ : y <⸢∨⸣ x) → 𝟘
      ʟ ε₁ ¬◇[<⸢∨⸣] ʟ ε₂ = ε₁ ¬◇ ε₂
      ʀ ε₁ ¬◇[<⸢∨⸣] ʀ ε₂ = ε₁ ¬◇ ε₂
      ʟʀ ¬◇[<⸢∨⸣] ()

      _⊚[<⸢∨⸣]_ : ∀ {x y z : A ∨ B} (ε₁ : y <⸢∨⸣ z) (ε₂ : x <⸢∨⸣ y) → x <⸢∨⸣ z
      ʟ ε₁ ⊚[<⸢∨⸣] ʟ ε₂ = ʟ (ε₁ ⊚ ε₂)
      ʀ ε₁ ⊚[<⸢∨⸣] ʀ ε₂ = ʀ (ε₁ ⊚ ε₂)
      ʀ ε₁ ⊚[<⸢∨⸣] ʟʀ = ʟʀ
      ʟʀ ⊚[<⸢∨⸣] ʟ ε₂ = ʟʀ

    instance
      irreflexive[<⸢∨⸣] : irreflexive _<⸢∨⸣_
      irreflexive[<⸢∨⸣] = record { ¬xRx = ¬xRx[<⸢∨⸣] }

      asymmetric[<⸢∨⸣] : asymmetric _<⸢∨⸣_
      asymmetric[<⸢∨⸣] = record { _¬◇_ = _¬◇[<⸢∨⸣]_ }

      transitive[<⸢∨⸣] : transitive _<⸢∨⸣_
      transitive[<⸢∨⸣] = record { _⊚_ = _⊚[<⸢∨⸣]_ }

      cor[<][∨] : cor[<] (A ∨ B)
      cor[<][∨] = record {}
      
module _ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : Set ℓ₂} {{_ : has[⋚?] A}} {{_ : has[⋚?] B}} where

  _⋚?⸢∨⸣_ : A ∨ B → A ∨ B → ⋚!
  ʟ x ⋚?⸢∨⸣ ʟ y = x ⋚? y
  ʟ x ⋚?⸢∨⸣ ʀ y = [<]
  ʀ x ⋚?⸢∨⸣ ʟ y = [>]
  ʀ x ⋚?⸢∨⸣ ʀ y = x ⋚? y

  instance
    has[⋚][∨] : has[⋚?] (A ∨ B)
    has[⋚][∨] = record { _⋚?_ = _⋚?⸢∨⸣_ }

  module _ {ℓᴿ₁ ℓᴿ₂} {{_ : has[<][ ℓᴿ₁ ] A}} {{_ : cor[<] A}} {{_ : has[<][ ℓᴿ₂ ] B}} {{_ : cor[<] B}} {{_ : cor[⋚?] A}} {{_ : cor[⋚?] B}} where
    abstract
      _⋚⋆⸢∨⸣_ : ∀ (x y : A ∨ B) → x ⋚⋆! y
      ʟ x ⋚⋆⸢∨⸣ ʟ y with x ⋚⋆ y
      … | [<] ε = [<] (ʟ ε)
      … | [≡] ε = [≡] (change[ (λ X → ʟ X ≡ ʟ y) ] ε ↯)
      … | [>] ε = [>] (ʟ ε)
      ʟ x ⋚⋆⸢∨⸣ ʀ y = [<] ʟʀ
      ʀ x ⋚⋆⸢∨⸣ ʟ y = [>] ʟʀ
      ʀ x ⋚⋆⸢∨⸣ ʀ y with x ⋚⋆ y
      … | [<] ε = [<] (ʀ ε)
      … | [≡] ε = [≡] (change[ (λ X → ʀ X ≡ ʀ y) ] ε ↯)
      … | [>] ε = [>] (ʀ ε)
      
      _⋚∿⸢∨⸣_ : ∀ (x y : A ∨ B) → x ⋚∿! y [ x ⋚?⸢∨⸣ y , x ⋚⋆⸢∨⸣ y ]
      ʟ x ⋚∿⸢∨⸣ ʟ y with x ⋚? y | x ⋚⋆ y | x ⋚∿ y
      … | [<] | [<] ε | [<] = [<]
      … | [≡] | [≡] ε | [≡] = [≡]
      … | [>] | [>] ε | [>] = [>]
      ʟ x ⋚∿⸢∨⸣ ʀ y = [<]
      ʀ x ⋚∿⸢∨⸣ ʟ y = [>]
      ʀ x ⋚∿⸢∨⸣ ʀ y with x ⋚? y | x ⋚⋆ y | x ⋚∿ y 
      … | [<] | [<] ε | [<] = [<]
      … | [≡] | [≡] ε | [≡] = [≡]
      … | [>] | [>] ε | [>] = [>]

    instance
      cor[⋚?][∨] : cor[⋚?] (A ∨ B)
      cor[⋚?][∨] = record { _⋚⋆_ = _⋚⋆⸢∨⸣_ ; _⋚∿_ = _⋚∿⸢∨⸣_ }
      
module _ {ℓ₁ ℓ₂ ℓᴿ₁ ℓᴿ₂} {A : Set ℓ₁} {B : Set ℓ₂} {{_ : has[<][ ℓᴿ₁ ] A}} {{_ : has[<][ ℓᴿ₂ ] B}} where

  module _ {{_ : cor[<] A}} {{_ : cor[<] B}} where

    _ = ⌁ AT irreflexive (_<⸢∨⸣_ AT (A ∨ B → A ∨ B → Set _))
    _ = ⌁ AT asymmetric  (_<⸢∨⸣_ AT (A ∨ B → A ∨ B → Set _))
    _ = ⌁ AT transitive  (_<⸢∨⸣_ AT (A ∨ B → A ∨ B → Set _))
    _ = ⌁ AT cor[<]      (A ∨ B)

  _ = ⌁ AT has[<][ _ ] (A ∨ B)

module _ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : Set ℓ₂} {{_ : has[⋚?] A}} {{_ : has[⋚?] B}} where
  module _ {ℓᴿ₁ ℓᴿ₂} {{_ : has[<][ ℓᴿ₁ ] A}} {{_ : cor[<] A}} {{_ : has[<][ ℓᴿ₂ ] B}} {{_ : cor[<] B}} {{_ : cor[⋚?] A}} {{_ : cor[⋚?] B}} where
    _ = ⌁ AT cor[⋚?] (A ∨ B)
  _ = ⌁ AT has[⋚?] (A ∨ B)
