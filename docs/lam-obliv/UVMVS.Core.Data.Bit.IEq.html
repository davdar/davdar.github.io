module UVMVS.Core.Data.Bit.IEq where

open import UVMVS.Core.Init
open import UVMVS.Core.Classes
open import UVMVS.Core.Data.Unit

open import UVMVS.Core.Data.Bit.Injection

data _≍ᴮ_ : (x y : 𝔹) → Set where
  ᴏ : ᴏ ≍ᴮ ᴏ
  ɪ : ɪ ≍ᴮ ɪ

data _≭ᴮ_ : (x y : 𝔹) → Set where
  ᴏɪ : ᴏ ≭ᴮ ɪ
  ɪᴏ : ɪ ≭ᴮ ᴏ

instance
  has[≍][𝔹] : has[≍][ L0 ] 𝔹
  has[≍][𝔹] = record { _≍_ = _≍ᴮ_ ; _≭_ = _≭ᴮ_ }

abstract
  isoto[≍ᴮ] : ∀ {x y} → x ≍ᴮ y → ιᴮ x ≍ ιᴮ y
  isoto[≍ᴮ] ᴏ = ʟ •
  isoto[≍ᴮ] ɪ = ʀ •
  
  isofr[≍ᴮ] : ∀ {x y} → ιᴮ x ≍ ιᴮ y → x ≍ᴮ y
  isofr[≍ᴮ] {ᴏ} {ᴏ} (ʟ •) = ᴏ
  isofr[≍ᴮ] {ᴏ} {ɪ} ()
  isofr[≍ᴮ] {ɪ} {ᴏ} ()
  isofr[≍ᴮ] {ɪ} {ɪ} (ʀ •) = ɪ
  
  isoto[≭ᴮ] : ∀ {x y} → x ≭ᴮ y → ιᴮ x ≭ ιᴮ y
  isoto[≭ᴮ] ᴏɪ = ʟʀ
  isoto[≭ᴮ] ɪᴏ = ʀʟ
  
  isofr[≭ᴮ] : ∀ {x y} → ιᴮ x ≭ ιᴮ y → x ≭ᴮ y
  isofr[≭ᴮ] {ᴏ} {ᴏ} (ʟ ())
  isofr[≭ᴮ] {ᴏ} {ɪ} ʟʀ = ᴏɪ
  isofr[≭ᴮ] {ɪ} {ᴏ} ʀʟ = ɪᴏ
  isofr[≭ᴮ] {ɪ} {ɪ} (ʀ ())

open derive[cor[≍]] 𝔹 (𝟙 ∨ 𝟙) ιᴮ isoto[≍ᴮ] isofr[≍ᴮ] isoto[≭ᴮ] isofr[≭ᴮ]

instance
  reflexive[≍ᴮ] = reflexive[≍]†
  symmetric[≍ᴮ] = symmetric[≍]†
  transitive[≍ᴮ] = transitive[≍]†
  irreflexive[≭ᴮ] = irreflexive[≭]†
  symmetric[≭ᴮ] = symmetric[≭]†
  cor[≍][𝔹] = cor[≍]†

_ = ⌁ AT reflexive    _≍ᴮ_
_ = ⌁ AT symmetric    _≍ᴮ_
_ = ⌁ AT transitive   _≍ᴮ_
_ = ⌁ AT irreflexive  _≭ᴮ_
_ = ⌁ AT symmetric    _≭ᴮ_
_ = ⌁ AT cor[≍] 𝔹
_ = ⌁ AT has[≍][ L0 ] 𝔹
