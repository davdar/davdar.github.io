_≡?ᴵ_ : ∀ {N} (n m : idx N) → ≡!
ᴢ ≡?ᴵ ᴢ = [≡]
ᴢ ≡?ᴵ ꜱ m = [≢]
ꜱ n ≡?ᴵ ᴢ = [≢]
ꜱ n ≡?ᴵ ꜱ m = n ≡?ᴵ m

instance
  has[≡?][idx] : ∀ {N} → has[≡?] (idx N)
  has[≡?][idx] = record { _≡?_ = _≡?ᴵ_ }

abstract
  corr[≡?ᴵ]<_,_> : ∀ {N} (x y : idx N) → (x ≡? y) ≡ (ιᴵ x ≡? ιᴵ y)
  corr[≡?ᴵ]< ᴢ , ᴢ > = ↯
  corr[≡?ᴵ]< ᴢ , ꜱ m > = ↯
  corr[≡?ᴵ]< ꜱ n , ᴢ > = ↯
  corr[≡?ᴵ]< ꜱ n , ꜱ m > = corr[≡?ᴵ]< n , m >

module _ {N : ℕ} where
  open derive[cor[≡?]] (idx N) ℕ ιᴵ corr[≡?ᴵ]<_,_>
  
  instance
    cor[≡?][idx] = cor[≡?]†

  _ = ⌁ AT cor[≡?] (idx N)
  _ = ⌁ AT has[≡?] (idx N)
