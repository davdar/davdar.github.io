-------------------
-- Partial Order --
-------------------

data ⊑! : Set where [⊑] [⋢] : ⊑!

record has[⊑][_] {ℓ} ℓᴿ (A : Set ℓ) : Set (ℓ ⩛ ↟ ℓᴿ) where
  infix 10 _⊑_
  field
    _⊑_ : ∀ (x y : A) → Set ℓᴿ

  data _⊑⋆!_ (x y : A) : Set (ℓ ⩛ ℓᴿ) where
    [⊑] : (ε : x ⊑ y)     → x ⊑⋆! y
    [⋢] : (ε : ¬ (x ⊑ y)) → x ⊑⋆! y
  
  data _⊑∿!_[_,_] (x y : A) : ∀ (ε₁ : ⊑!) (ε₂ : x ⊑⋆! y) → Set where
    [⊑] : ∀ {ε : x ⊑ y}     → x ⊑∿! y [ [⊑] , [⊑] ε ]
    [⋢] : ∀ {ε : ¬ (x ⊑ y)} → x ⊑∿! y [ [⋢] , [⋢] ε ]
open has[⊑][_] {{…}} public

record cor[⊑] {ℓ ℓᴿ} (A : Set ℓ) {{_ : has[⊑][ ℓᴿ ] A}} : Set (ℓ ⩛ ↟ ℓᴿ) where
  field
    {{reflexive[⊑]}} : reflexive (_⊑_ AT (A → A → Set ℓᴿ))
    {{antisymmetric[⊑]}} : antisymmetric (_⊑_ AT (A → A → Set ℓᴿ))
    {{transitive[⊑]}} : transitive (_⊑_ AT (A → A → Set ℓᴿ))
open cor[⊑] {{…}}

data ⧖! : Set where [≡] [⊏] [⊐] [▽] : ⧖!
  
record has[⊑?] {ℓ} (A : Set ℓ) : Set ℓ where
  infix 10 _⊑?_
  field
    _⊑?_ : A → A → ⊑!
  _⧖?_ : A → A → ⧖!
  x ⧖? y with x ⊑? y | y ⊑? x
  … | [⊑] | [⊑] = [≡]
  … | [⊑] | [⋢] = [⊏]
  … | [⋢] | [⊑] = [⊐]
  … | [⋢] | [⋢] = [▽]
open has[⊑?] {{…}} public

record cor[⊑?] {ℓ ℓᴿ} (A : Set ℓ) {{_ : has[⊑][ ℓᴿ ] A}} {{_ : cor[⊑] A}} {{_ : has[⊑?] A}} : Set (ℓ ⩛ ↟ ℓᴿ) where
  field
    _⊑⋆_ : ∀ (x y : A) → x ⊑⋆! y
    _⊑∿_ : ∀ (x y : A) → x ⊑∿! y [ x ⊑? y , x ⊑⋆ y ]
  is⊑[⊑]<_,_> : ∀ (x y : A) (ε : x ⊑ y) → (x ⊑? y) ≡ [⊑]
  is⊑[⊑]< x , y > ε with x ⊑? y | x ⊑⋆ y | x ⊑∿ y
  … | [⊑] | [⊑] ε′ | [⊑] = ↯
  … | [⋢] | [⋢] ε′ | [⋢] = contra ε′ ε
  is⋢[⊑]<_,_> : ∀ (x y : A) (ε : ¬ (x ⊑ y)) → (x ⊑? y) ≡ [⋢]
  is⋢[⊑]< x , y > ε with x ⊑? y | x ⊑⋆ y | x ⊑∿ y
  … | [⊑] | [⊑] ε′ | [⊑] = contra ε ε′
  … | [⋢] | [⋢] ε′ | [⋢] = ↯
  mk⊑[⊑]<_,_> : ∀ (x y : A) (ε : (x ⊑? y) ≡ [⊑]) → x ⊑ y
  mk⊑[⊑]< x , y > with x ⊑? y | x ⊑⋆ y | x ⊑∿ y
  … | [⊑] | [⊑] ε′ | [⊑] = λ _ → ε′
  … | [⋢] | [⋢] ε′ | [⋢] = λ ()
  mk⋢[⊑]<_,_> : ∀ (x y : A) (ε : (x ⊑? y) ≡ [⋢]) → ¬ (x ⊑ y)
  mk⋢[⊑]< x , y > with x ⊑? y | x ⊑⋆ y | x ⊑∿ y
  … | [⊑] | [⊑] ε′ | [⊑] = λ ()
  … | [⋢] | [⋢] ε′ | [⋢] = λ _ → ε′
open cor[⊑?] {{…}} public

module derive[cor[⊑]]
  {ℓ₁ ℓ₂ ℓᴿ₁ ℓᴿ₂}
  (A : Set ℓ₁) (B : Set ℓ₂) (ι : A → B) 
  {{_ : has[⊑][ ℓᴿ₁ ] A}}
  {{_ : has[⊑][ ℓᴿ₂ ] B}} {{_ : cor[⊑] B}}
  {{_ : injective ι}}
  (isoto[⊑] : ∀ {x y} → x ⊑ y → ι x ⊑ ι y)
  (isofr[⊑] : ∀ {x y} → ι x ⊑ ι y → x ⊑ y)
  where

  abstract
    xRx[⊑]†<_> : ∀ (x : A) → x ⊑ x
    xRx[⊑]†< x > = isofr[⊑] xRx< ι x >

    _⋈[⊑]†_ : ∀ {x y : A} → x ⊑ y → y ⊑ x → x ≡ y
    ε₁ ⋈[⊑]† ε₂ = inj< _ , _ > (isoto[⊑] ε₁ ⋈ isoto[⊑] ε₂)

    _⊚[⊑]†_ : ∀ {x y z : A} → y ⊑ z → x ⊑ y → x ⊑ z
    ε₁ ⊚[⊑]† ε₂ = isofr[⊑] (isoto[⊑] ε₁ ⊚ isoto[⊑] ε₂)

    reflexive[⊑]† : reflexive (_⊑_ AT (A → A → Set ℓᴿ₁))
    reflexive[⊑]† = record { xRx<_> = xRx[⊑]†<_> }
    antisymmetric[⊑]† : antisymmetric (_⊑_ AT (A → A → Set ℓᴿ₁))
    antisymmetric[⊑]† = record { _⋈_ = _⋈[⊑]†_ }
    transitive[⊑]† : transitive (_⊑_ AT (A → A → Set ℓᴿ₁))
    transitive[⊑]† = record { _⊚_ = _⊚[⊑]†_ }

  private
    instance
      _ = reflexive[⊑]†
      _ = antisymmetric[⊑]†
      _ = transitive[⊑]†

  cor[⊑]† : cor[⊑] A
  cor[⊑]† = record {}

  
module derive[cor[⊑?]]
  {ℓ₁ ℓ₂ ℓᴿ₁ ℓᴿ₂}
  (A : Set ℓ₁) (B : Set ℓ₂) (ι : A → B)
  {{_ : has[⊑][ ℓᴿ₁ ] A}} {{_ : cor[⊑] A}} {{_ : has[⊑?] A}}
  {{_ : has[⊑][ ℓᴿ₂ ] B}} {{_ : cor[⊑] B}} {{_ : has[⊑?] B}}  {{_ : cor[⊑?] B}}
  {{_ : injective ι}}
  (isoto[⊑] : ∀ {x y} → x ⊑ y → ι x ⊑ ι y)
  (isofr[⊑] : ∀ {x y} → ι x ⊑ ι y → x ⊑ y)
  (corr[⊑]<_,_> : ∀ x y → (x ⊑? y) ≡ (ι x ⊑? ι y))
  where

  abstract
    _⊑⋆†_ : ∀ (x y : A) → x ⊑⋆! y
    x ⊑⋆† y with ι x ⊑⋆ ι y
    … | [⊑] ε = [⊑] (isofr[⊑] ε)
    … | [⋢] ε = [⋢] (λ ε′ → ε (isoto[⊑] ε′))

    _⊑∿†_ : ∀ (x y : A) → x ⊑∿! y [ x ⊑? y , x ⊑⋆† y ]
    x ⊑∿† y with x ⊑? y | ι x ⊑? ι y | ι x ⊑⋆ ι y | corr[⊑]< x , y > | ι x ⊑∿ ι y
    … | [⊑] | [⊑] | [⊑] ε | ↯ | [⊑] = [⊑]
    … | [⋢] | [⋢] | [⋢] ε | ↯ | [⋢] = [⋢]

  cor[⊑?]† : cor[⊑?] A
  cor[⊑?]† = record { _⊑⋆_ = _⊑⋆†_ ; _⊑∿_ = _⊑∿†_ }

module synthesize[has[⊑]]
  {ℓ₁ ℓ₂} ℓᴿ
  (A : Set ℓ₁) (B : Set ℓ₂) (ι : A → B)
  {{_ : has[⊑][ ℓᴿ ] B}}
  {{_ : injective ι}}
  where
    
  data _⊑†_ (x y : A) : Set ℓᴿ where
    ⟨_⟩ : ι x ⊑ ι y → x ⊑† y

  has[⊑]† : has[⊑][ ℓᴿ ] A
  has[⊑]† = record { _⊑_ = _⊑†_}

module synthesize[cor[⊑]]
  {ℓ₁ ℓ₂} ℓᴿ
  (A : Set ℓ₁) (B : Set ℓ₂) (ι : A → B)
  {{_ : has[⊑][ ℓᴿ ] B}} {{_ : cor[⊑] B}}
  {{_ : injective ι}}
  where

  open synthesize[has[⊑]] ℓᴿ A B ι

  private
    instance _ = has[⊑]†

  abstract
    isoto[⊑]† : ∀ {x y : A} → x ⊑ y → ι x ⊑ ι y
    isoto[⊑]† ⟨ ε ⟩ = ε
  
    isofr[⊑]† : ∀ {x y : A} → ι x ⊑ ι y → x ⊑ y
    isofr[⊑]† ε = ⟨ ε ⟩

  open derive[cor[⊑]] A B ι isoto[⊑]† isofr[⊑]† public
  
module synthesize[has[⊑?]]
  {ℓ₁ ℓ₂}
  (A : Set ℓ₁) (B : Set ℓ₂) (ι : A → B) 
  {{_ : has[⊑?] B}} 
  {{_ : injective ι}}
  where
    
  _⊑?†_ : A → A → ⊑!
  x ⊑?† y = ι x ⊑? ι y

  has[⊑?]† : has[⊑?] A
  has[⊑?]† = record { _⊑?_ = _⊑?†_}

module synthesize[cor[⊑?]]
  {ℓ₁ ℓ₂} ℓᴿ
  (A : Set ℓ₁) (B : Set ℓ₂) (ι : A → B)
  {{_ : has[⊑][ ℓᴿ ] B}} {{_ : cor[⊑] B}} {{_ : has[⊑?] B}} {{_ : cor[⊑?] B}}
  {{_ : injective ι}}
  where

  open synthesize[has[⊑]] ℓᴿ A B ι
  open synthesize[cor[⊑]] ℓᴿ A B ι
  open synthesize[has[⊑?]] A B ι

  private
    instance
      _ = has[⊑]†
      _ = cor[⊑]†
      _ = has[⊑?]†

  abstract
    corr[⊑?]†<_,_> : ∀ (x y : A) → (x ⊑? y) ≡ (ι x ⊑? ι y)
    corr[⊑?]†< x , y > = ↯

  open derive[cor[⊑?]] A B ι isoto[⊑]† isofr[⊑]† corr[⊑?]†<_,_> public
