infixr 20 _∷_

data ⟬_⟭[_] {ℓ} (A : Set ℓ) : (N : ℕ) → Set ℓ where
  [] : ⟬ A ⟭[ ᴢ ]
  _∷_ : ∀ {n} (x : A) (xs : ⟬ A ⟭[ n ]) → ⟬ A ⟭[ ꜱ n ]

module _ {ℓ} {A : Set ℓ} where
  infixl 15 _⧻_
  infixl 27 _#[_] _#[_↦_]

  infix 10 ∈ⱽ
  syntax ∈ⱽ x i xs = ⟨ i ↦ x ⟩∈ xs
  data ∈ⱽ (x : A) : ∀ {N} (i : idx N) (xs : ⟬ A ⟭[ N ]) → Set where
    ᴢ : ∀ {N} {xs : ⟬ A ⟭[ N ]}
      → ⟨ ᴢ ↦ x ⟩∈ x ∷ xs
    ꜱ : ∀ {N} {i : idx N} {xs : ⟬ A ⟭[ N ]} {x′ : A}
      → ⟨ i ↦ x ⟩∈ xs
      → ⟨ ꜱ i ↦ x ⟩∈ x′ ∷ xs

  ∣_∣ⱽ : ∀ {N} (xs : ⟬ A ⟭[ N ]) → ℕ
  ∣_∣ⱽ {N} xs = N

  _#[_] : ∀ {N : ℕ} (xs : ⟬ A ⟭[ N ]) (i : idx N) → A
  (x ∷ xs) #[ ᴢ ] = x
  (x ∷ xs) #[ ꜱ i ] = xs #[ i ]
  
  _#[_↦_] : ∀ {N : ℕ} (xs : ⟬ A ⟭[ N ]) (i : idx N) (y : A) → ⟬ A ⟭[ N ]
  (x ∷ xs) #[ ᴢ ↦ y ] = y ∷ xs
  (x ∷ xs) #[ ꜱ i ↦ y ] = x ∷ xs #[ i ↦ y ]

  abstract
    reduc[lookup/∈]<_,_,_> : ∀ {N : ℕ} (i : idx N) (x : A) (xs : ⟬ A ⟭[ N ]) → ⟨ i ↦ x ⟩∈ xs → xs #[ i ] ≡ x
    reduc[lookup/∈]< ᴢ , x , x ∷ xs > ᴢ = ↯
    reduc[lookup/∈]< ꜱ i , x , x′ ∷ xs > (ꜱ ε) = reduc[lookup/∈]< i , x , xs > ε

    reduc[lookup/extend]<_,_,_> : ∀ {N : ℕ} (xs : ⟬ A ⟭[ N ]) (i : idx N) (x : A) → xs #[ i ↦ x ] #[ i ] ≡ x
    reduc[lookup/extend]< x′ ∷ xs , ᴢ , x > = ↯
    reduc[lookup/extend]< x′ ∷ xs , ꜱ i , x > rewrite reduc[lookup/extend]< xs , i , x > = ↯
  
  _⧻_ : ∀ {N₁ N₂} (xs : ⟬ A ⟭[ N₁ ]) (ys : ⟬ A ⟭[ N₂ ]) → ⟬ A ⟭[ N₁ + N₂ ]
  [] ⧻ ys = ys
  (x ∷ xs) ⧻ ys = x ∷ (xs ⧻ ys)

  foldrⱽ : ∀ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : Set ℓ₂} {N} → B → (A → B → B) → ⟬ A ⟭[ N ] → B
  foldrⱽ z f [] = z
  foldrⱽ z f (x ∷ xs) = f x (foldrⱽ z f xs)

module _ {ℓ} {A : Set ℓ} where
  𝐜<_> : ∀ N → A → ⟬ A ⟭[ N ]
  𝐜< ᴢ > x = []
  𝐜< ꜱ N > x = x ∷ 𝐜< N > x

  𝐜 : ∀ {N} → A → ⟬ A ⟭[ N ]
  𝐜 x = 𝐜< _ > x

  abstract
    dist[⧻/𝐜]<_,_,_> : ∀ N₁ N₂ (x : A) → 𝐜< N₁ > x ⧻ 𝐜< N₂ > x ≡ 𝐜< N₁ + N₂ > x
    dist[⧻/𝐜]< ᴢ , N₂ , x > = ↯
    dist[⧻/𝐜]< ꜱ N₁ , N₂ , x > rewrite dist[⧻/𝐜]< N₁ , N₂ , x > = ↯

-- has[map] and cor[map] --
---------------------------

mapⱽ : ∀ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : Set ℓ₂} {N} → (A → B) → ⟬ A ⟭[ N ] → ⟬ B ⟭[ N ]
mapⱽ f [] = []
mapⱽ f (x ∷ xs) = f x ∷ mapⱽ f xs

instance
  has[map][vec] : ∀ {ℓ} {N} → has[map] (λ (A : Set ℓ) → ⟬ A ⟭[ N ])
  has[map][vec] = record { map = mapⱽ }

abstract
  unit[mapⱽ]<_> : ∀ {ℓ} {A : Set ℓ} {N} (xs : ⟬ A ⟭[ N ]) → map id xs ≡ xs
  unit[mapⱽ]< [] > = ↯
  unit[mapⱽ]< x ∷ xs > rewrite unit[mapⱽ]< xs > = ↯
  
  homom[mapⱽ]<_,_,_> : ∀ {ℓ₁ ℓ₂ ℓ₃} {A : Set ℓ₁} {B : Set ℓ₂} {C : Set ℓ₃} {N} (g : B → C) (f : A → B) (xs : ⟬ A ⟭[ N ]) → mapⱽ (g ∘ f) xs ≡ mapⱽ g (mapⱽ f xs)
  homom[mapⱽ]< g , f , [] > = ↯
  homom[mapⱽ]< g , f , x ∷ xs > rewrite homom[mapⱽ]< g , f , xs > = ↯

instance
  cor[map][vec] : ∀ {ℓ} {N} → cor[map] (λ (A : Set ℓ) → ⟬ A ⟭[ N ])
  cor[map][vec] = record { unit[map]<_> = unit[mapⱽ]<_> ; homom[map]<_,_,_> = homom[mapⱽ]<_,_,_> }

abstract
  dist[mapⱽ/⧻]<_,_,_> : ∀ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : Set ℓ₂} {N₁ N₂} (f : A → B) (xs : ⟬ A ⟭[ N₁ ]) (ys : ⟬ A ⟭[ N₂ ]) → mapⱽ f (xs ⧻ ys) ≡ mapⱽ f xs ⧻ mapⱽ f ys
  dist[mapⱽ/⧻]<_,_,_> f [] ys = ↯
  dist[mapⱽ/⧻]<_,_,_> f (x ∷ xs) ys rewrite dist[mapⱽ/⧻]< f , xs , ys > = ↯

module _ {ℓ} {A : Set ℓ} {{_ : has[+] A}} where
  𝟎<_> : ∀ N → ⟬ A ⟭[ N ]
  𝟎< N > = 𝐜< N > zero

  𝟎 : ∀ {N} → ⟬ A ⟭[ N ]
  𝟎 = 𝟎< _ >

  _+ⱽ_ : ∀ {N} (xs : ⟬ A ⟭[ N ]) (ys : ⟬ A ⟭[ N ]) → ⟬ A ⟭[ N ]
  [] +ⱽ [] = []
  (x ∷ xs) +ⱽ (y ∷ ys) = (x + y) ∷ (xs +ⱽ ys) 
  instance
    has[+][vec] : ∀ {N} → has[+] (⟬ A ⟭[ N ])
    has[+][vec] = record { zero = 𝟎 ; _+_ = _+ⱽ_ }

  module _ {{_ : cor[+] A}} where
    abstract
      lunit[+ⱽ]<_> : ∀ {N} (xs : ⟬ A ⟭[ N ]) → 𝟎 + xs ≡ xs
      lunit[+ⱽ]< [] > = ↯
      lunit[+ⱽ]< x ∷ xs > rewrite lunit[+]< x > | lunit[+ⱽ]< xs > = ↯

      runit[+ⱽ]<_> : ∀ {N} (xs : ⟬ A ⟭[ N ]) → xs + 𝟎 ≡ xs
      runit[+ⱽ]< [] > = ↯
      runit[+ⱽ]< x ∷ xs > rewrite runit[+]< x > | runit[+ⱽ]< xs > = ↯

      commu[+ⱽ]<_,_> : ∀ {N} (xs ys : ⟬ A ⟭[ N ]) → xs + ys ≡ ys + xs
      commu[+ⱽ]< [] , [] > = ↯
      commu[+ⱽ]< x ∷ xs , y ∷ ys > rewrite commu[+]< x , y > | commu[+ⱽ]< xs , ys > = ↯

      assoc[+ⱽ]<_,_,_> : ∀ {N} (xs ys zs : ⟬ A ⟭[ N ]) → xs + (ys + zs) ≡ (xs + ys) + zs
      assoc[+ⱽ]< [] , [] , [] > = ↯
      assoc[+ⱽ]< x ∷ xs , y ∷ ys , z ∷ zs > rewrite assoc[+]< x , y , z > | assoc[+ⱽ]< xs , ys , zs > = ↯

    instance
      cor[+][vec] : ∀ {N} → cor[+] ⟬ A ⟭[ N ]
      cor[+][vec] = record
        { lunit[+]<_> = lunit[+ⱽ]<_>
        ; runit[+]<_> = runit[+ⱽ]<_>
        ; commu[+]<_,_> = commu[+ⱽ]<_,_>
        ; assoc[+]<_,_,_> = assoc[+ⱽ]<_,_,_>
        }
    
    module _ {{_ : split[+] A}} where
      abstract
        split[+ⱽ]<_,_> : ∀ {N} (xs ys : ⟬ A ⟭[ N ]) → xs + ys ≡ zero → xs ≡ zero ∧ ys ≡ zero
        split[+ⱽ]< [] , [] > ε = ⟨ ↯ , ↯ ⟩
        split[+ⱽ]< x ∷ xs , y ∷ ys > ε with x + y | xs + ys | split[+]< x , y > | split[+ⱽ]< xs , ys >
        split[+ⱽ]< x ∷ xs , y ∷ ys > ↯ | x+y | xs+ys | ε₁ | ε₂ with ε₁ ↯ | ε₂ ↯
        … | ⟨ ↯ , ↯ ⟩ | ⟨ ↯ , ↯ ⟩ = ⟨ ↯ , ↯ ⟩

      instance
        cor†[+][vec] : ∀ {N} → split[+] ⟬ A ⟭[ N ]
        cor†[+][vec] = record
          { split[+]<_,_> = split[+ⱽ]<_,_>
          }

    module _ {{_ : inj[+] A}} where
      abstract
        linj[+ⱽ]<_,_,_> : ∀ {N} (xs ys zs : ⟬ A ⟭[ N ]) → xs + ys ≡ xs + zs → ys ≡ zs
        linj[+ⱽ]< [] , [] , [] > ↯ = ↯
        linj[+ⱽ]< x ∷ xs , y ∷ ys , z ∷ zs > ε with x + y | xs + ys | linj[+]< x , y , z > | linj[+ⱽ]< xs , ys , zs >
        linj[+ⱽ]< x ∷ xs , y ∷ ys , z ∷ zs > ↯ | .(x + z) | .(xs +ⱽ zs) | H | IH rewrite H ↯ | IH ↯ = ↯

        rinj[+ⱽ]<_,_,_> : ∀ {N} (xs ys zs : ⟬ A ⟭[ N ]) → xs + zs ≡ ys + zs → xs ≡ ys
        rinj[+ⱽ]< [] , [] , [] > ↯ = ↯
        rinj[+ⱽ]< x ∷ xs , y ∷ ys , z ∷ zs > ε with x + z | xs + zs | rinj[+]< x , y , z > | rinj[+ⱽ]< xs , ys , zs >
        rinj[+ⱽ]< x ∷ xs , y ∷ ys , z ∷ zs > ↯ | .(y + z) | .(ys +ⱽ zs) | H | IH rewrite H ↯ | IH ↯ = ↯

      instance
        cor⋆[+][vec] : ∀ {N} → inj[+] ⟬ A ⟭[ N ]
        cor⋆[+][vec] = record
          { linj[+]<_,_,_> = linj[+ⱽ]<_,_,_>
          ; rinj[+]<_,_,_> = rinj[+ⱽ]<_,_,_>
          }

module _ {ℓ} {A : Set ℓ} {{_ : has[+] A}} {{_ : has[×] A}} where
  𝟏<_> : ∀ N → ⟬ A ⟭[ N ]
  𝟏< ᴢ > = []
  𝟏< ꜱ N > = one ∷ 𝟏< N >

  𝟏 : ∀ {N} → ⟬ A ⟭[ N ]
  𝟏 = 𝟏< _ >

  infixl 16 _⨰_
  _⨰_ : ∀ {N} → ⟬ A ⟭[ N ] → ⟬ A ⟭[ N ] → A
  [] ⨰ [] = zero
  (x ∷ xs) ⨰ (y ∷ ys) = x × y + xs ⨰ ys

  module _ {{_ : cor[+] A}} {{_ : cor[×] A}} where
    abstract
      lzero[⨰]<_> : ∀ {N} (xs : ⟬ A ⟭[ N ]) → zero ⨰ xs ≡ zero
      lzero[⨰]< [] > = ↯
      lzero[⨰]< x ∷ xs > rewrite lzero[×]< x > | lzero[⨰]< xs > | lunit[+]< zero AT A > = ↯

      rzero[⨰]<_> : ∀ {N} (xs : ⟬ A ⟭[ N ]) → xs ⨰ zero ≡ zero
      rzero[⨰]< [] > = ↯
      rzero[⨰]< x ∷ xs > rewrite rzero[×]< x > | rzero[⨰]< xs > | lunit[+]< zero AT A > = ↯

module _ {ℓ} {A : Set ℓ} {{_ : has[+] A}} {{_ : has[×] A}} where
  infixr 16 _⨵_
  _⨵_ : ∀ {N} → A → ⟬ A ⟭[ N ] → ⟬ A ⟭[ N ]
  x ⨵ ys = map (λ y → x × y) ys

  module _ {{_ : cor[+] A}} {{_ : cor[×] A}} where
    abstract
      lzero[⨵]<_> : ∀ {N} (xs : ⟬ A ⟭[ N ]) → zero ⨵ xs ≡ 𝟎
      lzero[⨵]< [] > = ↯
      lzero[⨵]< x ∷ xs > rewrite lzero[×]< x > | lzero[⨵]< xs > = ↯

      rzero[⨵]<_,_> : ∀ N (x : A) → x ⨵ 𝟎< N > ≡ zero
      rzero[⨵]< ᴢ , x > = ↯
      rzero[⨵]< ꜱ N , x > rewrite rzero[×]< x > | rzero[⨵]< N , x > = ↯

      lunit[⨵]<_> : ∀ {N} (xs : ⟬ A ⟭[ N ]) → one ⨵ xs ≡ xs
      lunit[⨵]< [] > = ↯
      lunit[⨵]< x ∷ xs > rewrite lunit[×]< x > | lunit[⨵]< xs > = ↯

      ldist[⨵]<_,_,_> : ∀ {N} (x : A) (ys zs : ⟬ A ⟭[ N ]) → x ⨵ (ys + zs) ≡ x ⨵ ys + x ⨵ zs
      ldist[⨵]< x , [] , [] > = ↯
      ldist[⨵]< x , y ∷ ys , z ∷ zs > rewrite ldist[×]< x , y , z > | ldist[⨵]< x , ys , zs > = ↯

      rdist[⨵]<_,_,_> : ∀ {N} (x y : A) (zs : ⟬ A ⟭[ N ]) → (x + y) ⨵ zs ≡ x ⨵ zs + y ⨵ zs
      rdist[⨵]< x , y , [] > = ↯
      rdist[⨵]< x , y , z ∷ zs > rewrite rdist[×]< x , y , z > | rdist[⨵]< x , y , zs > = ↯

      assoc[⨵]<_,_,_> : ∀ {N} (x y : A) (zs : ⟬ A ⟭[ N ]) → x ⨵ (y ⨵ zs) ≡ (x × y) ⨵ zs
      assoc[⨵]< x , y , [] > = ↯
      assoc[⨵]< x , y , z ∷ zs > rewrite assoc[×]< x , y , z > | assoc[⨵]< x , y , zs > = ↯

--   M×N     N = 2
--        ⌌----^----⌍   
--       ⌌[ [ a b ] ]
-- M = 3 {[ [ c d ] ]
--       ⌎[ [ e f ] ]
--
-- inner lists are row vectors [a,b] [c,d] [e,f]

⟬_⟭[_,_] : ∀ {ℓ} (A : Set ℓ) → ℕ → ℕ → Set ℓ
⟬ A ⟭[ M , N ] = ⟬ ⟬ A ⟭[ N ] ⟭[ M ]

-- slicing --

module _ {ℓ} {A : Set ℓ} where
  infixl 27 _[★,_]

  _[★,_] : ∀ {M N} (xs : ⟬ A ⟭[ M , N ]) (i : idx N) → ⟬ A ⟭[ M ]
  [] [★, i ] = []
  (xs ∷ xss) [★, i ] = xs #[ i ] ∷ xss [★, i ]

-- vertical cons --

module _ {ℓ} {A : Set ℓ} where
  infixr 20 _⁘_
  _⁘_ : ∀ {M N} → ⟬ A ⟭[ M ] → ⟬ A ⟭[ M , N ] → ⟬ A ⟭[ M , ꜱ N ]
  [] ⁘ [] = []
  (x ∷ xs) ⁘ (ys ∷ yss) = (x ∷ ys) ∷ xs ⁘ yss

  infixl 15 _⋕_
  _⋕_ : ∀ {M N₁ N₂} (xss : ⟬ A ⟭[ M , N₁ ]) (yss : ⟬ A ⟭[ M , N₂ ]) → ⟬ A ⟭[ M , N₁ + N₂ ]
  [] ⋕ [] = []
  xs ∷ xss ⋕ ys ∷ yss = (xs ⧻ ys) ∷ (xss ⋕ yss)

  abstract
    lunit[⋕]<_> : ∀ {M N} (xss : ⟬ A ⟭[ M , N ]) → 𝐜 [] ⋕ xss ≡ xss
    lunit[⋕]< [] > = ↯
    lunit[⋕]< xs ∷ xss > rewrite lunit[⋕]< xss > = ↯

    dist[𝐜/∷]<_,_,_> : ∀ M {N} (x : A) (xs : ⟬ A ⟭[ N ]) → 𝐜< M > (x ∷ xs) ≡ 𝐜< M > x ⁘ 𝐜< M > xs
    dist[𝐜/∷]< ᴢ , x , xs > = ↯
    dist[𝐜/∷]< ꜱ M , x , xs > rewrite dist[𝐜/∷]< M , x , xs > = ↯

    dist[⧻/⁘]<_,_,_,_> : ∀ {M₁ M₂ N} (ws : ⟬ A ⟭[ M₁ ]) (xss : ⟬ A ⟭[ M₁ , N ]) (ys : ⟬ A ⟭[ M₂ ]) (zss : ⟬ A ⟭[ M₂ , N ]) → ws ⁘ xss ⧻ ys ⁘ zss ≡ (ws ⧻ ys) ⁘ (xss ⧻ zss)
    dist[⧻/⁘]< [] , [] , ys , zss > = ↯
    dist[⧻/⁘]< w ∷ ws , xs ∷ xss , ys , zss > rewrite dist[⧻/⁘]< ws , xss , ys , zss > = ↯

    dist[⋕/⁘]<_,_,_> : ∀ {M N₁ N₂} (xs : ⟬ A ⟭[ M ]) (yss : ⟬ A ⟭[ M , N₁ ]) (zss : ⟬ A ⟭[ M , N₂ ]) → xs ⁘ yss ⋕ zss ≡ xs ⁘ (yss ⋕ zss)
    dist[⋕/⁘]< [] , [] , [] > = ↯
    dist[⋕/⁘]< x ∷ xs , ys ∷ yss , zs ∷ zss > rewrite dist[⋕/⁘]< xs , yss , zss > = ↯

  _ᵀ : ∀ {M N} → ⟬ A ⟭[ M , N ] → ⟬ A ⟭[ N , M ]
  [] ᵀ = 𝐜 []
  (xs ∷ xss) ᵀ = xs ⁘ xss ᵀ

module _ {ℓ} {A : Set ℓ} {{_ : has[+] A}} {{_ : has[×] A}} where
  𝐝<_> : ∀ N → ⟬ A ⟭[ N , N ]
  𝐝< ᴢ > = []
  𝐝< ꜱ N > = (one ∷ zero) ∷ (zero ⁘ 𝐝< N >)

  𝐝 : ∀ {N} → ⟬ A ⟭[ N , N ]
  𝐝 = 𝐝< _ >

---------------------------
-- dimension elimination --
---------------------------

module _ {ℓ} {A : Set ℓ} {{_ : has[+] A}} {{_ : has[×] A}} where
  infixl 16 _⨵²_
  _⨵²_ : ∀ {M N} → ⟬ A ⟭[ M ] → ⟬ A ⟭[ M , N ] → ⟬ A ⟭[ N ]
  [] ⨵² [] = zero
  x ∷ xs ⨵² ys ∷ yss = x ⨵ ys + xs ⨵² yss

  abstract
    dist[⨵²/⁘]<_,_,_> : ∀ {M N} (xs : ⟬ A ⟭[ M ]) (ys : ⟬ A ⟭[ M ]) (zss : ⟬ A ⟭[ M , N ]) → xs ⨵² ys ⁘ zss ≡ (xs ⨰ ys) ∷ (xs ⨵² zss)
    dist[⨵²/⁘]< [] , [] , [] > = ↯
    dist[⨵²/⁘]< x ∷ xs , y ∷ ys , zs ∷ zss > rewrite dist[⨵²/⁘]< xs , ys , zss > = ↯

  module _ {{_ : cor[+] A}} {{_ : cor[×] A}} where
    abstract
      lzero[⨵²]<_> : ∀ {M N} (xss : ⟬ A ⟭[ M , N ]) → 𝟎 ⨵² xss ≡ 𝟎
      lzero[⨵²]< [] > = ↯
      lzero[⨵²]<_> {ꜱ M} {N} (xs ∷ xss) rewrite lzero[⨵]< xs > | lzero[⨵²]< xss > | lunit[+]< 𝟎 AT ⟬ A ⟭[ N ] > = ↯

      runit[⨵²]<_> : ∀ {N} (xs : ⟬ A ⟭[ N ]) → xs ⨵² 𝐝 ≡ xs
      runit[⨵²]< [] > = ↯
      runit[⨵²]<_> {ꜱ N} (x ∷ xs)
        rewrite runit[×]< x >
              | rzero[⨵]< N , x >
              | dist[⨵²/⁘]< xs , 𝟎 , 𝐝 >
              | rzero[⨰]< xs >
              | runit[+]< x >
              | runit[⨵²]< xs >
              | lunit[+]< xs >
              = ↯

      rdist[⨵²]<_,_,_> : ∀ {M N} (xs ys : ⟬ A ⟭[ M ]) (zss : ⟬ A ⟭[ M , N ]) → (xs + ys) ⨵² zss ≡ xs ⨵² zss + ys ⨵² zss
      rdist[⨵²]<_,_,_> {M} {N} [] [] [] rewrite lunit[+]< 𝟎 AT ⟬ A ⟭[ N ] > = ↯
      rdist[⨵²]< x ∷ xs , y ∷ ys , zs ∷ zss >
        rewrite rdist[⨵]< x , y , zs >
              | rdist[⨵²]< xs , ys , zss >
              | assoc[+]< x ⨵ zs + y ⨵ zs , xs ⨵² zss , ys ⨵² zss >
              | ◇⸢≡⸣ assoc[+]< x ⨵ zs , y ⨵ zs , xs ⨵² zss >
              | commu[+]< y ⨵ zs , xs ⨵² zss >
              | assoc[+]< x ⨵ zs , xs ⨵² zss , y ⨵ zs >
              | ◇⸢≡⸣ assoc[+]< x ⨵ zs + xs ⨵² zss , y ⨵ zs , ys ⨵² zss >
              = ↯

      assoc[⨵²]<_,_,_> : ∀ {M N} (x : A) (ys : ⟬ A ⟭[ M ]) (zss : ⟬ A ⟭[ M , N ]) → x ⨵ (ys ⨵² zss) ≡ (x ⨵ ys) ⨵² zss
      assoc[⨵²]<_,_,_> {M} {N} x [] [] rewrite rzero[⨵]< N , x > = ↯
      assoc[⨵²]< x , y ∷ ys , zs ∷ zss > rewrite ldist[⨵]< x , y ⨵ zs , ys ⨵² zss > | assoc[⨵]< x , y , zs > | assoc[⨵²]< x , ys , zss > = ↯

      reduc[⨵²/𝐮]<_,_> : ∀ {M N} (x : idx M) (yss : ⟬ A ⟭[ M , N ]) → 𝐮 x ⨵² yss ≡ yss #[ x ]
      reduc[⨵²/𝐮]< ᴢ , ys ∷ yss > rewrite lunit[⨵]< ys > | lzero[⨵²]< yss > | runit[+]< ys > = ↯ 
      reduc[⨵²/𝐮]< ꜱ x , ys ∷ yss > rewrite lzero[⨵]< ys > | lunit[+]< 𝐮 x ⨵² yss > | reduc[⨵²/𝐮]< x , yss > = ↯

module _ {ℓ} {A : Set ℓ} {{_ : has[+] A}} {{_ : has[×] A}} where
  --              [ [   g     h     i   ] ]
  --           ✖  [ [   j     k     l   ] ]
  -- 
  -- [ [ a b ] ]  [ [ ag+bj ah+bk ai+bl ] ]
  -- [ [ c d ] ]  [ [ cg+dj ch+dk ci+dl ] ]
  -- [ [ e f ] ]  [ [ eg+fj eh+fk ei+fl ] ]
  infixl 16 _✖_
  _✖_ : ∀ {M N O} → ⟬ A ⟭[ M , N ] → ⟬ A ⟭[ N , O ] → ⟬ A ⟭[ M , O ]
  xss ✖ yss = map (λ xs → xs ⨵² yss) xss
