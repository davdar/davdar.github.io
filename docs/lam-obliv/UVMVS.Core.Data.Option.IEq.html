module UVMVS.Core.Data.Option.IEq where

open import UVMVS.Core.Init
open import UVMVS.Core.Classes
open import UVMVS.Core.Data.Option.Injection
open import UVMVS.Core.Data.Unit
open import UVMVS.Core.Data.Choice

module _ {ℓ ℓᴿ} {A : Set ℓ} {{_ : has[≍][ ℓᴿ ] A}} where

  data _≍ᴼ_ : (x y : ⦉ A ⦊) → Set ℓᴿ where
    • : • ≍ᴼ •
    ⟨_⟩ : ∀ {x y} (ε : x ≍ y) → ⟨ x ⟩ ≍ᴼ ⟨ y ⟩

  data _≭ᴼ_ : (x y : ⦉ A ⦊) → Set ℓᴿ where
    •⟨⟩ : ∀ {x} → • ≭ᴼ ⟨ x ⟩
    ⟨⟩• : ∀ {x} → ⟨ x ⟩ ≭ᴼ •
    ⟨_⟩ : ∀ {x y} (ε : x ≭ y) → ⟨ x ⟩ ≭ᴼ ⟨ y ⟩

  instance
    has[≍][option] : has[≍][ ℓᴿ ] ⦉ A ⦊
    has[≍][option] = record { _≍_ = _≍ᴼ_ ; _≭_ = _≭ᴼ_ }

  abstract
    isoto[≍ᴼ] : ∀ {x y} → x ≍ᴼ y → ιᴼ x ≍ ιᴼ y
    isoto[≍ᴼ] • = ʟ •
    isoto[≍ᴼ] ⟨ ε ⟩ = ʀ ε

    isofr[≍ᴼ] : ∀ {x y} → ιᴼ x ≍ ιᴼ y → x ≍ᴼ y
    isofr[≍ᴼ] {•} {•} (ʟ •) = •
    isofr[≍ᴼ] {•} {⟨ y ⟩} ()
    isofr[≍ᴼ] {⟨ x ⟩} {•} ()
    isofr[≍ᴼ] {⟨ x ⟩} {⟨ y ⟩} (ʀ ε) = ⟨ ε ⟩

    isoto[≭ᴼ] : ∀ {x y} → x ≭ᴼ y → ιᴼ x ≭ ιᴼ y
    isoto[≭ᴼ] •⟨⟩ = ʟʀ
    isoto[≭ᴼ] ⟨⟩• = ʀʟ
    isoto[≭ᴼ] ⟨ ε ⟩ = ʀ ε

    isofr[≭ᴼ] : ∀ {x y} → ιᴼ x ≭ ιᴼ y → x ≭ᴼ y
    isofr[≭ᴼ] {•} {•} (ʟ ())
    isofr[≭ᴼ] {•} {⟨ y ⟩} ʟʀ = •⟨⟩
    isofr[≭ᴼ] {⟨ x ⟩} {•} ʀʟ = ⟨⟩•
    isofr[≭ᴼ] {⟨ x ⟩} {⟨ y ⟩} (ʀ ε) = ⟨ ε ⟩

  module _ {{_ : cor[≍] A}} where
    open derive[cor[≍]] ⦉ A ⦊ (𝟙 ∨ A) ιᴼ isoto[≍ᴼ] isofr[≍ᴼ] isoto[≭ᴼ] isofr[≭ᴼ]
    
    instance
      reflexive[≍ᴼ] = reflexive[≍]†
      symmetric[≍ᴼ] = symmetric[≍]†
      transitive[≍ᴼ] = transitive[≍]†
      irreflexive[≭ᴼ] = irreflexive[≭]†
      symmetric[≭ᴼ] = symmetric[≭]†
      cor[≍][option] = cor[≍]†
    
    _ = ⌁ AT reflexive    _≍ᴼ_
    _ = ⌁ AT symmetric    _≍ᴼ_
    _ = ⌁ AT transitive   _≍ᴼ_
    _ = ⌁ AT irreflexive  _≭ᴼ_
    _ = ⌁ AT symmetric    _≭ᴼ_
    _ = ⌁ AT cor[≍]       ⦉ A ⦊
  _ = ⌁ AT has[≍][ ℓᴿ ] ⦉ A ⦊
