BEGIN_END : ∀ {ℓ} {A : Set ℓ} (x : A) → A
BEGIN x END = x

CASE_OF_ : ∀ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : Set ℓ₂} (x : A) (f : A → B) → B
CASE x OF f = f x

syntax at A x = x AT A
at : ∀ {ℓ} (A : Set ℓ) → A → A
at A x = x

------------------------
-- FUNCTIONAL HELPERS --
------------------------

id : ∀ {ℓ} {A : Set ℓ} (x : A) → A
id x = x

const : ∀ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : Set ℓ₂} (y : B) (x : A) → B
const y x = y

flip : ∀ {ℓ₁ ℓ₂ ℓ₃} {A : Set ℓ₁} {B : Set ℓ₂} {C : Set ℓ₃} (f : A → B → C) (y : B) (x : A) → C
flip f y x = f x y

_$_ : ∀ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : Set ℓ₂} (f : A → B) (x : A) → B
f $ x = f x

_∘_ : ∀ {ℓ₁ ℓ₂ ℓ₃} {A : Set ℓ₁} {B : Set ℓ₂} {C : Set ℓ₃} (g : B → C) (f : A → B) (x : A) → C
(g ∘ f) x = g (f x)

data _≡_ {ℓ} {A : Set ℓ} (x : A) : A → Set where
  instance
    ↯ : x ≡ x

{-# BUILTIN EQUALITY _≡_ #-}

syntax ≡<> A x y = x ≡< A > y
≡<> : ∀ {ℓ} (A : Set ℓ) → A → A → Set
≡<> _ = _≡_

abstract
  xRx⸢≡⸣<_> : ∀ {ℓ} {A : Set ℓ} (x : A) → x ≡ x
  xRx⸢≡⸣< _ > = ↯
  
  ◇⸢≡⸣ : ∀ {ℓ} {A : Set ℓ} {x y : A} → x ≡ y → y ≡ x
  ◇⸢≡⸣ ↯ = ↯
  
  _⊚⸢≡⸣_ : ∀ {ℓ} {A : Set ℓ} {x y z : A} → y ≡ z → x ≡ y → x ≡ z
  ↯ ⊚⸢≡⸣ ↯ = ↯

  cong[_] : ∀ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : Set ℓ₂} (F : A → B) {x y} (ε : x ≡ y) → F x ≡ F y
  cong[ F ] ↯ = ↯

subst[_] : ∀ {ℓ₁ ℓ₂} {A : Set ℓ₁} (B : (x : A) → Set ℓ₂) {x y} (ε : x ≡ y) → B x → B y
subst[ B ] ↯ x = x

subst2[_] : ∀ {ℓ₁ ℓ₂ ℓ₃} {A : Set ℓ₁} {B : Set ℓ₂} (C : (x : A) (y : B) → Set ℓ₃) {x₁ x₂ y₁ y₂} (ε₁ : x₁ ≡ x₂) (ε₂ : y₁ ≡ y₂) → C x₁ y₁ → C x₂ y₂
subst2[ B ] ↯ ↯ x = x

change[_] : ∀ {ℓ₁ ℓ₂} {A : Set ℓ₁} (B : (x : A) → Set ℓ₂) {x y} (ε : x ≡ y) → B y → B x
change[ B ] ↯ x = x

change : ∀ {ℓ} {A B : Set ℓ} (ε : A ≡ B) → B → A
change = change[ id ]

change2[_] : ∀ {ℓ₁ ℓ₂ ℓ₃} {A : Set ℓ₁} {B : Set ℓ₂} (C : (x : A) (y : B) → Set ℓ₃) {x₁ x₂ y₁ y₂} (ε₁ : x₁ ≡ x₂) (ε₂ : y₁ ≡ y₂) → C x₂ y₂ → C x₁ y₁
change2[ B ] ↯ ↯ x = x

irr[≡] : ∀ {ℓ} {A : Set ℓ} {x y : A} → .(x ≡ y) → x ≡ y
irr[≡] ↯ = ↯

data 𝟘 : Set where

¬ : ∀ {ℓ} (A : Set ℓ) → Set ℓ
¬ A = A → 𝟘

absurd : ∀ {ℓ} {A : Set ℓ} .(v : 𝟘) → A
absurd ()

contra : ∀ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : Set ℓ₂} .(ε₁ : ¬ A) .(ε₂ : A) → B
contra ε₁ ε₂ = absurd (ε₁ ε₂)

data _≢_ {ℓ} {A : Set ℓ} (x y : A) : Set ℓ where
  ⟨_⟩ : (ε : ¬ (x ≡ y)) → x ≢ y

syntax ≢<> A x y = x ≢< A > y
≢<> : ∀ {ℓ} (A : Set ℓ) → A → A → Set ℓ
≢<> _ = _≢_

contra[≡] : ∀ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : Set ℓ₂} {x y : A} .(ε₁ : x ≢ y) .(ε₂ : x ≡ y) → B
contra[≡] ε₁ ε₂ = absurd (CASE ε₁ OF λ where ⟨ ε′ ⟩ → ε′ ε₂)

¬xRx⸢≢⸣ : ∀ {ℓ} {A : Set ℓ} {x : A} → ¬ (x ≢ x)
¬xRx⸢≢⸣ ε = contra[≡] ε ↯

◇⸢≢⸣ : ∀ {ℓ} {A : Set ℓ} {x y : A} → x ≢ y → y ≢ x
◇⸢≢⸣ ⟨ ε ⟩ = ⟨ (λ where ↯ → ε ↯) ⟩

syntax ∃ST (λ x → e) = ∃ x ST e
record ∃ST {ℓ₁ ℓ₂} {A : Set ℓ₁} (B : ∀ (x : A) → Set ℓ₂) : Set (ℓ₁ ⩛ ℓ₂) where
  no-eta-equality
  constructor ⟨∃_,_⟩
  field
    dπ₁ : A
    dπ₂ : B dπ₁
open ∃ST public

syntax ∃ST⦂ A (λ x → e) = ∃ x ⦂ A ST e
∃ST⦂ : ∀ {ℓ₁ ℓ₂} (A : Set ℓ₁) (B : ∀ (x : A) → Set ℓ₂) → Set (ℓ₁ ⩛ ℓ₂)
∃ST⦂ A B = ∃ x ST B x

-- Borrowed heavily from Ulf Norell

delay : ∀ {ℓ} → Set ℓ → Set ℓ
delay A = 𝟙 → A

hide : ∀ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : A → Set ℓ₂} (f : ∀ (x : A) → B x) (x : A) → delay (B x)
hide f x • = f x

reveal : ∀ {ℓ} {A : Set ℓ} → delay A → A
reveal x = x •

data RECALL_IS_ {ℓ} {A : Set ℓ} (x : A) (y : delay A) : Set ℓ where
  Was : reveal y ≡ x → RECALL x IS y

remember/Π : ∀ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : A → Set ℓ₂} (f : ∀ (x : A) → B x) (x : A) → RECALL f x IS hide f x
remember/Π f x = Was ↯

remember : ∀ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : Set ℓ₂} (f : A → B) (x : A) → RECALL f x IS hide f x
remember = remember/Π
