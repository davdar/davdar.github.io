module LamObliv.Sampling.Weakening where

open import UVMVS

open import LamObliv.Common

open import LamObliv.Sampling.Syntax

module _ {R I} {Σ : tsto R I} where
  infixl 11 _⇧[Σ]ⱽ_
  _⇧[Σ]ⱽ_ : ∀ {τ} (v : Σ ⊢ⱽ τ) {I′} (Σ′ : tsto R I′) → Σ ⇧ⱽ Σ′ ⊢ⱽ τ
  Bit b ℓ ⇧[Σ]ⱽ Σ′ = Bit b ℓ
  Flip b ⇧[Σ]ⱽ Σ′ = Flip b
  Pair v₁ v₂ ⇧[Σ]ⱽ Σ′ = Pair (v₁ ⇧[Σ]ⱽ Σ′) (v₂ ⇧[Σ]ⱽ Σ′)
  Addr ε ⇧[Σ]ⱽ Σ′ = Addr (ε ⇧ᵋ Σ′)
  
  infixl 11 _⇧[Σ]ᴱ_
  _⇧[Σ]ᴱ_ : ∀ {N} {Γ : tcxt R N} {A A′ : acxt N} {τ} (e : Σ , Γ , A ⊢ᴱ τ , A′) {I′} (Σ′ : tsto R I′) → Σ ⇧ⱽ Σ′ , Γ , A ⊢ᴱ τ , A′
  Var ε₁ ε₂ ↯ ⇧[Σ]ᴱ Σ′ = Var ε₁ ε₂ ↯
  Val v ⇧[Σ]ᴱ Σ′ = Val (v ⇧[Σ]ⱽ Σ′)
  Bit b r ℓ ⇧[Σ]ᴱ Σ′ = Bit b r ℓ
  Flip ρ r ⇧[Σ]ᴱ Σ′ = Flip ρ r
  CastS ε₁ ε₂ ↯ ⇧[Σ]ᴱ Σ′ = CastS ε₁ ε₂ ↯
  CastS-V v ⇧[Σ]ᴱ Σ′ = CastS-V (v ⇧[Σ]ⱽ Σ′)
  CastP e ⇧[Σ]ᴱ Σ′ = CastP (e ⇧[Σ]ᴱ Σ′)
  MuxBit e₁ e₂ e₃ ⇧[Σ]ᴱ Σ′ = MuxBit (e₁ ⇧[Σ]ᴱ Σ′) (e₂ ⇧[Σ]ᴱ Σ′) (e₃ ⇧[Σ]ᴱ Σ′)
  MuxFlip e₁ e₂ e₃ ε₁ ε₂ ⇧[Σ]ᴱ Σ′ = MuxFlip (e₁ ⇧[Σ]ᴱ Σ′) (e₂ ⇧[Σ]ᴱ Σ′) (e₃ ⇧[Σ]ᴱ Σ′) ε₁ ε₂
  If e₁ e₂ e₃ ↯ ⇧[Σ]ᴱ Σ′ = If (e₁ ⇧[Σ]ᴱ Σ′) (e₂ ⇧[Σ]ᴱ Σ′) (e₃ ⇧[Σ]ᴱ Σ′) ↯
  Pair e₁ e₂ ⇧[Σ]ᴱ Σ′ = Pair (e₁ ⇧[Σ]ᴱ Σ′) (e₂ ⇧[Σ]ᴱ Σ′)
  Let e₁ e₂ ⇧[Σ]ᴱ Σ′ = Let (e₁ ⇧[Σ]ᴱ Σ′) (e₂ ⇧[Σ]ᴱ Σ′)
  LetPair e₁ e₂ ⇧[Σ]ᴱ Σ′ = LetPair (e₁ ⇧[Σ]ᴱ Σ′) (e₂ ⇧[Σ]ᴱ Σ′)
  Ref e ⇧[Σ]ᴱ Σ′ = Ref (e ⇧[Σ]ᴱ Σ′)
  Read/Write e₁ e₂ ⇧[Σ]ᴱ Σ′ = Read/Write (e₁ ⇧[Σ]ᴱ Σ′) (e₂ ⇧[Σ]ᴱ Σ′)
  
  infixl 11 _⇧[Σ]ᶜ_
  _⇧[Σ]ᶜ_  : ∀ {τ⁻ τ⁺} (c : Σ ⊢ τ⁻ ⇛ τ⁺) {I′} (Σ′ : tsto R I′) → Σ ⇧ⱽ Σ′ ⊢ τ⁻ ⇛ τ⁺
  □ ⇧[Σ]ᶜ Σ′ = □
  CastP[c] c ⇧[Σ]ᶜ Σ′ = CastP[c] (c ⇧[Σ]ᶜ Σ′)
  MuxBit[c,e,e] c₁ e₂ e₃ ⇧[Σ]ᶜ Σ′ = MuxBit[c,e,e] (c₁ ⇧[Σ]ᶜ Σ′) (e₂ ⇧[Σ]ᴱ Σ′) (e₃ ⇧[Σ]ᴱ Σ′)
  MuxBit[v,c,e] v₁ c₂ e₃ ⇧[Σ]ᶜ Σ′ = MuxBit[v,c,e] (v₁ ⇧[Σ]ⱽ Σ′) (c₂ ⇧[Σ]ᶜ Σ′) (e₃ ⇧[Σ]ᴱ Σ′)
  MuxBit[v,v,c] v₁ v₂ c₃ ⇧[Σ]ᶜ Σ′ = MuxBit[v,v,c] (v₁ ⇧[Σ]ⱽ Σ′) (v₂ ⇧[Σ]ⱽ Σ′) (c₃ ⇧[Σ]ᶜ Σ′)
  MuxFlip[c,e,e] c₁ e₂ e₃ ε₁ ε₂ ⇧[Σ]ᶜ Σ′ = MuxFlip[c,e,e] (c₁ ⇧[Σ]ᶜ Σ′) (e₂ ⇧[Σ]ᴱ Σ′) (e₃ ⇧[Σ]ᴱ Σ′) ε₁ ε₂
  MuxFlip[v,c,e] v₁ c₂ e₃ ε₁ ε₂ ⇧[Σ]ᶜ Σ′ = MuxFlip[v,c,e] (v₁ ⇧[Σ]ⱽ Σ′) (c₂ ⇧[Σ]ᶜ Σ′) (e₃ ⇧[Σ]ᴱ Σ′) ε₁ ε₂
  MuxFlip[v,v,c] v₁ v₂ c₃ ε₁ ε₂ ⇧[Σ]ᶜ Σ′ = MuxFlip[v,v,c] (v₁ ⇧[Σ]ⱽ Σ′) (v₂ ⇧[Σ]ⱽ Σ′) (c₃ ⇧[Σ]ᶜ Σ′) ε₁ ε₂
  If[c,e,e] c₁ e₂ e₃ ⇧[Σ]ᶜ Σ′ = If[c,e,e] (c₁ ⇧[Σ]ᶜ Σ′) (e₂ ⇧[Σ]ᴱ Σ′) (e₃ ⇧[Σ]ᴱ Σ′)
  Pair[c,e] c₁ e₂ ⇧[Σ]ᶜ Σ′ = Pair[c,e] (c₁ ⇧[Σ]ᶜ Σ′) (e₂ ⇧[Σ]ᴱ Σ′)
  Pair[v,c] v₁ c₂ ⇧[Σ]ᶜ Σ′ = Pair[v,c] (v₁ ⇧[Σ]ⱽ Σ′) (c₂ ⇧[Σ]ᶜ Σ′)
  Let[c,e] c₁ e₂ ⇧[Σ]ᶜ Σ′ = Let[c,e] (c₁ ⇧[Σ]ᶜ Σ′) (e₂ ⇧[Σ]ᴱ Σ′)
  LetPair[c,e] c₁ e₂ ⇧[Σ]ᶜ Σ′ = LetPair[c,e] (c₁ ⇧[Σ]ᶜ Σ′) (e₂ ⇧[Σ]ᴱ Σ′)
  Ref[c] c ⇧[Σ]ᶜ Σ′ = Ref[c] (c ⇧[Σ]ᶜ Σ′)
  Read/Write[c,e] c₁ e₂ ⇧[Σ]ᶜ Σ′ = Read/Write[c,e] (c₁ ⇧[Σ]ᶜ Σ′) (e₂ ⇧[Σ]ᴱ Σ′)
  Read/Write[v,c] v₁ c₂ ⇧[Σ]ᶜ Σ′ = Read/Write[v,c] (v₁ ⇧[Σ]ⱽ Σ′) (c₂ ⇧[Σ]ᶜ Σ′)

  infixl 11 _⇧[Σ]ᶻ_
  _⇧[Σ]ᶻ_  : ∀ {τ⁻ τ⁺} (c : Σ ⊢ τ⁺ ⇚ τ⁻) {I′} (Σ′ : tsto R I′) → Σ ⇧ⱽ Σ′ ⊢ τ⁺ ⇚ τ⁻
  `⊤ ⇧[Σ]ᶻ Σ′ = `⊤
  CastP[□] κ ⇧[Σ]ᶻ Σ′ = CastP[□] (κ ⇧[Σ]ᶻ Σ′)
  MuxBit[□,e,e] e₂ e₃ κ ⇧[Σ]ᶻ Σ′ = MuxBit[□,e,e] (e₂ ⇧[Σ]ᴱ Σ′) (e₃ ⇧[Σ]ᴱ Σ′) (κ ⇧[Σ]ᶻ Σ′)
  MuxBit[v,□,e] v₁ e₃ κ ⇧[Σ]ᶻ Σ′ = MuxBit[v,□,e] (v₁ ⇧[Σ]ⱽ Σ′) (e₃ ⇧[Σ]ᴱ Σ′) (κ ⇧[Σ]ᶻ Σ′)
  MuxBit[v,v,□] v₁ v₂ κ ⇧[Σ]ᶻ Σ′ = MuxBit[v,v,□] (v₁ ⇧[Σ]ⱽ Σ′) (v₂ ⇧[Σ]ⱽ Σ′)(κ ⇧[Σ]ᶻ Σ′)
  MuxFlip[□,e,e] e₂ e₃ ε₁ ε₂ κ ⇧[Σ]ᶻ Σ′ = MuxFlip[□,e,e] (e₂ ⇧[Σ]ᴱ Σ′) (e₃ ⇧[Σ]ᴱ Σ′) ε₁ ε₂ (κ ⇧[Σ]ᶻ Σ′)
  MuxFlip[v,□,e] v₁ e₃ ε₁ ε₂ κ ⇧[Σ]ᶻ Σ′ = MuxFlip[v,□,e] (v₁ ⇧[Σ]ⱽ Σ′) (e₃ ⇧[Σ]ᴱ Σ′) ε₁ ε₂ (κ ⇧[Σ]ᶻ Σ′)
  MuxFlip[v,v,□] v₁ v₂ ε₁ ε₂ κ ⇧[Σ]ᶻ Σ′ = MuxFlip[v,v,□] (v₁ ⇧[Σ]ⱽ Σ′) (v₂ ⇧[Σ]ⱽ Σ′) ε₁ ε₂ (κ ⇧[Σ]ᶻ Σ′)
  If[□,e,e] e₂ e₃ κ ⇧[Σ]ᶻ Σ′ = If[□,e,e] (e₂ ⇧[Σ]ᴱ Σ′) (e₃ ⇧[Σ]ᴱ Σ′) (κ ⇧[Σ]ᶻ Σ′)
  Pair[□,e] e₂ κ ⇧[Σ]ᶻ Σ′ = Pair[□,e] (e₂ ⇧[Σ]ᴱ Σ′) (κ ⇧[Σ]ᶻ Σ′)
  Pair[v,□] v₁ κ ⇧[Σ]ᶻ Σ′ = Pair[v,□] (v₁ ⇧[Σ]ⱽ Σ′)(κ ⇧[Σ]ᶻ Σ′)
  Let[□,e] e₂ κ ⇧[Σ]ᶻ Σ′ = Let[□,e] (e₂ ⇧[Σ]ᴱ Σ′) (κ ⇧[Σ]ᶻ Σ′)
  LetPair[□,e] e₂ κ ⇧[Σ]ᶻ Σ′ = LetPair[□,e] (e₂ ⇧[Σ]ᴱ Σ′) (κ ⇧[Σ]ᶻ Σ′)
  Ref[□] κ ⇧[Σ]ᶻ Σ′ = Ref[□] (κ ⇧[Σ]ᶻ Σ′)
  Read/Write[□,e] e₂ κ ⇧[Σ]ᶻ Σ′ = Read/Write[□,e] (e₂ ⇧[Σ]ᴱ Σ′) (κ ⇧[Σ]ᶻ Σ′)
  Read/Write[v,□] v₁ κ ⇧[Σ]ᶻ Σ′ = Read/Write[v,□] (v₁ ⇧[Σ]ⱽ Σ′)(κ ⇧[Σ]ᶻ Σ′)
  
  infixl 11 _⇧[Σ]ˢ_
  _⇧[Σ]ˢ_ : ∀ {I₀} {Σ₀ : tsto R I₀} (σ : vsto< Σ₀ > Σ) {I′} (Σ′ : tsto R I′) → vsto< Σ₀ > (Σ ⇧ⱽ Σ′)
  [] ⇧[Σ]ˢ Σ′ = []
  v ∷ σ ⇧[Σ]ˢ Σ′ = (v ⇧[Σ]ⱽ Σ′) ∷ (σ ⇧[Σ]ˢ Σ′)
