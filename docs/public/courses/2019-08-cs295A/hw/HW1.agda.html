module Lib where
  infix 4 _≡_

  data _≡_ {A : Set} (x : A) : A → Set where
    ↯ : x ≡ x

  {-# BUILTIN EQUALITY _≡_ #-}

  data 𝔹 : Set where
    True : 𝔹
    False : 𝔹

  data ℕ : Set where
    Z : ℕ
    S : ℕ → ℕ
  
  {-# BUILTIN NATURAL ℕ #-}
  
  _+_ : ℕ → ℕ → ℕ
  Z    + n  =  n
  (S m) + n  =  S (m + n)
  
  data animal : Set where
    Cat : animal
    Dog : animal
    Fish : animal

  is-fuzzy : animal → 𝔹
  is-fuzzy Cat = True
  is-fuzzy Dog = True
  is-fuzzy Fish = False

  is-burpy : animal → 𝔹
  is-burpy Cat = False
  is-burpy Dog = True
  is-burpy Fish = True

  postulate
    would-pet[_] : animal → Set

    would-pet-cat : would-pet[ Cat ]
    would-pet-dog : would-pet[ Dog ]
