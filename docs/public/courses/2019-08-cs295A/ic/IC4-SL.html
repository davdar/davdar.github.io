  Z : ℕ
  S : ℕ → ℕ

infixl 5 _+_
_+_ : ℕ → ℕ → ℕ
Z + n = n
S m + n = S (m + n)

_ = ↯

_ : (1 + 2) + 3 ≡ 1 + (2 + 3)
_ = ↯

infixl 6 _×_
_×_ : ℕ → ℕ → ℕ
Z × n = Z
S m × n = n + m × n

_ : 2 × 3 ≡ 6
_ = ↯

infixl 5 _∸_
_∸_ : ℕ → ℕ → ℕ
Z ∸ n = Z
m ∸ Z = m
S m ∸ S n = m ∸ n

_ : 2 + 2 × 3 ∸ 7 ≡ 1
_ = ↯

L1 : ∀ (m : ℕ) → m ≡ 4 → m + 2 ≡ 6
L1 .4 ↯ = ↯

L2 : ∀ (m : ℕ) → m ≡ 4 → m + 2 ≡ 6
L2 m ε rewrite ε = ↯

L3 : ∀ (m : ℕ) → 3 ≡ 4 → 1 ≡ 2
L3 m ()

L4 : ∀ (f g : ℕ → ℕ) (m : ℕ) → f m ≡ g m → 1 ≡ g m → f m ≡ 1
L4 f g m ε₁ ε₂ rewrite ε₁ | ◇ ε₂ = ↯

L5 : ∀ (f g : ℕ → ℕ) (m : ℕ) → f m ≡ g m → 1 ≡ g m → f m ≡ 1
L5 f g m ε₁ ε₂ = ◇ ε₂ ⊚ ε₁

lunit[+] : ∀ (m : ℕ) → 0 + m ≡ m
lunit[+] m = ↯

runit[+] : ∀ (m : ℕ) → m + 0 ≡ m
runit[+] Z = ↯
runit[+] (S m) rewrite runit[+] m = ↯

assoc[+] : ∀ (m n p : ℕ) → m + (n + p) ≡ (m + n) + p
assoc[+] Z n p = ↯
assoc[+] (S m) n p rewrite assoc[+] m n p = ↯

reduc[+] : ∀ (m n : ℕ) → m + S n ≡ S (m + n)
reduc[+] Z n = ↯
reduc[+] (S m) n rewrite reduc[+] m n = ↯

commu[+] : ∀ (m n : ℕ) → m + n ≡ n + m
commu[+] Z n rewrite runit[+] n = ↯
commu[+] (S m) n rewrite reduc[+] n m | commu[+] m n = ↯
