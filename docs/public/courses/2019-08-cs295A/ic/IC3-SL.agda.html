data 𝔹 : Set where
  True : 𝔹
  False : 𝔹

data color : Set where
  Red : color
  Green : color
  Blue : color

is-red : color → 𝔹
is-red Red = True
is-red Green = False
is-red Blue = False

data animal : Set where
  Cat : animal
  Dog : animal
  Fish : animal

david-has-one : animal → 𝔹
david-has-one Cat = True
david-has-one Dog = True
david-has-one Fish = False

postulate
  david-wants[_] : animal → Set

  dwc : david-wants[ Cat ] 
  dwd : david-wants[ Dog ]
  dwf : david-wants[ Fish ]

david-wants-all-animals : ∀ (a : animal) → david-wants[ a ]
david-wants-all-animals Cat = dwc
david-wants-all-animals Dog = dwd
david-wants-all-animals Fish = dwf

has-implies-wants : ∀ (a : animal) → david-has-one a ≡ True → david-wants[ a ]
has-implies-wants Cat has[a] = dwc
has-implies-wants Dog has[a] = dwd
has-implies-wants Fish ()
