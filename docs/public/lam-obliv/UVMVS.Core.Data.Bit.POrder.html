module UVMVS.Core.Data.Bit.POrder where

open import UVMVS.Core.Init
open import UVMVS.Core.Classes

open import UVMVS.Core.Data.Bit.Base

data _⊑ᴮ_ : (b₁ b₂ : 𝔹) → Set where
  ᴏ : ∀ {b} → ᴏ ⊑ᴮ b
  ɪ : ɪ ⊑ᴮ ɪ

instance
  has[⊑][𝔹] : has[⊑][ L0 ] 𝔹
  has[⊑][𝔹] = record { _⊑_ = _⊑ᴮ_ }

abstract
  xRx[⊑ᴮ]<_> : (b : 𝔹) → b ⊑ b
  xRx[⊑ᴮ]< ᴏ > = ᴏ
  xRx[⊑ᴮ]< ɪ > = ɪ

  _⋈[⊑ᴮ]_ : ∀ {b₁ b₂ : 𝔹} → b₁ ⊑ b₂ → b₂ ⊑ b₁ → b₁ ≡ b₂ 
  ᴏ ⋈[⊑ᴮ] ᴏ = ↯
  ɪ ⋈[⊑ᴮ] ɪ = ↯

  _⊚[⊑ᴮ]_ : ∀ {b₁ b₂ b₃ : 𝔹} → b₂ ⊑ b₃ → b₁ ⊑ b₂ → b₁ ⊑ b₃ 
  ε ⊚[⊑ᴮ] ᴏ = ᴏ
  ɪ ⊚[⊑ᴮ] ɪ = ɪ

instance
  reflexive[⊑ᴮ] : reflexive _⊑ᴮ_
  reflexive[⊑ᴮ] = record { xRx<_> = xRx[⊑ᴮ]<_> }

  antisymmetric[⊑ᴮ] : antisymmetric _⊑ᴮ_
  antisymmetric[⊑ᴮ] = record { _⋈_ = _⋈[⊑ᴮ]_ }

  transitive[⊑ᴮ] : transitive _⊑ᴮ_
  transitive[⊑ᴮ] = record { _⊚_ = _⊚[⊑ᴮ]_ }

  cor[⊑][𝔹] : cor[⊑] 𝔹
  cor[⊑][𝔹] = record {}

_⊑?ᴮ_ : (b₁ b₂ : 𝔹) → ⊑!
ᴏ ⊑?ᴮ b = [⊑]
ɪ ⊑?ᴮ ᴏ = [⋢]
ɪ ⊑?ᴮ ɪ = [⊑]

instance
  has[⊑?][𝔹] : has[⊑?] 𝔹
  has[⊑?][𝔹] = record { _⊑?_ = _⊑?ᴮ_ }

abstract
  _⊑⋆ᴮ_ : ∀ (b₁ b₂ : 𝔹) → b₁ ⊑⋆! b₂ 
  ᴏ ⊑⋆ᴮ b = [⊑] ᴏ
  ɪ ⊑⋆ᴮ ᴏ = [⋢] (λ ())
  ɪ ⊑⋆ᴮ ɪ = [⊑] ɪ

  _⊑∿ᴮ_ : ∀ (b₁ b₂ : 𝔹) → b₁ ⊑∿! b₂ [ b₁ ⊑? b₂ , b₁ ⊑⋆ᴮ b₂ ]
  ᴏ ⊑∿ᴮ b₂ = [⊑]
  ɪ ⊑∿ᴮ ᴏ = [⋢]
  ɪ ⊑∿ᴮ ɪ = [⊑]

instance
  cor[⊑?][𝔹] : cor[⊑?] 𝔹
  cor[⊑?][𝔹] = record { _⊑⋆_ = _⊑⋆ᴮ_ ; _⊑∿_ = _⊑∿ᴮ_ }

_ = ⌁ AT reflexive     _⊑ᴮ_
_ = ⌁ AT antisymmetric _⊑ᴮ_
_ = ⌁ AT transitive    _⊑ᴮ_
_ = ⌁ AT cor[⊑]        𝔹
_ = ⌁ AT has[⊑][ L0 ]  𝔹

_ = ⌁ AT cor[⊑?]       𝔹
_ = ⌁ AT has[⊑?]       𝔹
