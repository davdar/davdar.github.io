data _<ᴺ_ : (x y : ℕ) → Set where
  ᴢ : ∀ {x} → ᴢ <ᴺ ꜱ x
  ꜱ : ∀ {x y} (ε : x <ᴺ y) → ꜱ x <ᴺ ꜱ y

instance
  has[<][ℕ] : has[<][ L0 ] ℕ
  has[<][ℕ] = record { _<_ = _<ᴺ_ }

_⋚?ᴺ_ : ∀ (x y : ℕ) → ⋚!
ᴢ ⋚?ᴺ ᴢ = [≡]
ᴢ ⋚?ᴺ ꜱ y = [<]
ꜱ x ⋚?ᴺ ᴢ = [>]
ꜱ x ⋚?ᴺ ꜱ y = x ⋚?ᴺ y

instance
  has[⋚?][ℕ] : has[⋚?] ℕ
  has[⋚?][ℕ] = record { _⋚?_ = _⋚?ᴺ_ }

abstract
  isoto[<ᴺ] : ∀ {x y} → x <ᴺ y → ιᴺ x < ιᴺ y
  isoto[<ᴺ] ᴢ = []
  isoto[<ᴺ] (ꜱ ε) = [≡] ↯ (isoto[<ᴺ] ε)
  
  isofr[<ᴺ] : ∀ {x y} → ιᴺ x < ιᴺ y → x <ᴺ y
  isofr[<ᴺ] {ᴢ} {ᴢ} ()
  isofr[<ᴺ] {ᴢ} {ꜱ y} [] = ᴢ
  isofr[<ᴺ] {ꜱ x} {ᴢ} ()
  isofr[<ᴺ] {ꜱ x} {ꜱ y} ([<] ())
  isofr[<ᴺ] {ꜱ x} {ꜱ y} ([≡] ↯ ε) = ꜱ (isofr[<ᴺ] ε)
  
  corr[⋚?ᴺ]<_,_> : ∀ x y → (x ⋚?ᴺ y) ≡ (ιᴺ x ⋚? ιᴺ y)
  corr[⋚?ᴺ]< ᴢ , ᴢ > = ↯
  corr[⋚?ᴺ]< ᴢ , ꜱ y > = ↯
  corr[⋚?ᴺ]< ꜱ x , ᴢ > = ↯
  corr[⋚?ᴺ]< ꜱ x , ꜱ y > = corr[⋚?ᴺ]< x , y >
      
open derive[cor[<]] ℕ ⟬ 𝟙 ⟭ ιᴺ isoto[<ᴺ] isofr[<ᴺ]

instance
  irreflexive[<ᴺ] = irreflexive[<]†
  asymmetric[<ᴺ] = asymmetric[<]†
  transitive[<ᴺ] = transitive[<]†
  cor[<][ℕ] = cor[<]†

open derive[cor[⋚?]] ℕ ⟬ 𝟙 ⟭ ιᴺ isoto[<ᴺ] isofr[<ᴺ] corr[⋚?ᴺ]<_,_>

instance
  cor[⋚?][ℕ] = cor[⋚?]†

_ = ⌁ AT irreflexive    _<ᴺ_
_ = ⌁ AT asymmetric     _<ᴺ_
_ = ⌁ AT transitive     _<ᴺ_
_ = ⌁ AT has[<][ L0 ]   ℕ
_ = ⌁ AT cor[<]         ℕ
_ = ⌁ AT has[⋚?]        ℕ
_ = ⌁ AT cor[⋚?]        ℕ
