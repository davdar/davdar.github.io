module _ {ℓ ℓᴿ} {A : Set ℓ} {{_ : has[<][ ℓᴿ ] A}} where

  data _<ᴼ_ : (x y : ⦉ A ⦊) → Set ℓᴿ where
    • : ∀ {x} → • <ᴼ ⟨ x ⟩
    ⟨_⟩ : ∀ {x y} (ε : x < y) → ⟨ x ⟩ <ᴼ ⟨ y ⟩

  instance
    has[<][option] : has[<][ ℓᴿ ] ⦉ A ⦊
    has[<][option] = record { _<_ = _<ᴼ_ }

  abstract
    isoto[<ᴼ] : ∀ {x y} → x <ᴼ y → ιᴼ x < ιᴼ y
    isoto[<ᴼ] • = ʟʀ
    isoto[<ᴼ] ⟨ ε ⟩ = ʀ ε

    isofr[<ᴼ] : ∀ {x y} → ιᴼ x < ιᴼ y → x <ᴼ y
    isofr[<ᴼ] {•} {•} (ʟ ())
    isofr[<ᴼ] {•} {⟨ y ⟩} ʟʀ = •
    isofr[<ᴼ] {⟨ x ⟩} {•} ()
    isofr[<ᴼ] {⟨ x ⟩} {⟨ y ⟩} (ʀ ε) = ⟨ ε ⟩

  module _ {{_ : cor[<] A}} where

    open derive[cor[<]] ⦉ A ⦊ (𝟙 ∨ A) ιᴼ isoto[<ᴼ] isofr[<ᴼ]
    
    instance
      irreflexive[<ᴼ] = irreflexive[<]†
      asymmetric[<ᴼ] = asymmetric[<]†
      transitive[<ᴼ] = transitive[<]†
      cor[<][option] = cor[<]†
    
module _ {ℓ} {A : Set ℓ} {{_ : has[⋚?] A}} where

  _⋚?ᴼ_ : ∀ (x y : ⦉ A ⦊) → ⋚!
  • ⋚?ᴼ • = [≡]
  • ⋚?ᴼ ⟨ y ⟩ = [<]
  ⟨ x ⟩ ⋚?ᴼ • = [>]
  ⟨ x ⟩ ⋚?ᴼ ⟨ y ⟩ = x ⋚? y

  instance
    has[⋚?][option] : has[⋚?] ⦉ A ⦊
    has[⋚?][option] = record { _⋚?_ = _⋚?ᴼ_ }

  abstract
    corr[⋚?ᴼ]<_,_> : ∀ x y → (x ⋚?ᴼ y) ≡ (ιᴼ x ⋚? ιᴼ y)
    corr[⋚?ᴼ]< • , • > = ↯
    corr[⋚?ᴼ]< • , ⟨ y ⟩ > = ↯
    corr[⋚?ᴼ]< ⟨ x ⟩ , • > = ↯
    corr[⋚?ᴼ]< ⟨ x ⟩ , ⟨ y ⟩ > = ↯

  module _ {ℓᴿ} {{_ : has[<][ ℓᴿ ] A}} {{_ : cor[<] A}} {{_ : cor[⋚?] A}} where

    open derive[cor[⋚?]] ⦉ A ⦊ (𝟙 ∨ A) ιᴼ isoto[<ᴼ] isofr[<ᴼ] corr[⋚?ᴼ]<_,_>

    instance
      cor[⋚?][option] = cor[⋚?]†

module _ {ℓ ℓᴿ} {A : Set ℓ} {{_ : has[<][ ℓᴿ ] A}} where
  module _ {{_ : cor[<] A}} where
    _ = ⌁ AT irreflexive    (_<ᴼ_ AT (⦉ A ⦊ → ⦉ A ⦊ → Set ℓᴿ))
    _ = ⌁ AT asymmetric     (_<ᴼ_ AT (⦉ A ⦊ → ⦉ A ⦊ → Set ℓᴿ))
    _ = ⌁ AT transitive     (_<ᴼ_ AT (⦉ A ⦊ → ⦉ A ⦊ → Set ℓᴿ))
    _ = ⌁ AT cor[<]         ⦉ A ⦊
  _ = ⌁ AT has[<][ ℓᴿ ]  ⦉ A ⦊

module _ {ℓ} {A : Set ℓ} {{_ : has[⋚?] A}} where
  module _ {ℓᴿ} {{_ : has[<][ ℓᴿ ] A}} {{_ : cor[<] A}} {{_ : cor[⋚?] A}} where
    _ = ⌁ AT cor[⋚?] ⦉ A ⦊
  _ = ⌁ AT has[⋚?]  ⦉ A ⦊
