module UVMVS.Core.Data.Choice.IEq where

open import UVMVS.Core.Init
open import UVMVS.Core.Classes

module _ {ℓ₁ ℓ₂ ℓᴿ₁ ℓᴿ₂} {A : Set ℓ₁} {B : Set ℓ₂} {{_ : has[≍][ ℓᴿ₁ ] A}} {{_ : has[≍][ ℓᴿ₂ ] B}} where

  data _≍⸢∨⸣_ : (x y : A ∨ B) → Set (ℓᴿ₁ ⩛ ℓᴿ₂) where
    ʟ : ∀ {x y} (ε : x ≍ y) → ʟ x ≍⸢∨⸣ ʟ y
    ʀ : ∀ {x y} (ε : x ≍ y) → ʀ x ≍⸢∨⸣ ʀ y

  data _≭⸢∨⸣_ : (x y : A ∨ B) → Set (ℓᴿ₁ ⩛ ℓᴿ₂) where
    ʟʀ  : ∀ {x y} → ʟ x ≭⸢∨⸣ ʀ y
    ʀʟ  : ∀ {x y} → ʀ x ≭⸢∨⸣ ʟ y
    ʟ : ∀ {x y} (ε : x ≭ y) → ʟ x ≭⸢∨⸣ ʟ y
    ʀ : ∀ {x y} (ε : x ≭ y) → ʀ x ≭⸢∨⸣ ʀ y
  
  instance
    has[≍][∨] : has[≍][ ℓᴿ₁ ⩛ ℓᴿ₂ ] (A ∨ B)
    has[≍][∨] = record { _≍_ = _≍⸢∨⸣_ ; _≭_ = _≭⸢∨⸣_ }

  module _ {{_ : cor[≍] A}} {{_ : cor[≍] B}} where

    abstract
      sound[≍⸢∨⸣]<_,_> : ∀ (x y : A ∨ B) → x ≍⸢∨⸣ y → x ≡ y
      sound[≍⸢∨⸣]< ʟ x , ʟ y > (ʟ ε) rewrite sound[≍]< x , y > ε = ↯
      sound[≍⸢∨⸣]< ʀ x , ʀ y > (ʀ ε) rewrite sound[≍]< x , y > ε = ↯
      
      xRx[≍⸢∨⸣]<_> : ∀ (x : A ∨ B) → x ≍⸢∨⸣ x
      xRx[≍⸢∨⸣]< ʟ x > = ʟ xRx
      xRx[≍⸢∨⸣]< ʀ x > = ʀ xRx
      
      ¬xRx[≭⸢∨⸣] : ∀ {x : A ∨ B} → ¬ (x ≭⸢∨⸣ x)
      ¬xRx[≭⸢∨⸣] (ʟ ε) = ¬xRx ε
      ¬xRx[≭⸢∨⸣] (ʀ ε) = ¬xRx ε
      
      ◇[≭⸢∨⸣] : ∀ {x y} → x ≭⸢∨⸣ y → y ≭⸢∨⸣ x
      ◇[≭⸢∨⸣] ʟʀ = ʀʟ
      ◇[≭⸢∨⸣] ʀʟ = ʟʀ
      ◇[≭⸢∨⸣] (ʟ ε) = ʟ (◇ ε)
      ◇[≭⸢∨⸣] (ʀ ε) = ʀ (◇ ε)

    open make[≍] (A ∨ B) sound[≍⸢∨⸣]<_,_> xRx[≍⸢∨⸣]<_> ¬xRx[≭⸢∨⸣] ◇[≭⸢∨⸣]

    instance
      reflexive[≍⸢∨⸣] = reflexive[≍]†
      symmetric[≍⸢∨⸣] = symmetric[≍]†
      transitive[≍⸢∨⸣] = transitive[≍]†
      irreflexive[≭⸢∨⸣] = irreflexive[≭]†
      symmetric[≭⸢∨⸣] = symmetric[≭]†
      cor[≍][∨] = cor[≍]†
  
    _ = ⌁ AT reflexive         _≍⸢∨⸣_
    _ = ⌁ AT symmetric         _≍⸢∨⸣_
    _ = ⌁ AT transitive        _≍⸢∨⸣_
    _ = ⌁ AT irreflexive       _≭⸢∨⸣_
    _ = ⌁ AT symmetric         _≭⸢∨⸣_
    _ = ⌁ AT cor[≍]            (A ∨ B)
  _ = ⌁ AT has[≍][ ℓᴿ₁ ⩛ ℓᴿ₂ ] (A ∨ B)
