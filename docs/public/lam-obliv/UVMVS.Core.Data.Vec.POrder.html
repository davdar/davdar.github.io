module _ {ℓ ℓᴿ} {A : Set ℓ} {{_ : has[⊑][ ℓᴿ ] A}} where
  data _⊑ⱽ_ : ∀ {N} (xs ys : ⟬ A ⟭[ N ]) → Set (ℓ ⩛ ℓᴿ) where
    [] : [] ⊑ⱽ []
    _∷_ : ∀ {N} {xs ys : ⟬ A ⟭[ N ]} {x y} → x ⊑ y → xs ⊑ⱽ ys → (x ∷ xs) ⊑ⱽ (y ∷ ys)

  instance
    has[⊑][vec] : ∀ {N} → has[⊑][ ℓ ⩛ ℓᴿ ] ⟬ A ⟭[ N ]
    has[⊑][vec] = record { _⊑_ = _⊑ⱽ_}

  module _ {{_ : cor[⊑] A}} where
    abstract
      xRx[⊑ⱽ]<_> : ∀ {N} (xs : ⟬ A ⟭[ N ]) → xs ⊑ xs
      xRx[⊑ⱽ]< [] > = []
      xRx[⊑ⱽ]< x ∷ xs > = xRx ∷ xRx[⊑ⱽ]< xs >

      _⋈[⊑ⱽ]_ : ∀ {N} {xs ys : ⟬ A ⟭[ N ]} → xs ⊑ ys → ys ⊑ xs → xs ≡ ys
      [] ⋈[⊑ⱽ] [] = ↯
      (ε₁ ∷ εs₁) ⋈[⊑ⱽ] (ε₂ ∷ εs₂) rewrite ε₁ ⋈ ε₂ | εs₁ ⋈[⊑ⱽ] εs₂ = ↯

      _⊚[⊑ⱽ]_ : ∀ {N} {xs ys zs : ⟬ A ⟭[ N ]} → ys ⊑ zs → xs ⊑ ys → xs ⊑ zs
      [] ⊚[⊑ⱽ] [] = []
      (ε₁ ∷ εs₁) ⊚[⊑ⱽ] (ε₂ ∷ εs₂) = (ε₁ ⊚ ε₂) ∷ (εs₁ ⊚[⊑ⱽ] εs₂)

    instance
      reflexive[⊑ⱽ] : ∀ {N} → reflexive (_⊑ⱽ_ {N})
      reflexive[⊑ⱽ] = record { xRx<_> = xRx[⊑ⱽ]<_> }

      antisymmetric[⊑ⱽ] : ∀ {N} → antisymmetric (_⊑ⱽ_ {N})
      antisymmetric[⊑ⱽ] = record { _⋈_ = _⋈[⊑ⱽ]_ }

      transitive[⊑ⱽ] : ∀ {N} → transitive (_⊑ⱽ_ {N})
      transitive[⊑ⱽ] = record { _⊚_ = _⊚[⊑ⱽ]_ }

      cor[⊑][vec] : ∀ {N} → cor[⊑] ⟬ A ⟭[ N ]
      cor[⊑][vec] = record {}

module _ {ℓ} {A : Set ℓ} {{_ : has[⊑?] A}} where
  _⊑?ⱽ_ : ∀ {N} (xs ys : ⟬ A ⟭[ N ]) → ⊑!
  [] ⊑?ⱽ [] = [⊑]
  (x ∷ xs) ⊑?ⱽ (y ∷ ys) with x ⊑? y
  … | [⊑] = xs ⊑?ⱽ ys
  … | [⋢] = [⋢]

  instance
    has[⊑?][vec] : ∀ {N} → has[⊑?] ⟬ A ⟭[ N ]
    has[⊑?][vec] = record { _⊑?_ = _⊑?ⱽ_}

  module _ {ℓᴿ} {{_ : has[⊑][ ℓᴿ ] A}} {{_ : cor[⊑] A}} {{_ : cor[⊑?] A}} where
    abstract
      _⊑⋆ⱽ_ : ∀ {N} (xs ys : ⟬ A ⟭[ N ]) → xs ⊑⋆! ys
      [] ⊑⋆ⱽ [] = [⊑] []
      (x ∷ xs) ⊑⋆ⱽ (y ∷ ys) with x ⊑⋆ y
      … | [⋢] ε = [⋢] λ where (ε′ ∷ εs′) → ε ε′
      … | [⊑] ε with xs ⊑⋆ⱽ ys
      … | [⋢] ε′ = [⋢] λ where (ε″ ∷ εs′) → ε′ εs′
      … | [⊑] ε′ = [⊑] (ε ∷ ε′)

      _⊑∿ⱽ_ : ∀ {N} (xs ys : ⟬ A ⟭[ N ]) → xs ⊑∿! ys [ xs ⊑? ys , xs ⊑⋆ⱽ ys ]
      [] ⊑∿ⱽ [] = [⊑]
      (x ∷ xs) ⊑∿ⱽ (y ∷ ys) with x ⊑? y | x ⊑⋆ y | x ⊑∿ y
      … | [⋢] | [⋢] _ | [⋢] = [⋢]
      … | [⊑] | [⊑] _ | [⊑] with xs ⊑? ys | xs ⊑⋆ⱽ ys | xs ⊑∿ⱽ ys
      … | [⋢] | [⋢] _ | [⋢] = [⋢]
      … | [⊑] | [⊑] _ | [⊑] = [⊑]

    instance
      cor[⊑?][vec] : ∀ {N} → cor[⊑?] ⟬ A ⟭[ N ]
      cor[⊑?][vec] = record { _⊑⋆_ = _⊑⋆ⱽ_ ; _⊑∿_ = _⊑∿ⱽ_ }

module _ {ℓ ℓᴿ} {A : Set ℓ} {{_ : has[⊑][ ℓᴿ ] A}} {N : ℕ} where
  module _ {{_ : cor[⊑] A}} where
    _ = ⌁ AT reflexive      (_⊑ⱽ_ AT (⟬ A ⟭[ N ] → ⟬ A ⟭[ N ] → Set (ℓ ⩛ ℓᴿ)))
    _ = ⌁ AT antisymmetric  (_⊑ⱽ_ AT (⟬ A ⟭[ N ] → ⟬ A ⟭[ N ] → Set (ℓ ⩛ ℓᴿ)))
    _ = ⌁ AT transitive     (_⊑ⱽ_ AT (⟬ A ⟭[ N ] → ⟬ A ⟭[ N ] → Set (ℓ ⩛ ℓᴿ)))
    _ = ⌁ AT cor[⊑]         ⟬ A ⟭[ N ]
  _ = ⌁ AT has[⊑][ ℓ ⩛ ℓᴿ ] ⟬ A ⟭[ N ]

module _ {ℓ} {A : Set ℓ} {{_ : has[⊑?] A}} {N : ℕ} where
  module _ {ℓᴿ} {{_ : has[⊑][ ℓᴿ ] A}} {{_ : cor[⊑] A}} {{_ : cor[⊑?] A}} where
    _ = ⌁ AT cor[⊑?]  ⟬ A ⟭[ N ]
  _ = ⌁ AT has[⊑?]    ⟬ A ⟭[ N ]
