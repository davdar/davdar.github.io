module LamObliv.Common.ZContext where

open import UVMVS

open import LamObliv.Common.Syntax

module common-zcontext (R : ℕ) (val : ∀ (τ : type R) → Set) where
  open common-syntax R val

  -- closed zipper contexts ⸨ κ : τ⁺ ⇚ τ⁻ ⸩
  infix 10 _⇚_
  data _⇚_ (τ⁺ : type R) : ∀ (τ⁻ : type R) → Set where
    `⊤ : τ⁺ ⇚ τ⁺
    CastP[□] : ∀ {r}
      (κ : τ⁺ ⇚ Bit ⊥ P)
      → τ⁺ ⇚ Flip r
    MuxBit[□,e,e] : ∀ {r₁ r₂ r₃ ℓ₁ ℓ₂ ℓ₃}
      (e₂ : [] , [] ⊢ᴱ Bit r₂ ℓ₂ , [])
      (e₃ : [] , [] ⊢ᴱ Bit r₃ ℓ₃ , [])
      (κ : τ⁺ ⇚ Bit (r₁ ⊔ r₂ ⊔ r₃) (ℓ₁ ⊔ ℓ₂ ⊔ ℓ₃) `× Bit (r₁ ⊔ r₂ ⊔ r₃) (ℓ₁ ⊔ ℓ₂ ⊔ ℓ₃))
      → τ⁺ ⇚ Bit r₁ ℓ₁
    MuxBit[v,□,e] : ∀ {r₁ r₂ r₃ ℓ₁ ℓ₂ ℓ₃}
      (v₁ : val (Bit r₁ ℓ₁))
      (e₃ : [] , [] ⊢ᴱ Bit r₃ ℓ₃ , [])
      (κ : τ⁺ ⇚ Bit (r₁ ⊔ r₂ ⊔ r₃) (ℓ₁ ⊔ ℓ₂ ⊔ ℓ₃) `× Bit (r₁ ⊔ r₂ ⊔ r₃) (ℓ₁ ⊔ ℓ₂ ⊔ ℓ₃))
      → τ⁺ ⇚ Bit r₂ ℓ₂
    MuxBit[v,v,□] : ∀ {r₁ r₂ r₃ ℓ₁ ℓ₂ ℓ₃}
      (v₁ : val (Bit r₁ ℓ₁))
      (v₂ : val (Bit r₂ ℓ₂))
      (κ : τ⁺ ⇚ Bit (r₁ ⊔ r₂ ⊔ r₃) (ℓ₁ ⊔ ℓ₂ ⊔ ℓ₃) `× Bit (r₁ ⊔ r₂ ⊔ r₃) (ℓ₁ ⊔ ℓ₂ ⊔ ℓ₃))
      → τ⁺ ⇚ Bit r₃ ℓ₃
    MuxFlip[□,e,e] : ∀ {r₁ r₂ r₃ ℓ}
      (e₂ : [] , [] ⊢ᴱ Flip r₂ , [])
      (e₃ : [] , [] ⊢ᴱ Flip r₃ , [])
      (ε₁ : r₁ ⊓ r₂ ≡ ⊥)
      (ε₂ : r₁ ⊓ r₃ ≡ ⊥)
      (κ : τ⁺ ⇚ Flip (r₂ ⊔ r₃) `× Flip (r₂ ⊔ r₃))
      → τ⁺ ⇚ Bit r₁ ℓ
    MuxFlip[v,□,e] : ∀ {r₁ r₂ r₃ ℓ}
      (v₁ : val (Bit r₁ ℓ))
      (e₃ : [] , [] ⊢ᴱ Flip r₃ , [])
      (ε₁ : r₁ ⊓ r₂ ≡ ⊥)
      (ε₂ : r₁ ⊓ r₃ ≡ ⊥)
      (κ : τ⁺ ⇚ Flip (r₂ ⊔ r₃) `× Flip (r₂ ⊔ r₃))
      → τ⁺ ⇚ Flip r₂
    MuxFlip[v,v,□] : ∀ {r₁ r₂ r₃ ℓ}
      (v₁ : val (Bit r₁ ℓ))
      (v₂ : val (Flip r₂))
      (ε₁ : r₁ ⊓ r₂ ≡ ⊥)
      (ε₂ : r₁ ⊓ r₃ ≡ ⊥)
      (κ : τ⁺ ⇚ Flip (r₂ ⊔ r₃) `× Flip (r₂ ⊔ r₃))
      → τ⁺ ⇚ Flip r₃
    If[□,e,e] : ∀ {r τ}
      (e₂ : [] , [] ⊢ᴱ τ , [])
      (e₃ : [] , [] ⊢ᴱ τ , [])
      (κ : τ⁺ ⇚ τ)
      → τ⁺ ⇚ Bit r P
    Pair[□,e] : ∀ {τ₁ τ₂}
      (e₂ : [] , [] ⊢ᴱ τ₂ , [])
      (κ : τ⁺ ⇚ τ₁ `× τ₂)
      → τ⁺ ⇚ τ₁
    Pair[v,□] : ∀ {τ₁ τ₂}
      (v₁ : val τ₁)
      (κ : τ⁺ ⇚ τ₁ `× τ₂)
      → τ⁺ ⇚ τ₂
    Let[□,e] : ∀ {τ₁ τ₂ a}
      (e₂ : τ₁ ∷ [] , ᴏ ∷ [] ⊢ᴱ τ₂ , a ∷ [])
      (κ : τ⁺ ⇚ τ₂)
      → τ⁺ ⇚ τ₁
    LetPair[□,e] : ∀ {τ₁ τ₂ τ a₁ a₂}
      (e₂ : τ₁ ∷ τ₂ ∷ [] , ᴏ ∷ ᴏ ∷ [] ⊢ᴱ τ , a₁ ∷ a₂ ∷ [])
      (κ : τ⁺ ⇚ τ)
      → τ⁺ ⇚ τ₁ `× τ₂
    Ref[□] : ∀ {τ}
      (κ : τ⁺ ⇚ Ref τ)
      → τ⁺ ⇚ τ
    Read/Write[□,e] : ∀ {τ}
      (e₂ : [] , [] ⊢ᴱ Ref τ , [])
      (κ : τ⁺ ⇚ τ)
      → τ⁺ ⇚ τ
    Read/Write[v,□] : ∀ {τ}
      (v₁ : val τ)
      (κ : τ⁺ ⇚ τ)
      → τ⁺ ⇚ Ref τ

  -- composing contexts
  _⊚ᶻ_ : ∀ {τ₁ τ₂ τ₃} (c₂ : τ₃ ⇚ τ₂) (c₁ : τ₂ ⇚ τ₁) → τ₃ ⇚ τ₁
  κ′ ⊚ᶻ `⊤ = κ′
  κ′ ⊚ᶻ CastP[□] κ = CastP[□] (κ′ ⊚ᶻ κ)
  κ′ ⊚ᶻ MuxBit[□,e,e] e₂ e₃ κ = MuxBit[□,e,e] e₂ e₃ (κ′ ⊚ᶻ κ)
  κ′ ⊚ᶻ MuxBit[v,□,e] v₁ e₃ κ = MuxBit[v,□,e] v₁ e₃ (κ′ ⊚ᶻ κ)
  κ′ ⊚ᶻ MuxBit[v,v,□] v₁ v₂ κ = MuxBit[v,v,□] v₁ v₂ (κ′ ⊚ᶻ κ)
  κ′ ⊚ᶻ MuxFlip[□,e,e] e₂ e₃ ε₁ ε₂ κ = MuxFlip[□,e,e] e₂ e₃ ε₁ ε₂ (κ′ ⊚ᶻ κ)
  κ′ ⊚ᶻ MuxFlip[v,□,e] v₁ e₃ ε₁ ε₂ κ = MuxFlip[v,□,e] v₁ e₃ ε₁ ε₂ (κ′ ⊚ᶻ κ)
  κ′ ⊚ᶻ MuxFlip[v,v,□] v₁ v₂ ε₁ ε₂ κ = MuxFlip[v,v,□] v₁ v₂ ε₁ ε₂ (κ′ ⊚ᶻ κ)
  κ′ ⊚ᶻ If[□,e,e] e₂ e₃ κ = If[□,e,e] e₂ e₃ (κ′ ⊚ᶻ κ)
  κ′ ⊚ᶻ Pair[□,e] e₂ κ = Pair[□,e] e₂ (κ′ ⊚ᶻ κ)
  κ′ ⊚ᶻ Pair[v,□] v₁ κ = Pair[v,□] v₁ (κ′ ⊚ᶻ κ)
  κ′ ⊚ᶻ Let[□,e] e₂ κ = Let[□,e] e₂ (κ′ ⊚ᶻ κ)
  κ′ ⊚ᶻ LetPair[□,e] e₂ κ = LetPair[□,e] e₂ (κ′ ⊚ᶻ κ)
  κ′ ⊚ᶻ Ref[□] κ = Ref[□] (κ′ ⊚ᶻ κ)
  κ′ ⊚ᶻ Read/Write[□,e] e₂ κ = Read/Write[□,e] e₂ (κ′ ⊚ᶻ κ)
  κ′ ⊚ᶻ Read/Write[v,□] v₁ κ = Read/Write[v,□] v₁ (κ′ ⊚ᶻ κ)

  -- filling a context
  infixl  19 _⟁ᶻ_
  _⟁ᶻ_ : ∀ {τ⁻ τ⁺ : type R} (κ : τ⁺ ⇚ τ⁻) (e : [] , [] ⊢ᴱ τ⁻ , []) → [] , [] ⊢ᴱ τ⁺ , []
  `⊤ ⟁ᶻ e′ = e′
  CastP[□] κ ⟁ᶻ e′ = κ ⟁ᶻ CastP e′
  MuxBit[□,e,e] e₂ e₃ κ ⟁ᶻ e′ = κ ⟁ᶻ MuxBit e′ e₂ e₃
  MuxBit[v,□,e] v₁ e₃ κ ⟁ᶻ e′ = κ ⟁ᶻ MuxBit (Val v₁) e′ e₃
  MuxBit[v,v,□] v₁ v₂ κ ⟁ᶻ e′ = κ ⟁ᶻ MuxBit (Val v₁) (Val v₂) e′
  MuxFlip[□,e,e] e₂ e₃ ε₁ ε₂ κ ⟁ᶻ e′ = κ ⟁ᶻ MuxFlip e′ e₂ e₃ ε₁ ε₂
  MuxFlip[v,□,e] v₁ e₃ ε₁ ε₂ κ ⟁ᶻ e′ = κ ⟁ᶻ MuxFlip (Val v₁) e′ e₃ ε₁ ε₂
  MuxFlip[v,v,□] v₁ v₂ ε₁ ε₂ κ ⟁ᶻ e′ = κ ⟁ᶻ MuxFlip (Val v₁) (Val v₂) e′ ε₁ ε₂
  If[□,e,e] e₂ e₃ κ ⟁ᶻ e′ = κ ⟁ᶻ If e′ e₂ e₃ ↯
  Pair[□,e] e₂ κ ⟁ᶻ e′ = κ ⟁ᶻ Pair e′ e₂
  Pair[v,□] v₁ κ ⟁ᶻ e′ = κ ⟁ᶻ Pair (Val v₁) e′
  Let[□,e] e₂ κ ⟁ᶻ e′ = κ ⟁ᶻ Let e′ e₂
  LetPair[□,e] e₂ κ ⟁ᶻ e′ = κ ⟁ᶻ LetPair e′ e₂
  Ref[□] κ ⟁ᶻ e′ = κ ⟁ᶻ Ref e′
  Read/Write[□,e] e₂ κ ⟁ᶻ e′ = κ ⟁ᶻ Read/Write e′ e₂
  Read/Write[v,□] v₁ κ ⟁ᶻ e′ = κ ⟁ᶻ Read/Write (Val v₁) e′
