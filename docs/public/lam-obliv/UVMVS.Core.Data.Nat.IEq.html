module UVMVS.Core.Data.Nat.IEq where

open import UVMVS.Core.Init
open import UVMVS.Core.Classes
open import UVMVS.Core.Data.Unit
open import UVMVS.Core.Data.List

open import UVMVS.Core.Data.Nat.Injection

data _≍ᴺ_ : (n m : ℕ) → Set where
  ᴢ : ᴢ ≍ᴺ ᴢ
  ꜱ : ∀ {n m} (ε : n ≍ᴺ m) → ꜱ n ≍ᴺ ꜱ m

data _≭ᴺ_ : (n m : ℕ) → Set where
  ᴢꜱ : ∀ {n} → ᴢ ≭ᴺ ꜱ n
  ꜱᴢ : ∀ {n} → ꜱ n ≭ᴺ ᴢ
  ꜱ : ∀ {n m} (ε : n ≭ᴺ m) → ꜱ n ≭ᴺ ꜱ m

instance
  has[≍][ℕ] : has[≍][ L0 ] ℕ
  has[≍][ℕ] = record { _≍_ = _≍ᴺ_ ; _≭_ = _≭ᴺ_ }

abstract
  isoto[≍ᴺ] : ∀ {n m} → n ≍ᴺ m → ιᴺ n ≍ ιᴺ m
  isoto[≍ᴺ] ᴢ = []
  isoto[≍ᴺ] (ꜱ ε) = • ∷ isoto[≍ᴺ] ε
  
  isofr[≍ᴺ] : ∀ {n m} → ιᴺ n ≍ ιᴺ m → n ≍ᴺ m
  isofr[≍ᴺ] {ᴢ} {ᴢ} [] = ᴢ
  isofr[≍ᴺ] {ᴢ} {ꜱ m} ()
  isofr[≍ᴺ] {ꜱ n} {ᴢ} ()
  isofr[≍ᴺ] {ꜱ n} {ꜱ m} (ε ∷ εs) = ꜱ (isofr[≍ᴺ] εs)
  
  isoto[≭ᴺ] : ∀ {n m} → n ≭ᴺ m → ιᴺ n ≭ ιᴺ m
  isoto[≭ᴺ] ᴢꜱ = []≭[∷]
  isoto[≭ᴺ] ꜱᴢ = [∷]≭[]
  isoto[≭ᴺ] (ꜱ ε) = •∷ isoto[≭ᴺ] ε
  
  isofr[≭ᴺ] : ∀ {n m} → ιᴺ n ≭ ιᴺ m → n ≭ᴺ m
  isofr[≭ᴺ] {ᴢ} {ᴢ} ()
  isofr[≭ᴺ] {ᴢ} {ꜱ m} []≭[∷] = ᴢꜱ
  isofr[≭ᴺ] {ꜱ n} {ᴢ} [∷]≭[] = ꜱᴢ
  isofr[≭ᴺ] {ꜱ n} {ꜱ m} (() ∷•)
  isofr[≭ᴺ] {ꜱ n} {ꜱ m} (•∷ ε) = ꜱ (isofr[≭ᴺ] ε)

open derive[cor[≍]] ℕ ⟬ 𝟙 ⟭ ιᴺ isoto[≍ᴺ] isofr[≍ᴺ] isoto[≭ᴺ] isofr[≭ᴺ]

instance
  reflexive[≍ᴺ] = reflexive[≍]†
  symmetric[≍ᴺ] = symmetric[≍]†
  transitive[≍ᴺ] = transitive[≍]†
  irreflexive[≭ᴺ] = irreflexive[≭]†
  symmetric[≭ᴺ] = symmetric[≭]†
  cor[≍][ℕ] = cor[≍]†

_ = ⌁ AT reflexive    _≍ᴺ_
_ = ⌁ AT symmetric    _≍ᴺ_
_ = ⌁ AT transitive   _≍ᴺ_
_ = ⌁ AT irreflexive  _≭ᴺ_
_ = ⌁ AT symmetric    _≭ᴺ_
_ = ⌁ AT cor[≍]       ℕ   
_ = ⌁ AT has[≍][ L0 ] ℕ
