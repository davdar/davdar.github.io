module UVMVS.Lib.Data.Real where

open import UVMVS.Core

postulate
  ℝ : Set
  instance
    has[+][ℝ] : has[+] ℝ
    cor[+][ℝ] : cor[+] ℝ
    split[+][ℝ] : split[+] ℝ
    inj[+][ℝ] : inj[+] ℝ
    has[×][ℝ] : has[×] ℝ
    cor[×][ℝ] : cor[×] ℝ
    split[×][ℝ] : split[×] ℝ
    inj[×][ℝ] : inj[×] ℝ

  _-ᴿ_ : ℝ → ℝ → ℝ
  _/ᴿ_ : ℝ → ℝ → ℝ
  ㏒ᴿ : ℝ → ℝ
  expᴿ : ℝ → ℝ
