module _ {ℓ} {A : Set ℓ} {{_ : has[⊔] A}} where
  _⊔ⱽ_ : ∀ {N} (xs ys : ⟬ A ⟭[ N ]) → ⟬ A ⟭[ N ]
  [] ⊔ⱽ [] = []
  (x ∷ xs) ⊔ⱽ (y ∷ ys) = (x ⊔ y) ∷ (xs ⊔ⱽ ys)

  instance
    has[⊔][vec] : ∀ {N} → has[⊔] ⟬ A ⟭[ N ]
    has[⊔][vec] = record { _⊔_ = _⊔ⱽ_ }

  module _ {ℓᴿ} {{_ : has[⊑][ ℓᴿ ] A}} {{_ : cor[⊑] A}} {{_ : cor[⊔] A}} where
    abstract
      lbound[⊔ⱽ]<_,_> : ∀ {N} (xs ys : ⟬ A ⟭[ N ]) → xs ⊑ (xs ⊔ⱽ ys)
      lbound[⊔ⱽ]< [] , [] > = []
      lbound[⊔ⱽ]< x ∷ xs , y ∷ ys > = lbound[⊔]< x , y > ∷ lbound[⊔ⱽ]< xs , ys >

      rbound[⊔ⱽ]<_,_> : ∀ {N} (xs ys : ⟬ A ⟭[ N ]) → ys ⊑ (xs ⊔ⱽ ys)
      rbound[⊔ⱽ]< [] , [] > = []
      rbound[⊔ⱽ]< x ∷ xs , y ∷ ys > = rbound[⊔]< x , y > ∷ rbound[⊔ⱽ]< xs , ys >

      introd[⊔ⱽ]<_,_,_> : ∀ {N} (xs ys zs : ⟬ A ⟭[ N ]) → xs ⊑ zs → ys ⊑ zs → (xs ⊔ⱽ ys) ⊑ zs
      introd[⊔ⱽ]< [] , [] , [] > [] [] = []
      introd[⊔ⱽ]< x ∷ xs , y ∷ ys , z ∷ zs > (ε₁ ∷ εs₁) (ε₂ ∷ εs₂) = introd[⊔]< x , y , z > ε₁ ε₂ ∷ introd[⊔ⱽ]< xs , ys , zs > εs₁ εs₂

    instance
      cor[⊔][vec] : ∀ {N : ℕ} → cor[⊔] ⟬ A ⟭[ N ]
      cor[⊔][vec] = record
        { lbound[⊔]<_,_> = lbound[⊔ⱽ]<_,_>
        ; rbound[⊔]<_,_> = rbound[⊔ⱽ]<_,_>
        ; introd[⊔]<_,_,_> = introd[⊔ⱽ]<_,_,_>
        }

    _ = ⌁ AT ∀ {N} → cor[⊔] ⟬ A ⟭[ N ]

module _ {ℓ} {A : Set ℓ} {{_ : has[⊥] A}} where
  ⊥ⱽ[_] : ∀ N → ⟬ A ⟭[ N ]
  ⊥ⱽ[ ᴢ ] = []
  ⊥ⱽ[ ꜱ N ] = ⊥ ∷ ⊥ⱽ[ N ]

  instance
    has[⊥][vec] : ∀ {N} → has[⊥] ⟬ A ⟭[ N ]
    has[⊥][vec] = record { ⊥ = ⊥ⱽ[ _ ] }

  module _ {ℓᴿ} {{_ : has[⊑][ ℓᴿ ] A}} {{_ : cor[⊑] A}} {{_ : cor[⊥] A}} where
    abstract
      bound[⊥ⱽ]<_> : ∀ {N} (xs : ⟬ A ⟭[ N ]) → ⊥ⱽ[ N ] ⊑ xs
      bound[⊥ⱽ]< [] > = []
      bound[⊥ⱽ]< x ∷ xs > = bound[⊥]< x > ∷ bound[⊥ⱽ]< xs >

    instance
      cor[⊥][vec] : ∀ {N} → cor[⊥] ⟬ A ⟭[ N ]
      cor[⊥][vec] = record { bound[⊥]<_> = bound[⊥ⱽ]<_> }
