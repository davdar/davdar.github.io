module LamObliv.Common.Syntax where

open import UVMVS

-- security labels ⸨ ℓ : label ⸩
label : Set
label = 𝔹

pattern P = ᴏ
pattern S = ɪ

-- kinds ⸨ κ : kind ⸩
kind : Set
kind = 𝔹

pattern U = ᴏ
pattern L = ɪ

-- region variables ⸨ ρ : idx R ⸩
-- regions ⸨ r : region R ⸩
region : ∀ (R : ℕ) → Set
region R = ⟬ 𝔹 ⟭[ R ]

-- types ⸨ τ  : type R ⸩
data type (R : ℕ) : Set where
  Bit : ∀ (r : region R) (ℓ : label) → type R
  Flip : ∀ (r : region R) → type R
  _`×_ : ∀ (τ₁ : type R) (τ₂ : type R) → type R
  Ref : ∀ (τ : type R) → type R

-- kinding metafunction
𝒦 : ∀ {R} (τ : type R) → kind
𝒦 (Bit r ℓ) = U
𝒦 (Flip r) = L
𝒦 (τ₁ `× τ₂) = 𝒦 τ₁ ⩏ 𝒦 τ₂
𝒦 (Ref τ) = U

-- the resulting accessibility after a value at type with kind ⸨κ⸩ is consumed
consume : ∀ (κ : kind) → 𝔹
consume U = ᴏ
consume L = ɪ

-- typing context ⸨ Γ : tcxt ⸩
tcxt : ∀ (R : ℕ) (N : ℕ) → Set
tcxt R N = ⟬ type R ⟭[ N ]

-- accessibility context ⸨ A : acxt ⸩
acxt : ∀ (N : ℕ) → Set
acxt N = ⟬ 𝔹 ⟭[ N ]

-- typing store: ⸨ Σ : tsto R I ⸩
tsto : ∀ (R : ℕ) (I : ℕ) → Set
tsto R I = ⟬ type R ⟭[ I ]

module common-syntax (R : ℕ) (val : ∀ (τ : type R) → Set) where

  -- well typed expression ⸨ e : Γ , A ⊢ᴱ τ , A ⸩
  infix 10 _,_⊢ᴱ_,_
  data _,_⊢ᴱ_,_ : ∀ {N} (Γ : tcxt R N) (A : acxt N) (τ : type R) (A′ : acxt N) → Set where
    Var : ∀ {N} {Γ : tcxt R N} {A : acxt N} {A′ : acxt N} {τ : type R}
      {x : idx N}
      (ε₁ : ⟨ x ↦ τ ⟩∈ Γ)
      (ε₂ : ⟨ x ↦ ᴏ ⟩∈ A)
      (ε₃ : A′ ≡ A #[ x ↦ consume (𝒦 τ) ])
      → Γ , A ⊢ᴱ τ , A′
    Val : ∀ {N} {Γ : tcxt R N} {A : acxt N} {τ : type R}
      (v : val τ)
      → Γ , A ⊢ᴱ τ , A
    Bit : ∀ {N} {Γ : tcxt R N} {A : acxt N}
      (b : 𝔹)
      (r : region R)
      (ℓ : label)
      → Γ , A ⊢ᴱ Bit r ℓ , A
    Flip : ∀ {N} {Γ : tcxt R N} {A : acxt N}
      (ρ : idx R)
      (r : region R)
      → Γ , A ⊢ᴱ Flip (r #[ ρ ↦ ɪ ]) , A
    CastS : ∀ {N} {Γ : tcxt R N} {A A′ : acxt N} {r : region R}
      {x : idx N}
      (ε₁ : ⟨ x ↦ Flip r ⟩∈ Γ)
      (ε₂ : ⟨ x ↦ ᴏ ⟩∈ A)
      (ε₃ : A′ ≡ A #[ x ↦ ᴏ ])
      → Γ , A ⊢ᴱ Bit r S , A′
    CastS-V : ∀ {N} {Γ : tcxt R N} {A : acxt N} {r : region R}
      (v : val (Flip r))
      → Γ , A ⊢ᴱ Bit r S , A
    CastP : ∀ {N} {Γ : tcxt R N} {A A′ : acxt N} {r : region R}
      (e : Γ , A ⊢ᴱ Flip r , A′)
      → Γ , A ⊢ᴱ Bit ⊥ P , A′
    MuxBit : ∀ {N} {Γ : tcxt R N} {A A′ A″ A‴ : acxt N} {r₁ r₂ r₃ : region R} {ℓ₁ ℓ₂ ℓ₃ : label}
      (e₁ : Γ , A ⊢ᴱ Bit r₁ ℓ₁ , A′)
      (e₂ : Γ , A′ ⊢ᴱ Bit r₂ ℓ₂ , A″)
      (e₃ : Γ , A″ ⊢ᴱ Bit r₃ ℓ₃ , A‴)
      → Γ  , A ⊢ᴱ Bit (r₁ ⊔ r₂ ⊔ r₃) (ℓ₁ ⊔ ℓ₂ ⊔ ℓ₃) `× Bit (r₁ ⊔ r₂ ⊔ r₃) (ℓ₁ ⊔ ℓ₂ ⊔ ℓ₃) , A‴
    MuxFlip : ∀ {N} {Γ : tcxt R N} {A A′ A″ A‴ : acxt N} {r₁ r₂ r₃ : region R} {ℓ : label}
      (e₁ : Γ , A ⊢ᴱ Bit r₁ ℓ , A′)
      (e₂ : Γ , A′ ⊢ᴱ Flip r₂ , A″)
      (e₃ : Γ , A″ ⊢ᴱ Flip r₃ , A‴)
      (ε₁ : r₁ ⊓ r₂ ≡ ⊥)
      (ε₂ : r₁ ⊓ r₃ ≡ ⊥)
      → Γ  , A ⊢ᴱ Flip (r₂ ⊔ r₃) `× Flip (r₂ ⊔ r₃) , A‴
    If : ∀ {N} {Γ : tcxt R N} {A A′ A″ A″₁ A″₂ : acxt N} {τ : type R} {r}
      (e₁ : Γ , A ⊢ᴱ Bit r P , A′)
      (e₂ : Γ , A′ ⊢ᴱ τ , A″₁)
      (e₃ : Γ , A′ ⊢ᴱ τ , A″₂)
      (ε  : A″ ≡ A″₁ ⊔ A″₂)
      → Γ , A ⊢ᴱ τ , A″
    Pair : ∀ {N} {Γ : tcxt R N} {A A′ A″ : acxt N} {τ₁ τ₂ : type R}
      (e₁ : Γ , A ⊢ᴱ τ₁ , A′)
      (e₂ : Γ , A′ ⊢ᴱ τ₂ , A″)
      → Γ , A ⊢ᴱ τ₁ `× τ₂ , A″
    Let : ∀ {N} {Γ : tcxt R N} {A A′ A″ : acxt N} {τ₁ τ₂ : type R} {a : 𝔹}
      (e₁ : Γ , A ⊢ᴱ τ₁ , A′)
      (e₂ : τ₁ ∷ Γ , ᴏ ∷ A′ ⊢ᴱ τ₂ , a ∷ A″)
      → Γ , A ⊢ᴱ τ₂ , A″
    LetPair : ∀ {N} {Γ : tcxt R N} {A A′ A″ : acxt N} {τ₁ τ₂ τ : type R} {a₁ a₂}
      (e₁ : Γ , A ⊢ᴱ τ₁ `× τ₂ , A′)
      (e₂ : τ₁ ∷ τ₂ ∷ Γ , ᴏ ∷ ᴏ ∷ A′ ⊢ᴱ τ , a₁ ∷ a₂ ∷ A″)
      → Γ , A ⊢ᴱ τ , A″
    Ref : ∀ {N} {Γ : tcxt R N} {A A′ : acxt N} {τ : type R}
      (e : Γ , A ⊢ᴱ τ , A′)
      → Γ , A ⊢ᴱ Ref τ , A′
    Read/Write : ∀ {N} {Γ : tcxt R N} {A A′ A″ : acxt N} {τ : type R}
      (e₁ : Γ , A ⊢ᴱ τ , A′)
      (e₂ : Γ , A′ ⊢ᴱ Ref τ , A″)
      → Γ , A ⊢ᴱ τ , A″
