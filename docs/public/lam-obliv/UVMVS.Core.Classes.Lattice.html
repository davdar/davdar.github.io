    bound[⊥]<_> : ∀ (x : A) → ⊥ ⊑ x

record has[⊔] {ℓ} (A : Set ℓ) : Set ℓ where
  infixl 15 _⊔_
  field
    _⊔_ : ∀ (x y : A) → A
open has[⊔] {{…}} public

record cor[⊔] {ℓ ℓᴿ} (A : Set ℓ) {{_ : has[⊑][ ℓᴿ ] A}} {{_ : has[⊔] A}} : Set (ℓ ⩛ ↟ ℓᴿ) where
  field
    lbound[⊔]<_,_> : ∀ (x y : A) → x ⊑ (x ⊔ y)
    rbound[⊔]<_,_> : ∀ (x y : A) → y ⊑ (x ⊔ y)
    introd[⊔]<_,_,_> : ∀ (x y z : A) → x ⊑ z → y ⊑ z → (x ⊔ y) ⊑ z

  lbound[⊔] = λ {x} {y} → lbound[⊔]< x , y >
  rbound[⊔] = λ {x} {y} → rbound[⊔]< x , y >
  introd[⊔] = λ {x} {y} {z} → introd[⊔]< x , y , z >

  module _ {{_ : cor[⊑] A}} where
    is⊑[⊔]<_,_> : ∀ (x y : A) → x ⊑ y → x ⊔ y ≡ y
    is⊑[⊔]< x , y > ε = introd[⊔] ε xRx ⋈ rbound[⊔]

    is⊒[⊔]<_,_> : ∀ (x y : A) → y ⊑ x → x ⊔ y ≡ x
    is⊒[⊔]< x , y > ε = introd[⊔] xRx ε ⋈ lbound[⊔]

    if⊑[⊔]<_,_> : ∀ (x y : A) → x ⊔ y ≡ y → x ⊑ y
    if⊑[⊔]< x , y > ε with lbound[⊔]< x , y >
    … | ε′ rewrite ε = ε′

    if⊒[⊔]<_,_> : ∀ (x y : A) → x ⊔ y ≡ x → y ⊑ x
    if⊒[⊔]< x , y > ε with rbound[⊔]< x , y >
    … | ε′ rewrite ε = ε′

    idemp[⊔]<_> : ∀ (x : A) → x ⊔ x ≡ x
    idemp[⊔]< x > = introd[⊔] xRx xRx ⋈ lbound[⊔]

    commu[⊔]<_,_> : ∀ (x y : A) → x ⊔ y ≡ y ⊔ x
    commu[⊔]< x , y > = introd[⊔] rbound[⊔] lbound[⊔] ⋈ introd[⊔] rbound[⊔] lbound[⊔]

    assoc[⊔]<_,_,_> : ∀ (x y z : A) → x ⊔ (y ⊔ z) ≡ (x ⊔ y) ⊔ z
    assoc[⊔]< x , y , z > =
        introd[⊔] (lbound[⊔] ⊚ lbound[⊔]) (introd[⊔] (lbound[⊔] ⊚ rbound[⊔]) rbound[⊔])
      ⋈ introd[⊔] (introd[⊔] lbound[⊔] (rbound[⊔] ⊚ lbound[⊔])) (rbound[⊔] ⊚ rbound[⊔])
open cor[⊔] {{…}} public

record has[⊓] {ℓ} (A : Set ℓ) : Set ℓ where
  infixl 16 _⊓_
  field
    _⊓_ : ∀ (x y : A) → A
open has[⊓] {{…}} public

record cor[⊓] {ℓ ℓᴿ} (A : Set ℓ) {{_ : has[⊑][ ℓᴿ ] A}} {{_ : has[⊓] A}} : Set (ℓ ⩛ ↟ ℓᴿ) where
  field
    lbound[⊓]<_,_> : ∀ (x y : A) → (x ⊓ y) ⊑ x
    rbound[⊓]<_,_> : ∀ (x y : A) → (x ⊓ y) ⊑ y
    introd[⊓]<_,_,_> : ∀ (x y z : A) → x ⊑ y → x ⊑ z → x ⊑ (y ⊓ z)

  lbound[⊓] = λ {x} {y} → lbound[⊓]< x , y >
  rbound[⊓] = λ {x} {y} → rbound[⊓]< x , y >
  introd[⊓] = λ {x} {y} {z} → introd[⊓]< x , y , z >
  
  module _ {{_ : cor[⊑] A}} where
    is⊑[⊓]<_,_> : ∀ (x y : A) → x ⊑ y → x ⊓ y ≡ x
    is⊑[⊓]< _ , _ > ε = lbound[⊓] ⋈ introd[⊓] xRx ε
 
    is⊒[⊓]<_,_> : ∀ (x y : A) → y ⊑ x → x ⊓ y ≡ y
    is⊒[⊓]< _ , _ > ε = rbound[⊓] ⋈ introd[⊓] ε xRx
 
    idemp[⊓]<_> : ∀ (x : A) → x ⊓ x ≡ x
    idemp[⊓]< x > = lbound[⊓] ⋈ introd[⊓] xRx xRx

    commu[⊓]<_,_> : ∀ (x y : A) → x ⊓ y ≡ y ⊓ x
    commu[⊓]< x , y > = introd[⊓] rbound[⊓] lbound[⊓] ⋈ introd[⊓] rbound[⊓] lbound[⊓]

    assoc[⊓]<_,_,_> : ∀ (x y z : A) → x ⊓ (y ⊓ z) ≡ (x ⊓ y) ⊓ z
    assoc[⊓]< x , y , z > =
        introd[⊓] (introd[⊓] lbound[⊓] (lbound[⊓] ⊚ rbound[⊓])) (rbound[⊓] ⊚ rbound[⊓])
      ⋈ introd[⊓] (lbound[⊓] ⊚ lbound[⊓]) (introd[⊓] (rbound[⊓] ⊚ lbound[⊓]) rbound[⊓])
open cor[⊓] {{…}} public

record dist[⊓] {ℓ} (A : Set ℓ) {{_ : has[⊔] A}} {{_ : has[⊓] A}} : Set ℓ where
  field
    ldist[⊓]<_,_,_> : ∀ (x y z : A) → x ⊓ (y ⊔ z) ≡ x ⊓ y ⊔ x ⊓ z
    rdist[⊓]<_,_,_> : ∀ (x y z : A) → (x ⊔ y) ⊔ z ≡ x ⊓ z ⊔ y ⊓ z
open dist[⊓] {{…}} public

module _ {ℓ ℓᴿ} {A : Set ℓ}
  {{_ : has[⊑][ ℓᴿ ] A}} {{_ : cor[⊑] A}}
  {{_ : has[⊔] A}} {{_ : cor[⊔] A}}
  {{_ : has[⊥] A}} {{_ : cor[⊥] A}}
  where

  lunit[⊔]<_> : ∀ (x : A) → ⊥ ⊔ x ≡ x
  lunit[⊔]< x > = introd[⊔]< ⊥ , x , x > (bound[⊥]< x >) xRx ⋈ rbound[⊔]< ⊥ , x >

  runit[⊔]<_> : ∀ (x : A) → x ⊔ ⊥ ≡ x
  runit[⊔]< x > rewrite commu[⊔]< x , ⊥ > | lunit[⊔]< x > = ↯

module _ {ℓ ℓᴿ} {A : Set ℓ}
  {{_ : has[⊑][ ℓᴿ ] A}} {{_ : cor[⊑] A}}
  {{_ : has[⊔] A}} {{_ : cor[⊔] A}}
  {{_ : has[⊤] A}} {{_ : cor[⊤] A}}
  where

  lzero[⊔]<_> : ∀ (x : A) → ⊤ ⊔ x ≡ ⊤
  lzero[⊔]< x > = bound[⊤]< ⊤ ⊔ x > ⋈ lbound[⊔]< ⊤ , x >

  rzero[⊔]<_> : ∀ (x : A) → x ⊔ ⊤ ≡ ⊤
  rzero[⊔]< x > rewrite commu[⊔]< x , ⊤ > | lzero[⊔]< x > = ↯

module derive[cor[⊔]]
  {ℓ₁ ℓ₂ ℓᴿ₁ ℓᴿ₂}
  (A : Set ℓ₁) (B : Set ℓ₂) (ι : A → B)
  {{_ : has[⊑][ ℓᴿ₁ ] A}} {{_ : cor[⊑] A}}
  {{_ : has[⊔] A}} 
  {{_ : has[⊑][ ℓᴿ₂ ] B}} {{_ : cor[⊑] B}}
  {{_ : has[⊔] B}} {{_ : cor[⊔] B}}
  {{_ : injective ι}}
  (isoto[⊑] : ∀ {x y} → x ⊑ y → ι x ⊑ ι y)
  (isofr[⊑] : ∀ {x y} → ι x ⊑ ι y → x ⊑ y)
  (corr[⊔]<_,_> : ∀ x y → ι (x ⊔ y) ≡ ι x ⊔ ι y)
  where

  abstract
    lbound[⊔]†<_,_> : ∀ (x y : A) → x ⊑ x ⊔ y
    lbound[⊔]†< x , y > with lbound[⊔]< ι x , ι y >
    … | H rewrite ◇⸢≡⸣ corr[⊔]< x , y > = isofr[⊑] H

    rbound[⊔]†<_,_> : ∀ (x y : A) → y ⊑ x ⊔ y
    rbound[⊔]†< x , y > with rbound[⊔]< ι x , ι y >
    … | H rewrite ◇⸢≡⸣ corr[⊔]< x , y > = isofr[⊑] H

    introd[⊔]†<_,_,_> : ∀ (x y z : A) → x ⊑ z → y ⊑ z → x ⊔ y ⊑ z
    introd[⊔]†< x , y , z > ε₁ ε₂ with introd[⊔]< ι x , ι y , ι z >
    … | H rewrite ◇⸢≡⸣ corr[⊔]< x , y > = isofr[⊑] (H (isoto[⊑] ε₁) (isoto[⊑] ε₂))

  cor[⊔]† : cor[⊔] A
  cor[⊔]† = record
    { lbound[⊔]<_,_> = lbound[⊔]†<_,_>
    ; rbound[⊔]<_,_> = rbound[⊔]†<_,_>
    ; introd[⊔]<_,_,_> = introd[⊔]†<_,_,_>
    }

module derive[cor[⊓]]
  {ℓ₁ ℓ₂ ℓᴿ₁ ℓᴿ₂}
  (A : Set ℓ₁) (B : Set ℓ₂) (ι : A → B)
  {{_ : has[⊑][ ℓᴿ₁ ] A}} {{_ : cor[⊑] A}}
  {{_ : has[⊓] A}}
  {{_ : has[⊑][ ℓᴿ₂ ] B}} {{_ : cor[⊑] B}}
  {{_ : has[⊓] B}} {{_ : cor[⊓] B}}
  {{_ : injective ι}}
  (isoto[⊑] : ∀ {x y} → x ⊑ y → ι x ⊑ ι y)
  (isofr[⊑] : ∀ {x y} → ι x ⊑ ι y → x ⊑ y)
  (corr[⊓]<_,_> : ∀ x y → ι (x ⊓ y) ≡ ι x ⊓ ι y)
  where

  abstract
    lbound[⊓]†<_,_> : ∀ (x y : A) → x ⊓ y ⊑ x
    lbound[⊓]†< x , y > with lbound[⊓]< ι x , ι y >
    … | H rewrite ◇⸢≡⸣ corr[⊓]< x , y > = isofr[⊑] H

    rbound[⊓]†<_,_> : ∀ (x y : A) → x ⊓ y ⊑ y
    rbound[⊓]†< x , y > with rbound[⊓]< ι x , ι y >
    … | H rewrite ◇⸢≡⸣ corr[⊓]< x , y > = isofr[⊑] H

    introd[⊓]†<_,_,_> : ∀ (x y z : A) → x ⊑ y → x ⊑ z → x ⊑ y ⊓ z
    introd[⊓]†< x , y , z > ε₁ ε₂ with introd[⊓]< ι x , ι y , ι z >
    … | H rewrite ◇⸢≡⸣ corr[⊓]< y , z > = isofr[⊑] (H (isoto[⊑] ε₁) (isoto[⊑] ε₂))

  cor[⊓]† : cor[⊓] A
  cor[⊓]† = record
    { lbound[⊓]<_,_> = lbound[⊓]†<_,_>
    ; rbound[⊓]<_,_> = rbound[⊓]†<_,_>
    ; introd[⊓]<_,_,_> = introd[⊓]†<_,_,_>
    }
