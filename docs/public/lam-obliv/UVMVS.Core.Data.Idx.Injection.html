module UVMVS.Core.Data.Idx.Injection where

open import UVMVS.Core.Init
open import UVMVS.Core.Classes
open import UVMVS.Core.Data.Nat

open import UVMVS.Core.Data.Idx.Base

ιᴵ : ∀ {N} (x : idx N) → ℕ
ιᴵ ᴢ = ᴢ
ιᴵ (ꜱ x) = ꜱ (ιᴵ x)

abstract
  inj[ιᴵ]<_,_> : ∀ {N} (x y : idx N) → ιᴵ x ≡ ιᴵ y → x ≡ y
  inj[ιᴵ]< ᴢ , ᴢ > ↯ = ↯
  inj[ιᴵ]< ᴢ , ꜱ y > ()
  inj[ιᴵ]< ꜱ x , ᴢ > ()
  inj[ιᴵ]< ꜱ x , ꜱ y > ε with ιᴵ x | ιᴵ y | inj[ιᴵ]< x , y >
  inj[ιᴵ]< ꜱ x , ꜱ y > ↯ | fx | .fx | ε rewrite ε ↯ = ↯ 

module _ {N : ℕ} where
  instance
    injective[ιᴵ] : injective (ιᴵ {N})
    injective[ιᴵ] = record { inj<_,_> = inj[ιᴵ]<_,_> }
