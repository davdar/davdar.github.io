module UVMVS.Core.Data.Unit.IEq where

open import UVMVS.Core.Init
open import UVMVS.Core.Classes

data _≍¹_ : (x y : 𝟙) → Set where
  • : • ≍¹ •

data _≭¹_ (x y : 𝟙) : Set where

instance
  has[≍][𝟙] : has[≍][ L0 ] 𝟙
  has[≍][𝟙] = record { _≍_ = _≍¹_ ; _≭_ = _≭¹_ }

abstract
  sound[≍¹]<_,_> : ∀ (x y : 𝟙) → x ≍¹ y → x ≡ y
  sound[≍¹]< • , • > • = ↯
  
  xRx[≍¹]<_> : ∀ (x : 𝟙) → x ≍¹ x
  xRx[≍¹]< • > = •
  
  ¬xRx[≭¹] : ∀ {x : 𝟙} → ¬ (x ≭¹ x)
  ¬xRx[≭¹] ()
  
  ◇[≭¹] : ∀ {x y} → x ≭¹ y → y ≭¹ x
  ◇[≭¹] ()

open make[≍] 𝟙 sound[≍¹]<_,_> xRx[≍¹]<_> ¬xRx[≭¹] ◇[≭¹]

instance
  reflexive[≍¹] = reflexive[≍]†
  symmetric[≍¹] = symmetric[≍]†
  transitive[≍¹] = transitive[≍]†
  irreflexive[≭¹] = irreflexive[≭]†
  symmetric[≭¹] = symmetric[≭]†
  cor[≍][𝟙] = cor[≍]†

_ = ⌁ AT has[≍][ L0 ] 𝟙
_ = ⌁ AT reflexive    _≍¹_
_ = ⌁ AT symmetric    _≍¹_
_ = ⌁ AT transitive   _≍¹_
_ = ⌁ AT irreflexive  _≭¹_
_ = ⌁ AT symmetric    _≭¹_
_ = ⌁ AT cor[≍]       𝟙
