module _ {ℓ} {A : Set ℓ} where
  infix 10 _∈_
  data _∈_ (x : A) : ∀ (xs : ⟬ A ⟭) → Set ℓ where
    ᴢ : ∀ {xs} → x ∈ x ∷ xs
    ꜱ : ∀ {x′ xs} → x ∈ xs → x ∈ x′ ∷ xs
  
  infix 10 _∉_
  data _∉_ (x : A) : ∀ (xs : ⟬ A ⟭) → Set ℓ where
    [] : x ∉ []
    _∷_ : ∀ {x′ xs}
      → x ≢ x′
      → x ∉ xs
      → x ∉ x′ ∷ xs

  ∣_∣ : ∀ (xs : ⟬ A ⟭) → ℕ
  ∣ [] ∣ = ᴢ
  ∣ x ∷ xs ∣ = ꜱ ∣ xs ∣

  first : ⟬ A ⟭ → ⦉ A ⦊
  first [] = •
  first (x ∷ _) = ⟨ x ⟩

module _ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : Set ℓ₂} where
  fold : ∀ (f : A → B → B) (xs : ⟬ A ⟭) (i : B) → B
  fold f [] i = i
  fold f (x ∷ xs) i = fold f xs (f x i)

  foldr : ∀ (f : A → B → B) (xs : ⟬ A ⟭) (i : B) → B
  foldr f [] i = i
  foldr f (x ∷ xs) i = f x (fold f xs i)

module _ {ℓ} {A : Set ℓ} {{_ : has[⊔] A}} {{_ : has[⊥] A}} where
  joins : ∀ (xs : ⟬ A ⟭) → A
  joins xs = fold _⊔_ xs ⊥

module _ {ℓ} {A : Set ℓ} {{_ : has[⊓] A}} {{_ : has[⊤] A}} where
  meets : ∀ (xs : ⟬ A ⟭) → A
  meets xs = fold _⊓_ xs ⊤

mapᴸ : ∀ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : Set ℓ₂} (f : A → B) (xs : ⟬ A ⟭) → ⟬ B ⟭
mapᴸ f [] = []
mapᴸ f (x ∷ xs) = f x ∷ mapᴸ f xs
  
module _ {ℓ : level} where
  instance
    has[map][list] : has[map] (⟬_⟭ {ℓ})
    has[map][list] = record { map = mapᴸ }

  abstract
    unit[mapᴸ]<_> : ∀ {A : Set ℓ} (xs : ⟬ A ⟭) → map id xs ≡ xs
    unit[mapᴸ]< [] > = ↯
    unit[mapᴸ]< x ∷ xs > rewrite unit[mapᴸ]< xs > = ↯

    homom[mapᴸ]<_,_,_> : ∀ {A B C : Set ℓ} (g : B → C) (f : A → B) (xs : ⟬ A ⟭) → map (g ∘ f) xs ≡ map g (map f xs)
    homom[mapᴸ]< g , f , [] > = ↯
    homom[mapᴸ]< g , f , x ∷ xs > rewrite homom[mapᴸ]< g , f , xs > = ↯

  instance
    cor[map][list] : cor[map] ⟬_⟭
    cor[map][list] = record
      { unit[map]<_> = unit[mapᴸ]<_>
      ; homom[map]<_,_,_> = homom[mapᴸ]<_,_,_>
      }

module _ {ℓ} {A : Set ℓ} where
  _⧺ᴸ_ : ∀ (xs ys : ⟬ A ⟭) → ⟬ A ⟭
  [] ⧺ᴸ ys = ys
  (x ∷ xs) ⧺ᴸ ys = x ∷ (xs ⧺ᴸ ys)
  
  instance
    has[⧺][list] : has[⧺] ⟬ A ⟭
    has[⧺][list] = record { null = [] ; _⧺_ = _⧺ᴸ_ }

  abstract
    lunit[⧺ᴸ]<_> : ∀ (xs : ⟬ A ⟭) → null ⧺ xs ≡ xs
    lunit[⧺ᴸ]< n > = ↯
    
    runit[⧺ᴸ]<_> : ∀ (xs : ⟬ A ⟭) → xs ⧺ null ≡ xs
    runit[⧺ᴸ]< [] > = ↯
    runit[⧺ᴸ]< x ∷ xs > rewrite runit[⧺ᴸ]< xs > = ↯
    
    assoc[⧺ᴸ]<_,_,_> : ∀ (xs ys zs : ⟬ A ⟭) → xs ⧺ (ys ⧺ zs) ≡ (xs ⧺ ys) ⧺ zs
    assoc[⧺ᴸ]< [] , ys , zs > = ↯
    assoc[⧺ᴸ]< x ∷ xs , ys , zs > rewrite assoc[⧺ᴸ]< xs , ys , zs > = ↯

  instance
    cor[⧺][list] : cor[⧺] ⟬ A ⟭
    cor[⧺][list] = record
      { lunit[⧺]<_> = lunit[⧺ᴸ]<_>
      ; runit[⧺]<_> = runit[⧺ᴸ]<_>
      ; assoc[⧺]<_,_,_> = assoc[⧺ᴸ]<_,_,_>
      }

returnᴸ : ∀ {ℓ} {A : Set ℓ} → A → ⟬ A ⟭
returnᴸ x = x ∷ []

_≫=ᴸ_ : ∀ {ℓ} {A B : Set ℓ} → ⟬ A ⟭ → (A → ⟬ B ⟭) → ⟬ B ⟭
[] ≫=ᴸ f = []
(x ∷ xs) ≫=ᴸ f = f x ⧺ (xs ≫=ᴸ f)

instance
  has[≫=][list] : ∀ {ℓ} → has[≫=]  (⟬_⟭ {ℓ})
  has[≫=][list] = record { return = returnᴸ ; _≫=_ = _≫=ᴸ_ }

abstract
   ext[≫=ᴸ]<_,_,_> : ∀ {ℓ} {A B :  Set ℓ} (xs : ⟬ A ⟭) (f g : A → ⟬ B ⟭) → (∀ x → f x ≡ g x) → (xs ≫= f) ≡ (xs ≫= g)
   ext[≫=ᴸ]< [] , f , g > ε = ↯
   ext[≫=ᴸ]< x ∷ xs , f , g > ε rewrite ε x | ext[≫=ᴸ]< xs , f , g > ε = ↯

mzeroᴸ : ∀ {ℓ} {A : Set ℓ} → ⟬ A ⟭
mzeroᴸ = []

_⊞ᴸ_ : ∀ {ℓ} {A : Set ℓ} → ⟬ A ⟭ → ⟬ A ⟭ → ⟬ A ⟭
_⊞ᴸ_ = _⧺_

instance
  has[⊞][list] : ∀ {ℓ} → has[⊞] (⟬_⟭ {ℓ})
  has[⊞][list] = record { mzero = mzeroᴸ ; _⊞_ = _⊞ᴸ_ }

abstract
  lunit[⊞ᴸ]<_> : ∀ {ℓ} {A : Set ℓ} (xs : ⟬ A ⟭) → mzero ⊞ xs ≡ xs
  lunit[⊞ᴸ]<_> = lunit[⧺]<_>
  
  runit[⊞ᴸ]<_> : ∀ {ℓ} {A : Set ℓ} (xs : ⟬ A ⟭) → xs ⊞ mzero ≡ xs
  runit[⊞ᴸ]<_> = runit[⧺]<_>
  
  assoc[⊞ᴸ]<_,_,_> : ∀ {ℓ} {A : Set ℓ} (xs ys zs : ⟬ A ⟭) → xs ⊞ (ys ⊞ zs) ≡ (xs ⊞ ys) ⊞ zs
  assoc[⊞ᴸ]<_,_,_> = assoc[⧺]<_,_,_>
  
  lzero[≫=ᴸ]<_> : ∀ {ℓ} {A B : Set ℓ} (f : A → ⟬ B ⟭) → (mzero ≫= f) ≡ mzero
  lzero[≫=ᴸ]< f > = ↯
  
  rzero[≫=ᴸ]<_> : ∀ {ℓ} {A : Set ℓ} (xs : ⟬ A ⟭) → (xs ≫= (λ x → mzero {A = A})) ≡ mzero
  rzero[≫=ᴸ]< [] > = ↯
  rzero[≫=ᴸ]< x ∷ xs > = rzero[≫=ᴸ]< xs >
  
  ldist[≫=ᴸ]<_,_,_> : ∀ {ℓ} {A B : Set ℓ} (xs ys : ⟬ A ⟭) (f : A → ⟬ B ⟭) → ((xs ⊞ ys) ≫= f) ≡ ((xs ≫= f) ⊞ (ys ≫= f))
  ldist[≫=ᴸ]< [] , ys , f > = ↯
  ldist[≫=ᴸ]< x ∷ xs , ys , f > rewrite ldist[≫=ᴸ]< xs , ys , f > | assoc[⊞ᴸ]< f x , xs ≫= f , ys ≫= f > = ↯ 

instance
  cor[⊞][list] : ∀ {ℓ} → cor[⊞] (⟬_⟭ {ℓ})
  cor[⊞][list] = record
    { lunit[⊞]<_> = lunit[⊞ᴸ]<_>
    ; runit[⊞]<_> = runit[⊞ᴸ]<_>
    ; assoc[⊞]<_,_,_> = assoc[⊞ᴸ]<_,_,_> 
    ; lzero[≫=]<_> = lzero[≫=ᴸ]<_>
    ; rzero[≫=]<_> = rzero[≫=ᴸ]<_>
    ; ldist[≫=]<_,_,_> = ldist[≫=ᴸ]<_,_,_>
    }

abstract
  lunit[≫=ᴸ]<_,_> : ∀ {ℓ} {A B : Set ℓ} (x : A) (f : A → ⟬ B ⟭) → (return x ≫= f) ≡ f x
  lunit[≫=ᴸ]< x , f > rewrite runit[⊞]< f x > = ↯
  
  runit[≫=ᴸ]<_> : ∀ {ℓ} {A : Set ℓ} (xM : ⟬ A ⟭) → (xM ≫= return) ≡ xM
  runit[≫=ᴸ]< [] > = ↯
  runit[≫=ᴸ]< x ∷ xs > rewrite runit[≫=ᴸ]< xs > = ↯
  
  assoc[≫=ᴸ]<_,_,_> : ∀ {ℓ} {A B C : Set ℓ} (xM : ⟬ A ⟭) (f : A → ⟬ B ⟭) (g : B → ⟬ C ⟭) → (xM ≫= f ≫= g) ≡ (xM ≫= (λ x → f x ≫= g))
  assoc[≫=ᴸ]< [] , f , g > = ↯
  assoc[≫=ᴸ]< x ∷ xs , f , g > rewrite ldist[≫=]< f x , xs ≫= f , g > | assoc[≫=ᴸ]< xs , f , g > = ↯

instance
  cor[≫=][list] : ∀ {ℓ} → cor[≫=] (⟬_⟭ {ℓ})
  cor[≫=][list] = record
    { lunit[≫=]<_,_> = lunit[≫=ᴸ]<_,_>
    ; runit[≫=]<_> = runit[≫=ᴸ]<_>
    ; assoc[≫=]<_,_,_> = assoc[≫=ᴸ]<_,_,_>
    }

infixl 16 ⨳ᴸ[]
syntax ⨳ᴸ[] f xs ys = xs ⨳ᴸ[ f ] ys

⨳ᴸ[] : ∀ {ℓ} {A B C : Set ℓ} (f : A → B → C) (xs : ⟬ A ⟭) (ys : ⟬ B ⟭) → ⟬ C ⟭
⨳ᴸ[] f xs ys = do
  x ← xs
  y ← ys
  return $ f x y

module _ {ℓ} {A : Set ℓ} where
  module _  {{_ : has[⧺] A}} where
    someᴸ : ⟬ A ⟭
    someᴸ = null ∷ []

    _⨳ᴸ_ : ∀ (xs ys : ⟬ A ⟭) → ⟬ A ⟭
    _⨳ᴸ_ = ⨳ᴸ[] _⧺_

    instance
      has[⨳][list] : has[⨳] ⟬ A ⟭
      has[⨳][list] = record { some = someᴸ ; _⨳_ = _⨳ᴸ_ }

module _ {ℓ} {A : Set ℓ} {{_ : has[⧺] A}} {{_ : cor[⧺] A}} where
  abstract
    lzero[⨳ᴸ]<_> : ∀ (xs : ⟬ A ⟭) → null ⨳ xs ≡ null
    lzero[⨳ᴸ]< xs > = ↯

    rzero[⨳ᴸ]<_> : ∀ (xs : ⟬ A ⟭) → xs ⨳ null ≡ null
    rzero[⨳ᴸ]< [] > = ↯
    rzero[⨳ᴸ]< x ∷ xs > rewrite rzero[⨳ᴸ]< xs > = ↯

    lunit[⨳ᴸ]<_> : ∀ (xs : ⟬ A ⟭) → some ⨳ xs ≡ xs
    lunit[⨳ᴸ]< [] > = ↯
    lunit[⨳ᴸ]< x ∷ xs > rewrite lunit[⨳ᴸ]< xs > | lunit[⧺]< x > = ↯

    runit[⨳ᴸ]<_> : ∀ (xs : ⟬ A ⟭) → xs ⨳ some ≡ xs
    runit[⨳ᴸ]< [] > = ↯
    runit[⨳ᴸ]< x ∷ xs > rewrite runit[⨳ᴸ]< xs > | runit[⧺]< x > = ↯

    rdist[⨳ᴸ]<_,_,_> : ∀ (xs ys zs : ⟬ A ⟭) →  (xs ⧺ ys) ⨳ zs ≡ xs ⨳ zs ⧺ ys ⨳ zs
    rdist[⨳ᴸ]< [] , ys , zs > = ↯
    rdist[⨳ᴸ]< x ∷ xs , ys , zs >
      rewrite rdist[⨳ᴸ]< xs , ys , zs >
            | assoc[⊞]< zs ≫= (λ z → return (x ⧺ z)) , xs ⨳ᴸ zs , ys ⨳ᴸ zs >
            = ↯

    -- MONAD EQUATIONAL REASONING CHALLENGE PROBLEM
    assoc[⨳ᴸ]<_,_,_> : ∀ (xs ys zs : ⟬ A ⟭) → xs ⨳ (ys ⨳ zs) ≡ (xs ⨳ ys) ⨳ zs
    assoc[⨳ᴸ]< xs , ys , zs >
      rewrite ext[≫=ᴸ]< xs
                       , (λ x → do yz ← (do y ← ys ; z ← zs ; return $ y ⧺ z) ; return $ x ⧺ yz)
                       , (λ x → do y ← ys ; yz ← (do z ← zs ; return $ y ⧺ z) ; return $ x ⧺ yz)
                       > (λ x → assoc[≫=]< ys , (λ y → zs ≫= (λ z → return $ y ⧺ z)) , (λ yz → return $ x ⧺ yz) >)
            | ext[≫=ᴸ]< xs
                       , (λ x → do y ← ys ; yz ← (do z ← zs ; return $ y ⧺ z) ; return $ x ⧺ yz)
                       , (λ x → do y ← ys ; z ← zs ; yz ← return $ y ⧺ z ; return $ x ⧺ yz)
                       > (λ x →
                ext[≫=ᴸ]< ys
                         , (λ y → do yz ← (do z ← zs ; return $ y ⧺ z) ; return $ x ⧺ yz)
                         , (λ y → do z ← zs ; yz ← return $ y ⧺ z ; return $ x ⧺ yz)
                         > (λ y → assoc[≫=]< zs , (λ z → return $ y ⧺ z) , (λ yz → return $ x ⧺ yz) >))
            | ext[≫=ᴸ]< xs
                       , (λ x → do y ← ys ; z ← zs ; return $ x ⧺ (y ⧺ z))
                       , (λ x → do y ← ys ; z ← zs ; return $ x ⧺ y ⧺ z)
                       > (λ x →
                ext[≫=ᴸ]< ys
                         , (λ y → do z ← zs ; return $ x ⧺ (y ⧺ z))
                         , (λ y → do z ← zs ; return $ x ⧺ y ⧺ z)
                         > (λ y →
                  ext[≫=ᴸ]< zs
                           , (λ z → return $ x ⧺ (y ⧺ z))
                           , (λ z → return $ x ⧺ y ⧺ z)
                           > (λ z → change[ (λ X → return X ≡ return (x ⧺ y ⧺ z)) ] assoc[⧺]< x , y , z > ↯)))
            | assoc[≫=]< xs , (λ x → ys ≫= (λ y → return $ x ⧺ y)) , (λ xy → zs ≫= (λ z → return $ xy ⧺ z)) >
            | ext[≫=ᴸ]< xs
                       , (λ x → do xy ← (do y ← ys ; return $ x ⧺ y) ; z ← zs ; return $ xy ⧺ z)
                       , (λ x → do y ← ys ; xy ← return $ x ⧺ y ; z ← zs ; return $ xy ⧺ z)
                       > (λ x → assoc[≫=]< ys , (λ y → return $ x ⧺ y) , (λ xy → zs ≫= (λ z → return $ xy ⧺ z)) >)
            | ext[≫=ᴸ]< xs
                       , (λ x → do y ← ys ; ((do z ← zs ; return $ x ⧺ y ⧺ z) ⧺ []))
                       , (λ x → do y ← ys ; z ← zs ; return $ x ⧺ y ⧺ z)
                       > (λ x →
                ext[≫=ᴸ]< ys
                         , (λ y → (do z ← zs ; return $ x ⧺ y ⧺ z) ⧺ [])
                         , (λ y → do z ← zs ; return $ x ⧺ y ⧺ z)
                         > (λ y → runit[⊞]< zs ≫= (λ z → return $ x ⧺ y ⧺ z) >))
            = ↯
    
  instance
    cor[⨳][list] : cor[⨳] ⟬ A ⟭
    cor[⨳][list] = record
      { lzero[⨳]<_> = lzero[⨳ᴸ]<_>
      ; rzero[⨳]<_> = rzero[⨳ᴸ]<_>
      ; lunit[⨳]<_> = lunit[⨳ᴸ]<_>
      ; runit[⨳]<_> = runit[⨳ᴸ]<_>
      ; rdist[⨳]<_,_,_> = rdist[⨳ᴸ]<_,_,_>
      ; assoc[⨳]<_,_,_> = assoc[⨳ᴸ]<_,_,_>
      }
