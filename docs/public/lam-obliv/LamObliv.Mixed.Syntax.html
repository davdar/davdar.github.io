module LamObliv.Mixed.Syntax where

open import UVMVS

open import LamObliv.Common

-- public knowledge ⸨ Φ ⸩
pkn : ∀ G → Set
pkn G = 𝐼[ G ] 𝔹

pkn-wf : ∀ {G} → pkn G → Set
pkn-wf Φ = 𝔠[ Φ ] ≢ 0

data [_ː_]⊢bit[S][_] {R G} (Φ : pkn G) (E : pkn-wf Φ) (r : region R) : Set where
  BitS : ∀
    (𝒷 : 𝐼[ G ] 𝔹)
    → [ Φ ː E ]⊢bit[S][ r ]

data [_ː_]⊢bit[_,_] {R G} (Φ : pkn G) (E : pkn-wf Φ) (r : region R) : ∀ (ℓ : label) → Set where
  BitP : ∀
    (b : 𝔹)
    → [ Φ ː E ]⊢bit[ r , P ]
  BitS : ∀
    (b : [ Φ ː E ]⊢bit[S][ r ])
    → [ Φ ː E ]⊢bit[ r , S ]

data [_ː_]⊢flip[_] {R G} (Φ : pkn G) (E : pkn-wf Φ) (r : region R) : Set where
  Flip : ∀
    (𝒷 : 𝐼[ G ] 𝔹)
    (ε : 𝔲[ 𝒷 ¦ Φ ː E ])
    → [ Φ ː E ]⊢flip[ r ]

postulate
  cond-bit : ∀ {R G} {Φ : pkn G} {E : pkn-wf Φ} {r₁ r₂ r₃ : region R} {ℓ₁ ℓ₂ ℓ₃ : label}
    → [ Φ ː E ]⊢bit[ r₁ , ℓ₁ ]
    → [ Φ ː E ]⊢bit[ r₂ , ℓ₂ ]
    → [ Φ ː E ]⊢bit[ r₃ , ℓ₃ ]
    → [ Φ ː E ]⊢bit[ r₁ ⊔ r₂ ⊔ r₃ , ℓ₁ ⊔ ℓ₂ ⊔ ℓ₃ ] ∧ [ Φ ː E ]⊢bit[ r₁ ⊔ r₂ ⊔ r₃ , ℓ₁ ⊔ ℓ₂ ⊔ ℓ₃ ]

-- well-typed values: ⸨ Σ , Ψ ⊢ⱽ τ ⸩
infix 10 [_ː_]_⊢ⱽ_
data [_ː_]_⊢ⱽ_ {R I G} (Φ : pkn G) (E : pkn-wf Φ) (Σ : tsto R I) : ∀ (τ : type R) → Set where
  Bit : ∀ {r ℓ}
    (b : [ Φ ː E ]⊢bit[ r , ℓ ])
    → [ Φ ː E ] Σ ⊢ⱽ Bit r ℓ
  Flip : ∀ {r}
    (b : [ Φ ː E ]⊢flip[ r ])
    → [ Φ ː E ] Σ ⊢ⱽ Flip r
  Pair : ∀ {τ₁ τ₂}
    (v₁ : [ Φ ː E ] Σ ⊢ⱽ τ₁)
    (v₂ : [ Φ ː E ] Σ ⊢ⱽ τ₂)
    → [ Φ ː E ] Σ ⊢ⱽ τ₁ `× τ₂
  Addr : ∀ {τ}
    {i : idx I}
    (ε : ⟨ i ↦ τ ⟩∈ Σ)
    → [ Φ ː E ] Σ ⊢ⱽ Ref τ

module _ {R I G} (Φ : pkn G) (E : pkn-wf Φ) (Σ : tsto R I) where
  open common R (λ τ → [ Φ ː E ] Σ ⊢ⱽ τ) public using () renaming
    ( _,_⊢ᴱ_,_ to [_ː_]_,_,_⊢ᴱ_,_
    ; _⇛_      to [_ː_]_⊢_⇛_
    ; _⇚_      to [_ː_]_⊢_⇚_
    )

module _ {R I G} {Φ : pkn G} {E : pkn-wf Φ} {Σ : tsto R I} where
  open common R (λ τ → [ Φ ː E ] Σ ⊢ⱽ τ) public hiding
    ( _,_⊢ᴱ_,_
    ; _⇛_
    ; _⇚_
    )

vsto<_>[_ː_] : ∀ {R} {I₀} (Σ₀ : tsto R I₀) {I G} (Φ : pkn G) (E : pkn-wf Φ) (Σ : tsto R I) → Set
vsto< Σ₀ >[ Φ ː E ] Σ = ⟬ (λ τ → [ Φ ː E ] Σ ⊢ⱽ τ) ⟭[[ Σ₀ ]]

vsto[_ː_] : ∀ {R I G} (Φ : pkn G) (E : pkn-wf Φ) (Σ : tsto R I) → Set
vsto[ Φ ː E ] Σ = vsto< Σ >[ Φ ː E ] Σ
