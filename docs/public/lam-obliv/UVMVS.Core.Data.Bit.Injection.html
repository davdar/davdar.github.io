module UVMVS.Core.Data.Bit.Injection where

open import UVMVS.Core.Init
open import UVMVS.Core.Classes

---------------
-- INJECTION --
---------------

ιᴮ : 𝔹 → 𝟙 ∨ 𝟙
ιᴮ ᴏ = ʟ •
ιᴮ ɪ = ʀ •

abstract
  inj[ιᴮ]<_,_> : ∀ (x y : 𝔹) → ιᴮ x ≡ ιᴮ y → x ≡ y
  inj[ιᴮ]< ᴏ , ᴏ > ↯ = ↯
  inj[ιᴮ]< ᴏ , ɪ > ()
  inj[ιᴮ]< ɪ , ᴏ > ()
  inj[ιᴮ]< ɪ , ɪ > ↯ = ↯

instance
  injective[ιᴮ] : injective ιᴮ
  injective[ιᴮ] = record { inj<_,_> = inj[ιᴮ]<_,_> }

_ = ⌁ AT injective ιᴮ
