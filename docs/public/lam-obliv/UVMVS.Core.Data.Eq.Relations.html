module UVMVS.Core.Data.Eq.Relations where

open import UVMVS.Core.Init
open import UVMVS.Core.Classes

module _ {ℓ} {A : Set ℓ} where
  instance
    reflexive[≡] : reflexive (_≡_ AT (A → A → Set))
    reflexive[≡] = record { xRx<_> = xRx⸢≡⸣<_> }
  
    symmetric[≡] : symmetric (_≡_ AT (A → A → Set))
    symmetric[≡] = record { ◇ = ◇⸢≡⸣ }
  
    transitive[≡] : transitive (_≡_ AT (A → A → Set))
    transitive[≡] = record { _⊚_ = _⊚⸢≡⸣_ }

  _ = ⌁ AT reflexive  (_≡_ AT (A → A → Set))
  _ = ⌁ AT symmetric  (_≡_ AT (A → A → Set))
  _ = ⌁ AT transitive (_≡_ AT (A → A → Set))
