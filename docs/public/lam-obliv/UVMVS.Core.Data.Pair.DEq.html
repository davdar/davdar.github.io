module UVMVS.Core.Data.Pair.DEq where

open import UVMVS.Core.Init
open import UVMVS.Core.Classes

module _ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : Set ℓ₂} {{_ : has[≡?] A}} {{_ : has[≡?] B}} where

  _≡?⸢∧⸣_ : ∀ (x y : A ∧ B) → ≡!
  ⟨ x₁ , x₂ ⟩ ≡?⸢∧⸣ ⟨ y₁ , y₂ ⟩ with x₁ ≡? y₁
  … | [≢] = [≢]
  … | [≡] with x₂ ≡? y₂
  … | [≢] = [≢]
  … | [≡] = [≡]
  
  instance
    has[≡?][∧] : has[≡?] (A ∧ B)
    has[≡?][∧] = record { _≡?_ = _≡?⸢∧⸣_ }

  module _ {{_ : cor[≡?] A}} {{_ : cor[≡?] B}} where
    abstract
      _≡⋆⸢∧⸣_ : ∀ (x y : A ∧ B) → x ≡⋆! y
      ⟨ x₁ , x₂ ⟩ ≡⋆⸢∧⸣ ⟨ y₁ , y₂ ⟩ with x₁ ≡⋆ y₁
      … | [≢] ε = [≢] ⟨ (λ where ↯ → contra[≡] ε ↯) ⟩
      … | [≡] ε₁ with x₂ ≡⋆ y₂
      … | [≢] ε = [≢] ⟨ (λ where ↯ → contra[≡] ε ↯) ⟩
      … | [≡] ε₂ = [≡] (change2[ (λ X Y → ⟨ X , Y ⟩ ≡ ⟨ y₁ , y₂ ⟩) ] ε₁ ε₂ ↯)
      
      _≡∿⸢∧⸣_ : ∀ (x y : A ∧ B) → x ≡∿! y [ x ≡?⸢∧⸣ y , x ≡⋆⸢∧⸣ y ]
      ⟨ x₁ , x₂ ⟩ ≡∿⸢∧⸣ ⟨ y₁ , y₂ ⟩ with x₁ ≡? y₁ | x₁ ≡⋆ y₁ | x₁ ≡∿ y₁
      … | [≢] | [≢] _ | [≢] = [≢]
      … | [≡] | [≡] _ | [≡] with x₂ ≡? y₂ | x₂ ≡⋆ y₂ | x₂ ≡∿ y₂
      … | [≢] | [≢] _ | [≢] = [≢]
      … | [≡] | [≡] _ | [≡] = [≡]

    instance
      cor[≡?][∧] : cor[≡?] (A ∧ B)
      cor[≡?][∧] = record { _≡⋆_ = _≡⋆⸢∧⸣_ ; _≡∿_ = _≡∿⸢∧⸣_ }

    _ = ⌁ AT cor[≡?] (A ∧ B)
  _ = ⌁ AT has[≡?] (A ∧ B)
