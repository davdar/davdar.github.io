module UVMVS.Core.Classes.IEq where

open import UVMVS.Core.Init
open import UVMVS.Core.Classes.Relations
open import UVMVS.Core.Classes.Algebra

record has[≍][_] {ℓ} ℓᴿ (A : Set ℓ) : Set (ℓ ⩛ ↟ ℓᴿ) where
  infix 10 _≍_ _≭_
  field
    _≍_ : A → A → Set ℓᴿ
    _≭_ : A → A → Set ℓᴿ
open has[≍][_] {{…}} public

record cor[≍] {ℓ ℓᴿ} (A : Set ℓ) {{_ : has[≍][ ℓᴿ ] A}} : Set (ℓ ⩛ ↟ ℓᴿ) where
  field
    sound[≍]<_,_> : ∀ (x y : A) → x ≍ y → x ≡ y
    complete[≍]<_,_> : ∀ (x y : A) → x ≡ y → x ≍ y
    consistent[≭]<_,_> : ∀ (x y : A) → x ≭ y → ¬ (x ≍ y)
    {{reflexive[≍]}} : reflexive (_≍_ {A = A})
    {{symmetric[≍]}} : symmetric (_≍_ {A = A})
    {{transitive[≍]}} : transitive (_≍_ {A = A})
    {{irreflexive[≭]}} : irreflexive (_≭_ {A = A})
    {{symmetric[≭]}} : symmetric (_≭_ {A = A})
open cor[≍] {{…}} public

------------
-- Making --
------------

module make[≍]
  {ℓ ℓᴿ}
  (A : Set ℓ)
  {{_ : has[≍][ ℓᴿ ] A}}
  (sound[≍]†<_,_> : ∀ (x y : A) → x ≍ y → x ≡ y)
  (xRx[≍]†<_> : ∀ (x : A) → x ≍ x)
  (¬xRx[≭]† : ∀ {x : A} → ¬ (x ≭ x))
  (◇[≭]† : ∀ {x y : A} → x ≭ y → y ≭ x)
  where 

  abstract
    complete[≍]†<_,_> : ∀ (x y : A) → x ≡ y → x ≍ y
    complete[≍]†< x , .x > ↯ = xRx[≍]†< x >
    consistent[≭]†<_,_> : ∀ (x y : A) → x ≭ y → ¬ (x ≍ y)
    consistent[≭]†< x , y > ε ε′ rewrite sound[≍]†< x , y > ε′ = absurd (¬xRx[≭]† ε)
    ⊚[≍]† : ∀ {x y z : A} → y ≍ z → x ≍ y → x ≍ z
    ⊚[≍]† ε₁ ε₂ = complete[≍]†< _ , _ > (sound[≍]†< _ , _ > ε₁ ⊚⸢≡⸣ sound[≍]†< _ , _ > ε₂)
    ◇[≍]† : ∀ {x y : A} → x ≍ y → y ≍ x
    ◇[≍]† ε = complete[≍]†< _ , _ > (◇⸢≡⸣ (sound[≍]†< _ , _ > ε))

    reflexive[≍]† : reflexive (_≍_ AT (A → A → Set ℓᴿ))
    reflexive[≍]† = record { xRx<_> = xRx[≍]†<_> }
    symmetric[≍]† : symmetric (_≍_ AT (A → A → Set ℓᴿ))
    symmetric[≍]† = record { ◇ = ◇[≍]† }
    transitive[≍]† : transitive (_≍_ AT (A → A → Set ℓᴿ))
    transitive[≍]† = record { _⊚_ = ⊚[≍]† }
    irreflexive[≭]† : irreflexive (_≭_ AT (A → A → Set ℓᴿ))
    irreflexive[≭]† = record { ¬xRx = ¬xRx[≭]† }
    symmetric[≭]† : symmetric (_≭_ AT (A → A → Set ℓᴿ))
    symmetric[≭]† = record { ◇ = ◇[≭]† }

  private
    instance
      _ = reflexive[≍]†
      _ = symmetric[≍]†
      _ = transitive[≍]†
      _ = irreflexive[≭]†
      _ = symmetric[≭]†

  cor[≍]† : cor[≍] A
  cor[≍]† = record
    { sound[≍]<_,_> = sound[≍]†<_,_>
    ; complete[≍]<_,_> = complete[≍]†<_,_>
    ; consistent[≭]<_,_> = consistent[≭]†<_,_>
    }

--------------
-- Deriving --
--------------

module derive[cor[≍]]
  {ℓ₁ ℓ₂ ℓᴿ}
  (A : Set ℓ₁) (B : Set ℓ₂) (ι : A → B)
  {{_ : has[≍][ ℓᴿ ] A}} 
  {{_ : has[≍][ ℓᴿ ] B}} {{_ : cor[≍] B}}
  {{_ : injective ι}}
  (isoto[≍]† : ∀ {x y} → x ≍ y → ι x ≍ ι y)
  (isofr[≍]† : ∀ {x y} → ι x ≍ ι y → x ≍ y)
  (isoto[≭]† : ∀ {x y} → x ≭ y → ι x ≭ ι y)
  (isofr[≭]† : ∀ {x y} → ι x ≭ ι y → x ≭ y)
  where

  abstract
    sound[≍]†<_,_> : ∀ (x y : A) → x ≍ y → x ≡ y
    sound[≍]†< x , y > ε = inj< x , y > (sound[≍]< ι x , ι y > (isoto[≍]† ε))

    xRx[≍]†<_> : ∀ (x : A) → x ≍ x
    xRx[≍]†< x > = isofr[≍]† (complete[≍]< ι x , ι x > ↯)

    ¬xRx[≭]† : ∀ {x : A} → ¬ (x ≭ x)
    ¬xRx[≭]† ε = ¬xRx (isoto[≭]† ε)

    ◇[≭]† : ∀ {x y : A} → x ≭ y → y ≭ x
    ◇[≭]† ε = isofr[≭]† (◇ (isoto[≭]† ε))

  open make[≍] A sound[≍]†<_,_> xRx[≍]†<_> ¬xRx[≭]† ◇[≭]† public

------------------
-- Synthesizing --
------------------

module synthesize[has[≍]]
  {ℓ₁ ℓ₂} ℓᴿ
  (A : Set ℓ₁) (B : Set ℓ₂) (ι : A → B)
  {{_ : has[≍][ ℓᴿ ] B}}
  {{_ : injective ι}}
  where

  data _≍†_ (x y : A) : Set ℓᴿ where
    ⟨_⟩ : ι x ≍ ι y → x ≍† y

  data _≭†_ (x y : A) : Set ℓᴿ where
    ⟨_⟩ : ι x ≭ ι y → x ≭† y

  has[≍]† = record
    { _≍_ = _≍†_
    ; _≭_ = _≭†_
    }
  
module synthesize[cor[≍]]
  {ℓ₁ ℓ₂ ℓᴿ}
  (A : Set ℓ₁) (B : Set ℓ₂) (ι : A → B)
  {{_ : has[≍][ ℓᴿ ] B}} {{_ : cor[≍] B}}
  {{_ : injective ι}}
  where

  open synthesize[has[≍]] ℓᴿ A B ι

  private
    instance _ = has[≍]†

  abstract
    isoto[≍]† : ∀ {x y : A} → x ≍ y → ι x ≍ ι y
    isoto[≍]† ⟨ ε ⟩ = ε

    isofr[≍]† : ∀ {x y : A} → ι x ≍ ι y → x ≍ y
    isofr[≍]† ε = ⟨ ε ⟩

    isoto[≭]† : ∀ {x y : A} → x ≭ y → ι x ≭ ι y
    isoto[≭]† ⟨ ε ⟩ = ε

    isofr[≭]† : ∀ {x y : A} → ι x ≭ ι y → x ≭ y
    isofr[≭]† ε = ⟨ ε ⟩

  open derive[cor[≍]] A B ι isoto[≍]† isofr[≍]† isoto[≭]† isofr[≭]† public
