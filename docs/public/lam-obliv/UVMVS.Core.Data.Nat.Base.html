--------------
-- addition --
--------------

_+ᴺ_ : ℕ → ℕ → ℕ
ᴢ +ᴺ y = y
ꜱ x +ᴺ y = ꜱ (x +ᴺ y)

instance
  has[+][ℕ] : has[+] ℕ
  has[+][ℕ] = record { zero = 0 ; _+_ = _+ᴺ_ }

  has[⧺][ℕ] : has[⧺] ℕ
  has[⧺][ℕ] = record { null = 0 ; _⧺_ = _+ᴺ_ }

abstract
  lunit[+ᴺ]<_> : ∀ (x : ℕ) → zero + x ≡ x
  lunit[+ᴺ]< x > = ↯
  
  runit[+ᴺ]<_> : ∀ (x : ℕ) → x + zero ≡ x
  runit[+ᴺ]< ᴢ > = ↯
  runit[+ᴺ]< ꜱ x > rewrite runit[+ᴺ]< x > = ↯
  
  reduc[+ᴺ]<_,_> : ∀ (x y : ℕ) → x + ꜱ y ≡ ꜱ (x + y)
  reduc[+ᴺ]< ᴢ , y > = ↯
  reduc[+ᴺ]< ꜱ x , y > rewrite reduc[+ᴺ]< x , y > = ↯
  
  commu[+ᴺ]<_,_> : ∀ (x y : ℕ) → x + y ≡ y + x
  commu[+ᴺ]< ᴢ , y > rewrite runit[+ᴺ]< y > = ↯
  commu[+ᴺ]< ꜱ x , y > rewrite commu[+ᴺ]< x , y > | reduc[+ᴺ]< y , x > = ↯
  
  assoc[+ᴺ]<_,_,_> : ∀ (x y z : ℕ) → x + (y + z) ≡ (x + y) + z
  assoc[+ᴺ]< ᴢ , y , z > = ↯
  assoc[+ᴺ]< ꜱ x , y , z > rewrite assoc[+ᴺ]< x , y , z > = ↯

instance
  cor[+][ℕ] : cor[+] ℕ
  cor[+][ℕ] = record
    { lunit[+]<_> = lunit[+ᴺ]<_>
    ; runit[+]<_> = runit[+ᴺ]<_>
    ; assoc[+]<_,_,_> = assoc[+ᴺ]<_,_,_>
    ; commu[+]<_,_> = commu[+ᴺ]<_,_>
    }
  cor[⧺][ℕ] : cor[⧺] ℕ
  cor[⧺][ℕ] = record
    { lunit[⧺]<_> = lunit[+ᴺ]<_>
    ; runit[⧺]<_> = runit[+ᴺ]<_>
    ; assoc[⧺]<_,_,_> = assoc[+ᴺ]<_,_,_>
    }
  com[⧺][ℕ] : com[⧺] ℕ
  com[⧺][ℕ] = record
    { commu[⧺]<_,_> = commu[+ᴺ]<_,_>
    }

abstract
  split[+ᴺ]<_,_> : ∀ (x y : ℕ) → x + y ≡ zero → x ≡ zero ∧ y ≡ zero
  split[+ᴺ]< ᴢ , y > ε = ⟨ ↯ , ε ⟩
  split[+ᴺ]< ꜱ x , y > ()

instance
  split[+][ℕ] : split[+] ℕ
  split[+][ℕ] = record
    { split[+]<_,_> = split[+ᴺ]<_,_> }

abstract
  linj[+ᴺ]<_,_,_> : ∀ (x y z : ℕ) → x + y ≡ x + z → y ≡ z
  linj[+ᴺ]< ᴢ , y , z > ε = ε
  linj[+ᴺ]< ꜱ x , y , z > ε with x + y | x + z | linj[+ᴺ]< x , y , z >
  linj[+ᴺ]< ꜱ x , y , z > ↯ | nm | no | IH = IH ↯
  
  rinj[+ᴺ]<_,_,_> : ∀ (x y z : ℕ) → x + z ≡ y + z → x ≡ y
  rinj[+ᴺ]< x , y , z > ε
    rewrite commu[+ᴺ]< x , z >
          | commu[+ᴺ]< y , z >
          | linj[+ᴺ]< z , x , y > ε = ↯

instance
  inj[+][ℕ] : inj[+] ℕ
  inj[+][ℕ] = record
    { linj[+]<_,_,_> = linj[+ᴺ]<_,_,_>
    ; rinj[+]<_,_,_> = rinj[+ᴺ]<_,_,_>
    }

--------------------
-- multiplication --
--------------------

_×ᴺ_ : ∀ (x y : ℕ) → ℕ
ᴢ ×ᴺ y = ᴢ
ꜱ x ×ᴺ y = y + x ×ᴺ y

instance
  has[×][ℕ] : has[×] ℕ
  has[×][ℕ] = record { one = ꜱ ᴢ ; _×_ = _×ᴺ_ }

  has[⨳][ℕ] : has[⨳] ℕ
  has[⨳][ℕ] = record { some = ꜱ ᴢ ; _⨳_ = _×ᴺ_ }

abstract
  lzero[×ᴺ]<_> : ∀ (x : ℕ) → zero × x ≡ zero
  lzero[×ᴺ]< x > = ↯
  
  rzero[×ᴺ]<_> : ∀ (x : ℕ) → x × zero ≡ zero
  rzero[×ᴺ]< ᴢ > = ↯
  rzero[×ᴺ]< ꜱ x > rewrite rzero[×ᴺ]< x > = ↯
  
  lunit[×ᴺ]<_> : ∀ (x : ℕ) → one × x ≡ x
  lunit[×ᴺ]< x > rewrite runit[+]< x > = ↯
  
  runit[×ᴺ]<_> : ∀ (x : ℕ) → x × one ≡ x
  runit[×ᴺ]< ᴢ > = ↯
  runit[×ᴺ]< ꜱ x > rewrite runit[×ᴺ]< x > = ↯
  
  reduc[×ᴺ]<_,_> : ∀ (x y : ℕ) → x × ꜱ y ≡ (x × y) + x
  reduc[×ᴺ]< ᴢ , y > = ↯
  reduc[×ᴺ]< ꜱ x , y >
    rewrite reduc[×ᴺ]< x , y >
          | reduc[+ᴺ]< y + x × y , x >
          | assoc[+]< y , x × y , x > = ↯
  
  commu[×ᴺ]<_,_> : ∀ (x y : ℕ) → x × y ≡ y × x
  commu[×ᴺ]< ᴢ , y > rewrite rzero[×ᴺ]< y > = ↯
  commu[×ᴺ]< ꜱ x , y >
    rewrite reduc[×ᴺ]< y , x >
          | commu[+]< y , x × y >
          | commu[×ᴺ]< x , y > = ↯
  
  rdist[×ᴺ]<_,_,_> : ∀ (x y z : ℕ) → (x + y) × z ≡ (x × z) + (y × z)
  rdist[×ᴺ]< ᴢ , y , z > = ↯
  rdist[×ᴺ]< ꜱ x , y , z > rewrite rdist[×ᴺ]< x , y , z > | assoc[+]< z , x × z , y × z > = ↯
  
  ldist[×ᴺ]<_,_,_> : ∀ (x y z : ℕ) → x × (y + z) ≡ (x × y) + (x × z)
  ldist[×ᴺ]< x , y , z >
    rewrite commu[×ᴺ]< x , y + z >
          | rdist[×ᴺ]< y , z , x >
          | commu[×ᴺ]< y , x >
          | commu[×ᴺ]< x , z > = ↯
  
  assoc[×ᴺ]<_,_,_> : ∀ (x y z : ℕ) → x × (y × z) ≡ (x × y) × z
  assoc[×ᴺ]< ᴢ , y , z > = ↯
  assoc[×ᴺ]< ꜱ x , y , z > rewrite assoc[×ᴺ]< x , y , z > | rdist[×ᴺ]< y , x × y , z > = ↯
  
  discr[×ᴺ]<_,_> : ∀ (x y : ℕ) → x × y ≡ zero → x ≡ zero ∨ y ≡ zero
  discr[×ᴺ]< ᴢ , y > ε = ʟ ↯
  discr[×ᴺ]< ꜱ x , y > ε with split[+]< y , x × y > ε
  … | ⟨ ε₁ , ε₂ ⟩ = ʀ ε₁

instance
  cor[×][ℕ] : cor[×] ℕ
  cor[×][ℕ] = record
    { lzero[×]<_> = lzero[×ᴺ]<_>
    ; rzero[×]<_> = rzero[×ᴺ]<_>
    ; lunit[×]<_> = lunit[×ᴺ]<_>
    ; runit[×]<_> = runit[×ᴺ]<_>
    ; assoc[×]<_,_,_> = assoc[×ᴺ]<_,_,_>
    ; commu[×]<_,_> = commu[×ᴺ]<_,_>
    ; ldist[×]<_,_,_> = ldist[×ᴺ]<_,_,_>
    ; rdist[×]<_,_,_> = rdist[×ᴺ]<_,_,_>
    ; discr[×]<_,_> = discr[×ᴺ]<_,_>
    }
  cor[⨳][ℕ] : cor[⨳] ℕ
  cor[⨳][ℕ] = record
    { lzero[⨳]<_> = lzero[×ᴺ]<_>
    ; rzero[⨳]<_> = rzero[×ᴺ]<_>
    ; lunit[⨳]<_> = lunit[×ᴺ]<_>
    ; runit[⨳]<_> = runit[×ᴺ]<_>
    ; rdist[⨳]<_,_,_> = rdist[×ᴺ]<_,_,_>
    ; assoc[⨳]<_,_,_> = assoc[×ᴺ]<_,_,_>
    }
  com[⨳][ℕ] : com[⨳] ℕ
  com[⨳][ℕ] = record
    { ldist[⨳]<_,_,_> = ldist[×ᴺ]<_,_,_>
    ; commu[⨳]<_,_> = commu[×ᴺ]<_,_>
    }

abstract
  split[×ᴺ]<_,_> : ∀ (x y : ℕ) → x × y ≡ one → x ≡ one ∧ y ≡ one
  split[×ᴺ]< ᴢ , y > ()
  split[×ᴺ]< ꜱ ᴢ , y > ε rewrite runit[+]< y > = ⟨ ↯ , ε ⟩
  split[×ᴺ]< ꜱ (ꜱ x) , ᴢ > ε rewrite rzero[×ᴺ]< x > with ε
  … | ()
  split[×ᴺ]< ꜱ (ꜱ x) , ꜱ y > ε rewrite reduc[+ᴺ]< y , y + x × ꜱ y > with ε
  … | ()

instance
  split[×][ℕ] : split[×] ℕ
  split[×][ℕ] = record
    { split[×]<_,_> = split[×ᴺ]<_,_>
    }

abstract
  rinj[×ᴺ]<_,_,_> : ∀ (x y z : ℕ) → z ≢ 0 → x × z ≡ y × z → x ≡ y
  rinj[×ᴺ]< ᴢ , ᴢ , z > ε₁ ε₂ = ↯
  rinj[×ᴺ]< ᴢ , ꜱ y , z > ε₁ ε₂ = contra[≡] (¬lzero[+]< z , y × z > ε₁) (◇ ε₂)
  rinj[×ᴺ]< ꜱ x , ᴢ , z > ε₁ ε₂ = contra[≡] (¬lzero[+]< z , x × z > ε₁) ε₂
  rinj[×ᴺ]< ꜱ x , ꜱ y , z > ε₁ ε₂ rewrite rinj[×ᴺ]< x , y , z > ε₁ (linj[+]< z , x × z , y × z > ε₂ ) = ↯
  
  linj[×ᴺ]<_,_,_> : ∀ (x y z : ℕ) → x ≢ 0 → x × y ≡ x × z → y ≡ z
  linj[×ᴺ]< x , y , z > ε₁ ε₂
    rewrite commu[×ᴺ]< x , y >
          | commu[×ᴺ]< x , z >
          | rinj[×ᴺ]< y , z , x > ε₁ ε₂ = ↯

instance
  inj[×][ℕ] : inj[×] ℕ
  inj[×][ℕ] = record
    { linj[×]<_,_,_> = linj[×ᴺ]<_,_,_>
    ; rinj[×]<_,_,_> = rinj[×ᴺ]<_,_,_>
    }

--------------------
-- exponentiation --
--------------------

module _ {ℓ} {A : Set ℓ} {{_ : has[+] A}} {{_ : has[×] A}} where
  infixr 17 _^^_
  _^^_ : ∀ (x : A) (y : ℕ) → A
  x ^^ ᴢ = one
  x ^^ ꜱ y = x × x ^^ y
  
  module _ {{_ : cor[+] A}} {{_ : cor[×] A}} where
    abstract
      lzero[^^]<_> : ∀ (x : ℕ) → one ^^ x ≡ one
      lzero[^^]< ᴢ > = ↯
      lzero[^^]< ꜱ x > rewrite lunit[×]< one ^^ x > | lzero[^^]< x > = ↯
      
      rzero[^^]<_> : ∀ (x : A) → x ^^ zero ≡ one
      rzero[^^]< x > = ↯
      
      runit[^^]<_> : ∀ (x : A) → x ^^ one ≡ x
      runit[^^]< x > rewrite runit[×]< x > = ↯
      
      ldist[^^]<_,_,_> : ∀ (x : A) (y z : ℕ) → x ^^ (y + z) ≡ x ^^ y × x ^^ z
      ldist[^^]< x , ᴢ , ᴢ > rewrite lunit[×]< one AT A > = ↯
      ldist[^^]< x , ᴢ , ꜱ z > rewrite lunit[×]< x × x ^^ z > | runit[+]< x × x ^^ z > = ↯
      ldist[^^]< x , ꜱ y , ᴢ > rewrite runit[+]< y > | runit[×]< x × x ^^ y > = ↯
      ldist[^^]< x , ꜱ y , ꜱ z >
        rewrite reduc[+ᴺ]< y , z >
              | ldist[^^]< x , y , z >
              | assoc[×]< x , x ^^ y , x ^^ z >
              | commu[×]< x , x ^^ y >
              | ◇ assoc[×]< x ^^ y , x , x ^^ z >
              | assoc[×]< x , x ^^ y , x × x ^^ z >
              | commu[×]< x , x ^^ y >
              = ↯
      
      rdist[^^]<_,_,_> : ∀ (x y : A) (z : ℕ) → (x × y) ^^ z ≡ x ^^ z × y ^^ z
      rdist[^^]< x , y , ᴢ > rewrite lunit[×]< one AT A > = ↯
      rdist[^^]< x , y , ꜱ z >
        rewrite rdist[^^]< x , y , z >
              | ◇ assoc[×]< x , y , x ^^ z × y ^^ z >
              | assoc[×]< y , x ^^ z , y ^^ z >
              | commu[×]< y , x ^^ z >
              | ◇ assoc[×]< x ^^ z , y , y ^^ z >
              | assoc[×]< x , x ^^ z , y × y ^^ z >
              = ↯
      
      assoc[^^]<_,_,_> : ∀ (x : A) (y z : ℕ) → (x ^^ y) ^^ z ≡ x ^^ (y × z)
      assoc[^^]< x , ᴢ , ᴢ > = ↯
      assoc[^^]< x , ᴢ , ꜱ z > rewrite lunit[×]< (one AT A) ^^ z > | lzero[^^]< z > = ↯
      assoc[^^]< x , ꜱ y , ᴢ > rewrite rzero[×]< y > = ↯
      assoc[^^]< x , ꜱ y , ꜱ z >
        rewrite rdist[^^]< x , x ^^ y , z > 
              | assoc[^^]< x , y , z >
              | ldist[^^]< x , z , y × ꜱ z >
              | commu[×]< y , ꜱ z >
              | ldist[^^]< x , y , z × y >
              | commu[×]< z , y >
              | ◇ assoc[×]< x , x ^^ y , x ^^ z × x ^^ (y × z) >
              | assoc[×]< x ^^ y , x ^^ z , x ^^ (y × z) >
              | assoc[×]< x ^^ z , x ^^ y , x ^^ (y × z) >
              | commu[×]< x ^^ y , x ^^ z >
              = ↯
