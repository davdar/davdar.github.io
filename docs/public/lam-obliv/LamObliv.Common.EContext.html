module LamObliv.Common.EContext where

open import UVMVS

open import LamObliv.Common.Syntax

module common-econtext (R : ℕ) (val : ∀ (τ : type R) → Set) where
  open common-syntax R val

  -- closed evaluation contexts ⸨ c : τ⁻ ⇛ τ⁺ ⸩
  infix 10 _⇛_
  data _⇛_ (τ⁻ : type R) : ∀ (τ⁺ : type R) → Set where
    □ : τ⁻ ⇛ τ⁻
    CastP[c] : ∀ {r}
      (c : τ⁻ ⇛ Flip r)
      → τ⁻ ⇛ Bit ⊥ P
    MuxBit[c,e,e] : ∀ {r₁ r₂ r₃ ℓ₁ ℓ₂ ℓ₃}
      (c₁ : τ⁻ ⇛ Bit r₁ ℓ₁)
      (e₂ : [] , [] ⊢ᴱ Bit r₂ ℓ₂ , [])
      (e₃ : [] , [] ⊢ᴱ Bit r₃ ℓ₃ , [])
      → τ⁻ ⇛ Bit (r₁ ⊔ r₂ ⊔ r₃) (ℓ₁ ⊔ ℓ₂ ⊔ ℓ₃) `× Bit (r₁ ⊔ r₂ ⊔ r₃) (ℓ₁ ⊔ ℓ₂ ⊔ ℓ₃)
    MuxBit[v,c,e] : ∀ {r₁ r₂ r₃ ℓ₁ ℓ₂ ℓ₃}
      (v₁ : val (Bit r₁ ℓ₁))
      (c₂ : τ⁻ ⇛ Bit r₂ ℓ₂)
      (e₃ : [] , [] ⊢ᴱ Bit r₃ ℓ₃ , [])
      → τ⁻ ⇛ Bit (r₁ ⊔ r₂ ⊔ r₃) (ℓ₁ ⊔ ℓ₂ ⊔ ℓ₃) `× Bit (r₁ ⊔ r₂ ⊔ r₃) (ℓ₁ ⊔ ℓ₂ ⊔ ℓ₃)
    MuxBit[v,v,c] : ∀ {r₁ r₂ r₃ ℓ₁ ℓ₂ ℓ₃}
      (v₁ : val (Bit r₁ ℓ₁))
      (v₂ : val (Bit r₂ ℓ₂))
      (c₃ : τ⁻ ⇛ Bit r₃ ℓ₃)
      → τ⁻ ⇛ Bit (r₁ ⊔ r₂ ⊔ r₃) (ℓ₁ ⊔ ℓ₂ ⊔ ℓ₃) `× Bit (r₁ ⊔ r₂ ⊔ r₃) (ℓ₁ ⊔ ℓ₂ ⊔ ℓ₃)
    MuxFlip[c,e,e] : ∀ {r₁ r₂ r₃ ℓ}
      (c₁ : τ⁻ ⇛ Bit r₁ ℓ)
      (e₂ : [] , [] ⊢ᴱ Flip r₂ , [])
      (e₃ : [] , [] ⊢ᴱ Flip r₃ , [])
      (ε₁ : r₁ ⊓ r₂ ≡ ⊥)
      (ε₂ : r₁ ⊓ r₃ ≡ ⊥)
      → τ⁻ ⇛ Flip (r₂ ⊔ r₃) `× Flip (r₂ ⊔ r₃)
    MuxFlip[v,c,e] : ∀ {r₁ r₂ r₃ ℓ}
      (v₁ : val (Bit r₁ ℓ))
      (c₂ : τ⁻ ⇛ Flip r₂)
      (e₃ : [] , [] ⊢ᴱ Flip r₃ , [])
      (ε₁ : r₁ ⊓ r₂ ≡ ⊥)
      (ε₂ : r₁ ⊓ r₃ ≡ ⊥)
      → τ⁻ ⇛ Flip (r₂ ⊔ r₃) `× Flip (r₂ ⊔ r₃)
    MuxFlip[v,v,c] : ∀ {r₁ r₂ r₃ ℓ}
      (v₁ : val (Bit r₁ ℓ))
      (v₂ : val (Flip r₂))
      (c₃ : τ⁻ ⇛ Flip r₃)
      (ε₁ : r₁ ⊓ r₂ ≡ ⊥)
      (ε₂ : r₁ ⊓ r₃ ≡ ⊥)
      → τ⁻ ⇛ Flip (r₂ ⊔ r₃) `× Flip (r₂ ⊔ r₃)
    If[c,e,e] : ∀ {r τ}
      (c₁ : τ⁻ ⇛ Bit r P)
      (e₂ : [] , [] ⊢ᴱ τ , [])
      (e₃ : [] , [] ⊢ᴱ τ , [])
      → τ⁻ ⇛ τ
    Pair[c,e] : ∀ {τ₁ τ₂}
      (c₁ : τ⁻ ⇛ τ₁)
      (e₂ : [] , [] ⊢ᴱ τ₂ , [])
      → τ⁻ ⇛ τ₁ `× τ₂
    Pair[v,c] : ∀ {τ₁ τ₂}
      (v₁ : val τ₁)
      (c₂ : τ⁻ ⇛ τ₂)
      → τ⁻ ⇛ τ₁ `× τ₂
    Let[c,e] : ∀ {τ₁ τ₂ a}
      (c₁ : τ⁻ ⇛ τ₁)
      (e₂ : τ₁ ∷ [] , ᴏ ∷ [] ⊢ᴱ τ₂ , a ∷ [])
      → τ⁻ ⇛ τ₂
    LetPair[c,e] : ∀ {τ₁ τ₂ τ a₁ a₂}
      (c₁ : τ⁻ ⇛ τ₁ `× τ₂)
      (e₂ : τ₁ ∷ τ₂ ∷ [] , ᴏ ∷ ᴏ ∷ [] ⊢ᴱ τ , a₁ ∷ a₂ ∷ [])
      → τ⁻ ⇛ τ
    Ref[c] : ∀ {τ}
      (c : τ⁻ ⇛ τ)
      → τ⁻ ⇛ Ref τ
    Read/Write[c,e] : ∀ {τ}
      (c₁ : τ⁻ ⇛ τ)
      (e₂ : [] , [] ⊢ᴱ Ref τ , [])
      → τ⁻ ⇛ τ
    Read/Write[v,c] : ∀ {τ}
      (v₁ : val τ)
      (c : τ⁻ ⇛ Ref τ)
      → τ⁻ ⇛ τ

  -- composing contexts
  _⊚ᶜ_ : ∀ {τ₁ τ₂ τ₃} (c₂ : τ₂ ⇛ τ₃) (c₁ : τ₁ ⇛ τ₂) → τ₁ ⇛ τ₃
  □ ⊚ᶜ c′ = c′
  CastP[c] c ⊚ᶜ c′ = CastP[c] (c ⊚ᶜ c′)
  MuxBit[c,e,e] c₁ e₂ e₃ ⊚ᶜ c′ = MuxBit[c,e,e] (c₁ ⊚ᶜ c′) e₂ e₃
  MuxBit[v,c,e] v₁ c₂ e₃ ⊚ᶜ c′ = MuxBit[v,c,e] v₁ (c₂ ⊚ᶜ c′) e₃
  MuxBit[v,v,c] v₁ v₂ c₃ ⊚ᶜ c′ = MuxBit[v,v,c] v₁ v₂ (c₃ ⊚ᶜ c′)
  MuxFlip[c,e,e] c₁ e₂ e₃ ε₁ ε₂ ⊚ᶜ c′ = MuxFlip[c,e,e] (c₁ ⊚ᶜ c′) e₂ e₃ ε₁ ε₂
  MuxFlip[v,c,e] v₁ c₂ e₃ ε₁ ε₂ ⊚ᶜ c′ = MuxFlip[v,c,e] v₁ (c₂ ⊚ᶜ c′) e₃ ε₁ ε₂
  MuxFlip[v,v,c] v₁ v₂ c₃ ε₁ ε₂ ⊚ᶜ c′ = MuxFlip[v,v,c] v₁ v₂ (c₃ ⊚ᶜ c′) ε₁ ε₂
  If[c,e,e] c₁ e₂ e₃ ⊚ᶜ c′ = If[c,e,e] (c₁ ⊚ᶜ c′) e₂ e₃
  Pair[c,e] c₁ e₂ ⊚ᶜ c′ = Pair[c,e] (c₁ ⊚ᶜ c′) e₂
  Pair[v,c] v₁ c₂ ⊚ᶜ c′ = Pair[v,c] v₁ (c₂ ⊚ᶜ c′)
  Let[c,e] c₁ e₂ ⊚ᶜ c′ = Let[c,e] (c₁ ⊚ᶜ c′) e₂
  LetPair[c,e] c₁ e₂ ⊚ᶜ c′ = LetPair[c,e] (c₁ ⊚ᶜ c′) e₂
  Ref[c] c ⊚ᶜ c′ = Ref[c] (c ⊚ᶜ c′)
  Read/Write[c,e] c₁ e₂ ⊚ᶜ c′ = Read/Write[c,e] (c₁ ⊚ᶜ c′) e₂
  Read/Write[v,c] v₁ c₁ ⊚ᶜ c′ = Read/Write[v,c] v₁ (c₁ ⊚ᶜ c′)

  -- filling an evaluation context hole ⸨ c ⸩ with expression ⸨ e ⸩, 𝑖.𝑒., ⸨ c[e] ⸩
  infixl  19 _⟁ᶜ_
  _⟁ᶜ_ : ∀ {τ⁻ τ⁺ : type R} (c : τ⁻ ⇛ τ⁺) (e : [] , [] ⊢ᴱ τ⁻ , []) → [] , [] ⊢ᴱ τ⁺ , []
  □ ⟁ᶜ e′ = e′
  CastP[c] c ⟁ᶜ e′ = CastP (c ⟁ᶜ e′)
  MuxBit[c,e,e] c₁ e₂ e₃ ⟁ᶜ e′ = MuxBit (c₁ ⟁ᶜ e′) e₂ e₃
  MuxBit[v,c,e] v₁ c₂ e₃ ⟁ᶜ e′ = MuxBit (Val v₁) (c₂ ⟁ᶜ e′) e₃
  MuxBit[v,v,c] v₁ v₂ c₃ ⟁ᶜ e′ = MuxBit (Val v₁) (Val v₂) (c₃ ⟁ᶜ e′)
  MuxFlip[c,e,e] c₁ e₂ e₃ ε₁ ε₂ ⟁ᶜ e′ = MuxFlip (c₁ ⟁ᶜ e′) e₂ e₃ ε₁ ε₂
  MuxFlip[v,c,e] v₁ c₂ e₃ ε₁ ε₂ ⟁ᶜ e′ = MuxFlip (Val v₁) (c₂ ⟁ᶜ e′) e₃ ε₁ ε₂
  MuxFlip[v,v,c] v₁ v₂ c₃ ε₁ ε₂ ⟁ᶜ e′ = MuxFlip (Val v₁) (Val v₂) (c₃ ⟁ᶜ e′) ε₁ ε₂
  If[c,e,e] c e₁ e₂ ⟁ᶜ e′ = If (c ⟁ᶜ e′) e₁ e₂ ↯
  Pair[c,e] c₁ e₂ ⟁ᶜ e′ = Pair (c₁ ⟁ᶜ e′) e₂
  Pair[v,c] v₁ c₂ ⟁ᶜ e′ = Pair (Val v₁) (c₂ ⟁ᶜ e′)
  Let[c,e] c e ⟁ᶜ e′ = Let (c ⟁ᶜ e′) e
  LetPair[c,e] c e ⟁ᶜ e′ = LetPair (c ⟁ᶜ e′) e
  Ref[c] c ⟁ᶜ e′ = Ref (c ⟁ᶜ e′)
  Read/Write[c,e] c₁ e₂ ⟁ᶜ e′ = Read/Write (c₁ ⟁ᶜ e′) e₂
  Read/Write[v,c] v₁ c₂ ⟁ᶜ e′ = Read/Write (Val v₁) (c₂ ⟁ᶜ e′)

  -- zip is injective
  inj[⟁ᶜ]<_,_,_> : ∀ {τ⁻ τ⁺} (c : τ⁻ ⇛ τ⁺) (e₁ e₂ : [] , [] ⊢ᴱ τ⁻ , []) → c ⟁ᶜ e₁ ≡ c ⟁ᶜ e₂ → e₁ ≡ e₂
  inj[⟁ᶜ]< □ , e₁ , e₂ > ε = ε
  inj[⟁ᶜ]< CastP[c] c , e₁ , e₂ > ε with c ⟁ᶜ e₁ | c ⟁ᶜ e₂ | inj[⟁ᶜ]< c , e₁ , e₂ > | ε
  … | X₁ | X₂ | IH | ↯ = IH ↯
  inj[⟁ᶜ]< MuxBit[c,e,e] c _ _ , e₁ , e₂ > ε with c ⟁ᶜ e₁ | c ⟁ᶜ e₂ | inj[⟁ᶜ]< c , e₁ , e₂ > | ε
  … | X₁ | X₂ | IH | ↯ = IH ↯
  inj[⟁ᶜ]< MuxBit[v,c,e] _ c _ , e₁ , e₂ > ε with c ⟁ᶜ e₁ | c ⟁ᶜ e₂ | inj[⟁ᶜ]< c , e₁ , e₂ > | ε
  … | X₁ | X₂ | IH | ↯ = IH ↯
  inj[⟁ᶜ]< MuxBit[v,v,c] _ _ c , e₁ , e₂ > ε with c ⟁ᶜ e₁ | c ⟁ᶜ e₂ | inj[⟁ᶜ]< c , e₁ , e₂ > | ε
  … | X₁ | X₂ | IH | ↯ = IH ↯
  inj[⟁ᶜ]< MuxFlip[c,e,e] c _ _ ε₁ ε₂ , e₁ , e₂ > ε with c ⟁ᶜ e₁ | c ⟁ᶜ e₂ | inj[⟁ᶜ]< c , e₁ , e₂ > | ε
  … | X₁ | X₂ | IH | ↯ = IH ↯
  inj[⟁ᶜ]< MuxFlip[v,c,e] _ c _ ε₁ ε₂ , e₁ , e₂ > ε with c ⟁ᶜ e₁ | c ⟁ᶜ e₂ | inj[⟁ᶜ]< c , e₁ , e₂ > | ε
  … | X₁ | X₂ | IH | ↯ = IH ↯
  inj[⟁ᶜ]< MuxFlip[v,v,c] _ _ c ε₁ ε₂ , e₁ , e₂ > ε with c ⟁ᶜ e₁ | c ⟁ᶜ e₂ | inj[⟁ᶜ]< c , e₁ , e₂ > | ε
  … | X₁ | X₂ | IH | ↯ = IH ↯
  inj[⟁ᶜ]< If[c,e,e] c _ _ , e₁ , e₂ > ε with c ⟁ᶜ e₁ | c ⟁ᶜ e₂ | inj[⟁ᶜ]< c , e₁ , e₂ > | ε
  … | X₁ | X₂ | IH | ↯ = IH ↯
  inj[⟁ᶜ]< Pair[c,e] c _ , e₁ , e₂ > ε with c ⟁ᶜ e₁ | c ⟁ᶜ e₂ | inj[⟁ᶜ]< c , e₁ , e₂ > | ε
  … | X₁ | X₂ | IH | ↯ = IH ↯
  inj[⟁ᶜ]< Pair[v,c] _ c , e₁ , e₂ > ε with c ⟁ᶜ e₁ | c ⟁ᶜ e₂ | inj[⟁ᶜ]< c , e₁ , e₂ > | ε
  … | X₁ | X₂ | IH | ↯ = IH ↯
  inj[⟁ᶜ]< Let[c,e] c _ , e₁ , e₂ > ε with c ⟁ᶜ e₁ | c ⟁ᶜ e₂ | inj[⟁ᶜ]< c , e₁ , e₂ > | ε
  … | X₁ | X₂ | IH | ↯ = IH ↯
  inj[⟁ᶜ]< LetPair[c,e] c _ , e₁ , e₂ > ε with c ⟁ᶜ e₁ | c ⟁ᶜ e₂ | inj[⟁ᶜ]< c , e₁ , e₂ > | ε
  … | X₁ | X₂ | IH | ↯ = IH ↯
  inj[⟁ᶜ]< Ref[c] c , e₁ , e₂ > ε with c ⟁ᶜ e₁ | c ⟁ᶜ e₂ | inj[⟁ᶜ]< c , e₁ , e₂ > | ε
  … | X₁ | X₂ | IH | ↯ = IH ↯
  inj[⟁ᶜ]< Read/Write[c,e] c _ , e₁ , e₂ > ε with c ⟁ᶜ e₁ | c ⟁ᶜ e₂ | inj[⟁ᶜ]< c , e₁ , e₂ > | ε
  … | X₁ | X₂ | IH | ↯ = IH ↯
  inj[⟁ᶜ]< Read/Write[v,c] _ c , e₁ , e₂ > ε with c ⟁ᶜ e₁ | c ⟁ᶜ e₂ | inj[⟁ᶜ]< c , e₁ , e₂ > | ε
  … | X₁ | X₂ | IH | ↯ = IH ↯
 
  -- filling a context with disequal expressions yields disequal expressions
  ≢[⟁ᶜ] : ∀ {τ⁻ τ⁺} (c : τ⁻ ⇛ τ⁺) {e₁ e₂ : [] , [] ⊢ᴱ τ⁻ , []} → e₁ ≢ e₂ → c ⟁ᶜ e₁ ≢ c ⟁ᶜ e₂
  ≢[⟁ᶜ] c ε = ⟨ (λ ε′ → contra[≡] ε (inj[⟁ᶜ]< c , _ , _ > ε′)) ⟩

  -- if a list of expressions does not contain ⸨ e ⸩, then filling a
  -- hole ⸨ c ⸩ with that list of expressions does not contain
  -- ⸨ c[e] ⸩
  ∉[⟁ᶜ] : ∀ {τ⁻ τ⁺} {e : [] , [] ⊢ᴱ τ⁻ , []} {es : ⟬ [] , [] ⊢ᴱ τ⁻ , [] ⟭} (c : τ⁻ ⇛ τ⁺) → e ∉ es → c ⟁ᶜ e ∉ map (_⟁ᶜ_ c) es
  ∉[⟁ᶜ] c [] = []
  ∉[⟁ᶜ] c (ε ∷ εs) = ≢[⟁ᶜ] c ε ∷ ∉[⟁ᶜ] c εs
