---------
-- _≤_ --
---------

data _≤ᴺ_ : (x y : ℕ) → Set where
  ᴢ : ∀ {x} → ᴢ ≤ᴺ x
  ꜱ : ∀ {x y} (ε : x ≤ᴺ y) → ꜱ x ≤ᴺ ꜱ y

xRx[≤ᴺ]<_> : ∀ (x : ℕ) → x ≤ᴺ x
xRx[≤ᴺ]< ᴢ > = ᴢ
xRx[≤ᴺ]< ꜱ x > = ꜱ xRx[≤ᴺ]< x >

_⋈[≤ᴺ]_ : ∀ {x y : ℕ} → x ≤ᴺ y → y ≤ᴺ x → x ≡ y
ᴢ ⋈[≤ᴺ] ᴢ = ↯
ꜱ ε₁ ⋈[≤ᴺ] ꜱ ε₂ rewrite ε₁ ⋈[≤ᴺ] ε₂ = ↯

_⊚[≤ᴺ]_ : ∀ {x y z : ℕ} → y ≤ᴺ z → x ≤ᴺ y → x ≤ᴺ z
ε₁ ⊚[≤ᴺ] ᴢ = ᴢ
ꜱ ε₁ ⊚[≤ᴺ] ꜱ ε₂ = ꜱ (ε₁ ⊚[≤ᴺ] ε₂)

instance
  reflexive[≤ᴺ] : reflexive _≤ᴺ_
  reflexive[≤ᴺ] = record { xRx<_> = xRx[≤ᴺ]<_> }
  antisymmetric[≤ᴺ] : antisymmetric _≤ᴺ_
  antisymmetric[≤ᴺ] = record { _⋈_ = _⋈[≤ᴺ]_ }
  transitive[≤ᴺ] : transitive _≤ᴺ_
  transitive[≤ᴺ] = record { _⊚_ = _⊚[≤ᴺ]_ }

weaken[≤ᴺ] : ∀ {n m} (ε : n ≤ᴺ m) → n ≤ᴺ ꜱ m
weaken[≤ᴺ] ᴢ = ᴢ
weaken[≤ᴺ] (ꜱ ε) = ꜱ (weaken[≤ᴺ] ε)

---------
-- _⊴_ --
---------

infix 10 _⊴ᴺ_ 

data _⊴ᴺ_ : (n m : ℕ) → Set where
  ᴢ : ∀ {n} → n ⊴ᴺ n
  ꜱ : ∀ {n m} → n ⊴ᴺ m → n ⊴ᴺ ꜱ m

zero[⊴ᴺ] : ∀ {n} → 0 ⊴ᴺ n
zero[⊴ᴺ] {ᴢ} = ᴢ
zero[⊴ᴺ] {ꜱ n} = ꜱ zero[⊴ᴺ]

succ[⊴ᴺ] : ∀ {n m} (ε : n ⊴ᴺ m) → ꜱ n ⊴ᴺ ꜱ m
succ[⊴ᴺ] ᴢ = ᴢ
succ[⊴ᴺ] (ꜱ ε) = ꜱ (succ[⊴ᴺ] ε)

isoto[⊴ᴺ] : ∀ {n m} (ε : n ⊴ᴺ m) → n ≤ᴺ m
isoto[⊴ᴺ] ᴢ = xRx
isoto[⊴ᴺ] (ꜱ ε) = weaken[≤ᴺ] (isoto[⊴ᴺ] ε)

isofr[⊴ᴺ] : ∀ {n m} (ε : n ≤ᴺ m) → n ⊴ᴺ m
isofr[⊴ᴺ] ᴢ = zero[⊴ᴺ]
isofr[⊴ᴺ] (ꜱ ε) = succ[⊴ᴺ] (isofr[⊴ᴺ] ε)

private
  𝔐[xRx] : deriving[xRx] ℕ _⊴ᴺ_ ℕ _≤ᴺ_ id
  𝔐[xRx] = derive[xRx] isofr[⊴ᴺ]
  𝔐[⋈] : deriving[⋈] ℕ _⊴ᴺ_ ℕ _≤ᴺ_ id
  𝔐[⋈] = derive[⋈] isoto[⊴ᴺ]
  𝔐[⊚] : deriving[⊚] ℕ _⊴ᴺ_ ℕ _≤ᴺ_ id
  𝔐[⊚] = derive[⊚] isoto[⊴ᴺ] isofr[⊴ᴺ]
  module 𝔐 where
    open deriving[xRx] 𝔐[xRx] public
    open deriving[⋈] 𝔐[⋈] public
    open deriving[⊚] 𝔐[⊚] public
instance
  reflexive[⊴ᴺ] = 𝔐.reflexive[≼]†
  antisymmetric[⊴ᴺ] = 𝔐.antisymmetric[≼]†
  transitive[⊴ᴺ] = 𝔐.transitive[≼]†

-- _<_ --

xRx[<ᴺ]<_> : ∀ n → n <ᴺ ꜱ n
xRx[<ᴺ]< ᴢ > = ᴢ
xRx[<ᴺ]< ꜱ n > = ꜱ xRx[<ᴺ]< n >

weaken[<ᴺ] : ∀ {n m} (ε : n <ᴺ m) → n <ᴺ ꜱ m
weaken[<ᴺ] ᴢ = ᴢ
weaken[<ᴺ] (ꜱ ε) = ꜱ (weaken[<ᴺ] ε)

data _⊲ᴺ_ : (n m : ℕ) → Set where
  ᴢ : ∀ {n} → n ⊲ᴺ ꜱ n
  ꜱ : ∀ {n m} (ε : n ⊲ᴺ m) → n ⊲ᴺ ꜱ m

zero[⊲ᴺ] : ∀ {n} → 0 ⊲ᴺ ꜱ n
zero[⊲ᴺ] {ᴢ} = ᴢ
zero[⊲ᴺ] {ꜱ n} = ꜱ zero[⊲ᴺ]

succ[⊲ᴺ] : ∀ {n m} (ε : n ⊲ᴺ m) → ꜱ n ⊲ᴺ ꜱ m
succ[⊲ᴺ] ᴢ = ᴢ
succ[⊲ᴺ] (ꜱ ε) = ꜱ (succ[⊲ᴺ] ε)

isoto[⊲ᴺ] : ∀ {n m} (ε : n ⊲ᴺ m) → n <ᴺ m
isoto[⊲ᴺ] ᴢ = xRx[<ᴺ]< _ >
isoto[⊲ᴺ] (ꜱ ε) = weaken[<ᴺ] (isoto[⊲ᴺ] ε)

isofr[⊲ᴺ] : ∀ {n m} (ε : n <ᴺ m) → n ⊲ᴺ m
isofr[⊲ᴺ] ᴢ = zero[⊲ᴺ]
isofr[⊲ᴺ] (ꜱ ε) = succ[⊲ᴺ] (isofr[⊲ᴺ] ε)
