infixl 11 _⇧ᴺ_ _⇧ᴵ_ _⇧ⱽ_ _⇧ᵋ_
infixl 15 _∸ᴺ_ _∸ⱽ_

_∸ᴺ_ : ∀ N (i : idx N) → ℕ
ꜱ N ∸ᴺ ᴢ = N
ꜱ N ∸ᴺ ꜱ i = ꜱ (N ∸ᴺ i )

_∸ⱽ_ : ∀ {ℓ} {A : Set ℓ} {N} (xs : ⟬ A ⟭[ N ]) (i : idx N) → ⟬ A ⟭[ N ∸ᴺ i ]
(x ∷ X) ∸ⱽ ᴢ = X
(x ∷ X) ∸ⱽ ꜱ i = x ∷ (X ∸ⱽ i)

_⇧ᴺ_ : ℕ → ℕ → ℕ
x ⇧ᴺ ᴢ = x
x ⇧ᴺ ꜱ y = ꜱ x ⇧ᴺ y

_⇧ᴵ_ : ∀ {N} (i : idx N) N′ → idx (N ⇧ᴺ N′)
i ⇧ᴵ ᴢ = i
i ⇧ᴵ ꜱ N′ = ꜱ i ⇧ᴵ N′

_⇧ⱽ_ : ∀ {ℓ} {A : Set ℓ} {N N′} (xs : ⟬ A ⟭[ N ]) (xs′ : ⟬ A ⟭[ N′ ]) → ⟬ A ⟭[ N ⇧ᴺ N′ ]
X ⇧ⱽ [] = X
X ⇧ⱽ x ∷ X′ = (x ∷ X) ⇧ⱽ X′

_⇧ᵋ_ : ∀ {ℓ} {A : Set ℓ} {N N′} {i : idx N} {X : ⟬ A ⟭[ N ]} {x : A} (ε : ⟨ i ↦ x ⟩∈ X) (X′ : ⟬ A ⟭[ N′ ]) → ⟨ i ⇧ᴵ N′ ↦ x ⟩∈ (X ⇧ⱽ X′)
ε ⇧ᵋ [] = ε
ε ⇧ᵋ x′ ∷ X′ = ꜱ ε ⇧ᵋ X′

abstract
  dist[⇧/extend]<_,_,_,_> : ∀ {ℓ} {A : Set ℓ} {N N′} (X : ⟬ A ⟭[ N ]) (i : idx N) (x : A) (X′ : ⟬ A ⟭[ N′ ]) → X #[ i ↦ x ] ⇧ⱽ X′ ≡ (X ⇧ⱽ X′) #[ (i ⇧ᴵ N′) ↦ x ]
  dist[⇧/extend]< X , i , x , []      > = ↯
  dist[⇧/extend]< X , i , x , x′ ∷ X′ > = dist[⇧/extend]< x′ ∷ X , ꜱ i , x , X′ >
  
  reduc[⇧/lookup]<_,_,_> : ∀ {ℓ} {A : Set ℓ} {N N′} (X : ⟬ A ⟭[ N ]) (i : idx N) (X′ : ⟬ A ⟭[ N′ ]) → (X ⇧ⱽ X′) #[ i ⇧ᴵ N′ ] ≡ X #[ i ]
  reduc[⇧/lookup]< X , i , [] > = ↯
  reduc[⇧/lookup]< X , i , x ∷ X′ > = reduc[⇧/lookup]< x ∷ X , ꜱ i , X′ >

  dist[∸/⊔]<_,_,_> : ∀ {ℓ} {A : Set ℓ} {{_ : has[⊔] A}} {N} (X₁ : ⟬ A ⟭[ N ]) (X₂ : ⟬ A ⟭[ N ]) (i : idx N) → (X₁ ⊔ X₂) ∸ⱽ i ≡ (X₁ ∸ⱽ i) ⊔ (X₂ ∸ⱽ i)
  dist[∸/⊔]< x₁ ∷ X₁ , x₂ ∷ X₂ , ᴢ > = ↯
  dist[∸/⊔]< x₁ ∷ X₁ , x₂ ∷ X₂ , ꜱ i > rewrite dist[∸/⊔]< X₁ , X₂ , i > = ↯
