module UVMVS.Core.Classes.Relations where

open import UVMVS.Core.Init

proper : ∀ {ℓ ℓᴿ} {A : Set ℓ} (_≼_ : A → A → Set ℓᴿ) (x : A) → Set ℓᴿ
proper _≼_ x = x ≼ x

_⇉_ : ∀ {ℓ₁₁ ℓ₁₂ ℓ₂₁ ℓ₂₂ ℓᴿ₁ ℓᴿ₂} {A₁ : Set ℓ₁₁} {B₁ : Set ℓ₁₂} {A₂ : Set ℓ₂₁} {B₂ : Set ℓ₂₂}
      (_≼₁_ : A₁ → B₁ → Set ℓᴿ₁) (_≼₂_ : A₂ → B₂ → Set ℓᴿ₂)
    → (f : A₁ → A₂)
    → (g : B₁ → B₂)
    → Set (ℓ₁₁ ⩛ ℓ₁₂ ⩛ ℓᴿ₁ ⩛ ℓᴿ₂)
(_≼₁_ ⇉ _≼₂_) f g = ∀ {x y} → x ≼₁ y → f x ≼₂ g y

record reflexive {ℓ ℓᴿ} {A : Set ℓ} (_≼_ : A → A → Set ℓᴿ) : Set (ℓ ⩛ ℓᴿ) where
  field
    xRx<_> : ∀ x → x ≼ x
  xRx : ∀ {x} → x ≼ x
  xRx = xRx< _ >
  xRx[≡] : ∀ {x y} → x ≡ y → x ≼ y
  xRx[≡] ↯ = xRx
open reflexive {{…}} public

record irreflexive {ℓ ℓᴿ} {A : Set ℓ} (_≼_ : A → A → Set ℓᴿ) : Set (ℓ ⩛ ℓᴿ) where
  field
    ¬xRx : ∀ {x} → ¬ (x ≼ x)
open irreflexive {{…}} public

record symmetric {ℓ ℓᴿ} {A : Set ℓ} (_≼_ : A → A → Set ℓᴿ) : Set (ℓ ⩛ ℓᴿ) where
  field
    ◇ : ∀ {x y} → x ≼ y → y ≼ x
open symmetric {{…}} public

record antisymmetric {ℓ ℓᴿ} {A : Set ℓ} (_≼_ : A → A → Set ℓᴿ) : Set (ℓ ⩛ ℓᴿ) where
  infixl 19 _⋈_
  field
    _⋈_ : ∀ {x y} → x ≼ y → y ≼ x → x ≡ y
open antisymmetric {{…}} public

record asymmetric {ℓ ℓᴿ} {A : Set ℓ} (_≼_ : A → A → Set ℓᴿ) : Set (ℓ ⩛ ℓᴿ) where
  field
    _¬◇_ : ∀ {x y} → x ≼ y → y ≼ x → 𝟘
open asymmetric {{…}} public

record transitive {ℓ ℓᴿ} {A : Set ℓ} (_≼_ : A → A → Set ℓᴿ) : Set (ℓ ⩛ ℓᴿ) where
  infixl 19 _⊚_
  field
    _⊚_ : ∀ {x y z} → y ≼ z → x ≼ y → x ≼ z
open transitive {{…}} public

record injective {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : Set ℓ₂} (ι : A → B) : Set (ℓ₁ ⩛ ℓ₂) where
  field
    inj<_,_> : ∀ (x y : A) → ι x ≡ ι y → x ≡ y
open injective {{…}} public

instance
  injective[id] : ∀ {ℓ} {A : Set ℓ} → injective (id {A = A})
  injective[id] = record { inj<_,_> = λ x y → id }

record deriving[xRx] {ℓ₁ ℓ₂ ℓᴿ₁ ℓᴿ₂}
  (A : Set ℓ₁) (_≼ᴬ_ : (x y : A) → Set ℓᴿ₁)
  (B : Set ℓ₂) (_≼ᴮ_ : (x y : B) → Set ℓᴿ₂) {{_ : reflexive _≼ᴮ_}}
  (ι : A → B)
  : Set (ℓ₁ ⩛ ℓᴿ₁ ⩛ ℓᴿ₂)
  where
  constructor derive[xRx]
  field
    isofr : ∀ {x y} → ι x ≼ᴮ ι y → x ≼ᴬ y

  private
    xRx†<_> : ∀ x → x ≼ᴬ x
    xRx†< x > = isofr xRx< ι x >

  reflexive[≼]† : reflexive _≼ᴬ_
  reflexive[≼]† = record { xRx<_> = xRx†<_> }
  
record deriving[⋈] {ℓ₁ ℓ₂ ℓᴿ₁ ℓᴿ₂}
  (A : Set ℓ₁) (_≼ᴬ_ : (x y : A) → Set ℓᴿ₁)
  (B : Set ℓ₂) (_≼ᴮ_ : (x y : B) → Set ℓᴿ₂) {{_ : antisymmetric _≼ᴮ_}}
  (ι : A → B) {{_ : injective ι}}
  : Set (ℓ₁ ⩛ ℓᴿ₁ ⩛ ℓᴿ₂)
  where
  constructor derive[⋈]

  field
    isoto : ∀ {x y} → x ≼ᴬ y → ι x ≼ᴮ ι y

  private
    _⋈†_ : ∀ {x y} → x ≼ᴬ y → y ≼ᴬ x → x ≡ y
    ε₁ ⋈† ε₂ = inj< _ , _ > (isoto ε₁ ⋈ isoto ε₂)

  antisymmetric[≼]† : antisymmetric _≼ᴬ_
  antisymmetric[≼]† = record { _⋈_ = _⋈†_ }

record deriving[⊚] {ℓ₁ ℓ₂ ℓᴿ₁ ℓᴿ₂}
  (A : Set ℓ₁) (_≼ᴬ_ : (x y : A) → Set ℓᴿ₁)
  (B : Set ℓ₂) (_≼ᴮ_ : (x y : B) → Set ℓᴿ₂) {{_ : transitive _≼ᴮ_}}
  (ι : A → B)
  : Set (ℓ₁ ⩛ ℓᴿ₁ ⩛ ℓᴿ₂)
  where
  constructor derive[⊚]
  field
    isoto : ∀ {x y} → x ≼ᴬ y → ι x ≼ᴮ ι y
    isofr : ∀ {x y} → ι x ≼ᴮ ι y → x ≼ᴬ y

  private
    _⊚†_ : ∀ {x y z} → y ≼ᴬ z → x ≼ᴬ y → x ≼ᴬ z
    ε₁ ⊚† ε₂ = isofr (isoto ε₁ ⊚ isoto ε₂)

  transitive[≼]† : transitive _≼ᴬ_
  transitive[≼]† = record { _⊚_ = _⊚†_ }
