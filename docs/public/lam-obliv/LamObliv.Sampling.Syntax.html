module LamObliv.Sampling.Syntax where

open import UVMVS

open import LamObliv.Common

-- well-typed values: ⸨ v : Σ ⊢ⱽ τ ⸩
infix 10 _⊢ⱽ_
data _⊢ⱽ_ {R I} (Σ : tsto R I) : ∀ (τ : type R) → Set where
  Bit : ∀ {r}
    (b : 𝔹)
    (ℓ : label)
    → Σ ⊢ⱽ Bit r ℓ
  Flip : ∀ {r}
    (b : 𝔹)
    → Σ ⊢ⱽ Flip r
  Pair : ∀ {τ₁ : type R} {τ₂ : type R}
    (e₁ : Σ ⊢ⱽ τ₁)
    (e₂ : Σ ⊢ⱽ τ₂)
    → Σ ⊢ⱽ τ₁ `× τ₂
  Addr : ∀ {τ}
    {i : idx I}
    (ε : ⟨ i ↦ τ ⟩∈ Σ)
    → Σ ⊢ⱽ Ref τ

module _ {R} {I} (Σ : tsto R I) where
  open common R (λ τ → Σ ⊢ⱽ τ) public using () renaming
    ( _,_⊢ᴱ_,_ to _,_,_⊢ᴱ_,_
    ; _⇛_      to _⊢_⇛_
    ; _⇚_      to _⊢_⇚_
    )

module _ {R} {I} {Σ : tsto R I} where
  open common R (λ τ → Σ ⊢ⱽ τ) public hiding
    ( _,_⊢ᴱ_,_
    ; _⇛_
    ; _⇚_
    )

vsto<_> : ∀ {R} {I₀} (Σ₀ : tsto R I₀) {I} (Σ : tsto R I) → Set
vsto< Σ₀ > Σ = ⟬ (λ τ → Σ ⊢ⱽ τ) ⟭[[ Σ₀ ]]

-- value store ⸨ σ : vsto Σ ⸩ 
vsto : ∀ {R : ℕ} {I : ℕ} (Σ : tsto R I) → Set
vsto Σ = vsto< Σ > Σ
