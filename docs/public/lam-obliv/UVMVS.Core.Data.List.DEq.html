module UVMVS.Core.Data.List.DEq where

open import UVMVS.Core.Init
open import UVMVS.Core.Classes

module _ {ℓ} {A : Set ℓ} {{_ : has[≡?] A}} where
  _≡?ᴸ_ : ∀ (xs ys : ⟬ A ⟭) → ≡!
  [] ≡?ᴸ [] = [≡]
  [] ≡?ᴸ (y ∷ ys) = [≢]
  (x ∷ xs) ≡?ᴸ [] = [≢]
  (x ∷ xs) ≡?ᴸ (y ∷ ys) with x ≡? y
  … | [≢] = [≢]
  … | [≡] = xs ≡?ᴸ ys

  instance
    has[≡?][list] : has[≡?] ⟬ A ⟭
    has[≡?][list] = record { _≡?_ = _≡?ᴸ_ }
  
  module _ {{_ : cor[≡?] A}} where
    abstract
      _≡⋆ᴸ_ : ∀ (xs ys : ⟬ A ⟭) → xs ≡⋆! ys
      [] ≡⋆ᴸ [] = [≡] ↯
      [] ≡⋆ᴸ (y ∷ ys) = [≢] ⟨ (λ ()) ⟩
      (x ∷ xs) ≡⋆ᴸ [] = [≢] ⟨ (λ ()) ⟩
      (x ∷ xs) ≡⋆ᴸ (y ∷ ys) with x ≡⋆ y
      … | [≢] ε = [≢] ⟨ (λ where ↯ → contra[≡] ε ↯) ⟩
      … | [≡] ε with xs ≡⋆ᴸ ys
      … | [≢] ε′ = [≢] ⟨ (λ where ↯ → contra[≡] ε′ ↯) ⟩
      … | [≡] ε′ = [≡] (change2[ (λ X Y → X ∷ Y ≡ y ∷ ys) ] ε ε′ ↯ )
      
      _≡∿ᴸ_ : ∀ (xs ys : ⟬ A ⟭) → xs ≡∿! ys [ xs ≡?ᴸ ys , xs ≡⋆ᴸ ys ]
      [] ≡∿ᴸ [] = [≡]
      [] ≡∿ᴸ (y ∷ ys) = [≢]
      (x ∷ xs) ≡∿ᴸ [] = [≢]
      (x ∷ xs) ≡∿ᴸ (y ∷ ys) with x ≡? y | x ≡⋆ y | x ≡∿ y
      … | [≢] | [≢] ε | [≢] = [≢]
      … | [≡] | [≡] ε | [≡] with xs ≡?ᴸ ys | xs ≡⋆ᴸ ys | xs ≡∿ᴸ ys
      … | [≢] | [≢] ε′ | [≢] = [≢]
      … | [≡] | [≡] ε′ | [≡] = [≡]
  
    instance
      cor[≡?][list] : cor[≡?] ⟬ A ⟭
      cor[≡?][list] = record { _≡⋆_ = _≡⋆ᴸ_ ; _≡∿_ = _≡∿ᴸ_ }
  
    _ = ⌁ AT cor[≡?] ⟬ A ⟭
  _ = ⌁ AT has[≡?] ⟬ A ⟭
