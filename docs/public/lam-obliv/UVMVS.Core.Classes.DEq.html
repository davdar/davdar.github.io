module UVMVS.Core.Classes.DEq where

open import UVMVS.Core.Init
open import UVMVS.Core.Classes.Relations
open import UVMVS.Core.Classes.Algebra

infix 10 _≡⋆!_ _≡∿!_[_,_]

data ≡! : Set where [≢] [≡] : ≡!

data _≡⋆!_ {ℓ} {A : Set ℓ} (x y : A) : Set ℓ where
  [≡] : ∀ (ε : x ≡ y) → x ≡⋆! y
  [≢] : ∀ (ε : x ≢ y) → x ≡⋆! y

data _≡∿!_[_,_] {ℓ} {A : Set ℓ} (x y : A) : ∀ (ε₁ : ≡!) (ε₃ : x ≡⋆! y) → Set where
  [≡] : ∀ {ε : x ≡ y} → x ≡∿! y [ [≡] , [≡] ε ]
  [≢] : ∀ {ε : x ≢ y} → x ≡∿! y [ [≢] , [≢] ε ]

record has[≡?] {ℓ} (A : Set ℓ) : Set ℓ where
  infix 10 _≡?_

  field
    _≡?_ : ∀ (x y : A) → ≡!
open has[≡?] {{…}} public

record cor[≡?] {ℓ} (A : Set ℓ) {{_ : has[≡?] A}} : Set ℓ where
  infix 10 _≡⋆_
  infix 10 _≡∿_

  field
    _≡⋆_ : ∀ (x y : A) → x ≡⋆! y
    _≡∿_ : ∀ (x y : A) → x ≡∿! y [ x ≡? y , x ≡⋆ y ]

  is≡[≡]<_,_> : ∀ (x y : A) (ε : x ≡ y) → (x ≡? y) ≡ [≡]
  is≡[≡]< x , .x > ↯ with x ≡? x | x ≡⋆ x | x ≡∿ x
  … | [≡] | [≡] _ | [≡] = ↯
  … | [≢] | [≢] ε | [≢] = contra[≡] ε ↯
  is≢[≡]<_,_> : ∀ (x y : A) (ε : x ≢ y) → (x ≡? y) ≡ [≢]
  is≢[≡]< x , y > ε with x ≡? y | x ≡⋆ y | x ≡∿ y 
  … | [≡] | [≡] ↯ | [≡] = contra[≡] ε ↯
  … | [≢] | [≢] _ | [≢] = ↯

  mk≡[≡]<_,_> : ∀ (x y : A) → (x ≡? y) ≡ [≡] → x ≡ y
  mk≡[≡]< x , y > with x ≡? y | x ≡⋆ y | x ≡∿ y
  … | [≡] | [≡] ↯ | [≡] = λ _ → ↯
  … | [≢] | [≢] ε | [≢] = λ ()
  mk≢[≡]<_,_> : ∀ (x y : A) → (x ≡? y) ≡ [≢] → x ≢ y
  mk≢[≡]< x , y > with x ≡? y | x ≡⋆ y | x ≡∿ y
  … | [≡] | [≡] ↯ | [≡] = λ ()
  … | [≢] | [≢] ε | [≢] = λ _ → ε
open cor[≡?] {{…}} public

--------------
-- Deriving --
--------------

module derive[cor[≡?]]
  {ℓ₁ ℓ₂}
  (A : Set ℓ₁) (B : Set ℓ₂) (ι : A → B)
  {{_ : has[≡?] A}}
  {{_ : has[≡?] B}} {{_ : cor[≡?] B}}
  {{_ : injective ι}}
  (corr[≡?]<_,_> : ∀ x y → (x ≡? y) ≡ (ι x ≡? ι y))
  where

  abstract
    _≡⋆†_ : ∀ (x y : A) → x ≡⋆! y
    x ≡⋆† y with ι x ≡⋆ ι y
    … | [≡] ε = [≡] (inj< x , y > ε)
    … | [≢] ε = [≢] ⟨ (λ where ↯ → contra[≡] ε ↯) ⟩

    _≡∿†_ : ∀ (x y : A) → x ≡∿! y [ x ≡? y , x ≡⋆† y ]
    x ≡∿† y with x ≡? y | ι x ≡? ι y | ι x ≡⋆ ι y | corr[≡?]< x , y > | ι x ≡∿ ι y
    … | [≡] | [≡] | [≡] ε | ↯ | [≡] = [≡]
    … | [≢] | [≢] | [≢] ε | ↯ | [≢] = [≢]

  cor[≡?]† : cor[≡?] A
  cor[≡?]† = record
    { _≡⋆_ = _≡⋆†_
    ; _≡∿_ = _≡∿†_
    }

------------------
-- Synthesizing --
------------------

module synthesize[has[≡?]]
  {ℓ₁ ℓ₂}
  (A : Set ℓ₁) (B : Set ℓ₂) (ι : A → B)
  {{_ : has[≡?] B}}
  {{_ : injective ι}}
  where

  _≡?†_ : ∀ (x y : A) → ≡!
  x ≡?† y = ι x ≡? ι y

  has[≡?]† : has[≡?] A
  has[≡?]† = record { _≡?_ = _≡?†_ }

module synthesize[cor[≡?]]
  {ℓ₁ ℓ₂}
  (A : Set ℓ₁) (B : Set ℓ₂) (ι : A → B)
  {{_ : has[≡?] B}} {{_ : cor[≡?] B}}
  {{_ : injective ι}}
  where

  open synthesize[has[≡?]] A B ι

  private
    instance
      _ = has[≡?]†

  open derive[cor[≡?]] A B ι (λ x y → ↯) public
