module _ {ℓ} {A : Set ℓ} {{_ : has[≡?] A}} where

  _≡?ⱽ_ : ∀ {N} (xs ys : ⟬ A ⟭[ N ]) → ≡!
  [] ≡?ⱽ [] = [≡]
  (x ∷ xs) ≡?ⱽ (y ∷ ys) with x ≡? y
  … | [≢] = [≢]
  … | [≡] = xs ≡?ⱽ ys
  
  instance
    has[≡?][vec] : ∀ {N} → has[≡?] ⟬ A ⟭[ N ]
    has[≡?][vec] = record { _≡?_ = _≡?ⱽ_ }

  abstract
    corr[≡?ⱽ]<_,_> : ∀ {N} (xs ys : ⟬ A ⟭[ N ]) → (xs ≡? ys) ≡ (ιⱽ xs ≡? ιⱽ ys)
    corr[≡?ⱽ]< [] , [] > = ↯
    corr[≡?ⱽ]< x ∷ xs , y ∷ ys > with x ≡? y
    … | [≢] = ↯
    … | [≡] = corr[≡?ⱽ]< xs , ys >
  
  module _ {N : ℕ} where
    module _ {{_ : cor[≡?] A}} where

      open derive[cor[≡?]] ⟬ A ⟭[ N ] ⟬ A ⟭ ιⱽ corr[≡?ⱽ]<_,_>
    
      instance
        cor[≡?][vec] = cor[≡?]†
  
      _ = ⌁ AT cor[≡?] ⟬ A ⟭[ N ]
    _ = ⌁ AT has[≡?] ⟬ A ⟭[ N ]
