module LamObliv.Common.Substitution where

open import UVMVS

open import LamObliv.Common.Syntax

module common-substitution (R : ℕ) (val : ∀ (τ : type R) → Set) where
  open common-syntax R val

  -- generalized substitution for an arbitrary variable in an arbitrary context
  s[∈][_,_,_,_][_,_,_]<_> :
    ∀ (τ τ′ : type R)
      (mk-var : ∀ {N} {Γ : tcxt R N} {A : acxt N} {x : idx N} (ε₁ : ⟨ x ↦ τ ⟩∈ Γ) (ε₂ : ⟨ x ↦ ᴏ ⟩∈ A) → Γ , A ⊢ᴱ τ′ , A #[ x ↦ consume (𝒦 τ′) ])
      (mk-val : ∀ {N} {Γ : tcxt R N} {A : acxt N} (v : val τ) → Γ , A ⊢ᴱ τ′ , A)
      N′ (Γ′ : tcxt R N′) (A′ : acxt N′)
      {N} (x : idx N) {Γ : tcxt R N} {A : acxt N} 
      (v : val (Γ #[ x ]))
      {y : idx N}
      (ε₁ : ⟨ y ↦ τ ⟩∈ Γ)
      (ε₂ : ⟨ y ↦ ᴏ ⟩∈ A)
    → (Γ ∸ⱽ x) ⇧ⱽ Γ′ , (A ∸ⱽ x) ⇧ⱽ A′ ⊢ᴱ τ′ , (A #[ y ↦ consume (𝒦 τ′) ] ∸ⱽ x) ⇧ⱽ A′
  s[∈][ τ , τ′ , mk-var , mk-val ][ N′ , Γ′ , A′ ]< ᴢ > {τ″ ∷ Γ} {a ∷ A} v ᴢ ε₂ = mk-val v
  s[∈][ τ , τ′ , mk-var , mk-val ][ N′ , Γ′ , A′ ]< ᴢ > {τ″ ∷ Γ} {a ∷ A} v {ꜱ y} (ꜱ ε₁) (ꜱ ε₂)
    rewrite dist[⇧/extend]< A , y , consume (𝒦 τ′) , A′ >
          = mk-var (ε₁ ⇧ᵋ Γ′) (ε₂ ⇧ᵋ A′)
  s[∈][ τ , τ′ , mk-var , mk-val ][ N′ , Γ′ , A′ ]< ꜱ x > {τ″ ∷ Γ} {a ∷ A} v {ᴢ} ᴢ ᴢ
    rewrite dist[⇧/extend]< a ∷ A ∸ⱽ ꜱ x , ᴢ , consume (𝒦 τ′) , A′ >
          = mk-var (ᴢ ⇧ᵋ Γ′) (ᴢ ⇧ᵋ A′)
  s[∈][ τ , τ′ , mk-var , mk-val ][ N′ , Γ′ , A′ ]< ꜱ x > {τ″ ∷ Γ} {a ∷ A} v (ꜱ ε₁) (ꜱ ε₂) = s[∈][ τ , τ′ , mk-var , mk-val ][ ꜱ N′ , τ″ ∷ Γ′ , a ∷ A′ ]< x > v ε₁ ε₂
  
  -- generalized substitution for an arbitrary variable
  s[∈][_,_,_,_]<_> :
    ∀ (τ τ′ : type R)
      (mk-var : ∀ {N} {Γ : tcxt R N} {A : acxt N} {x : idx N} (ε₁ : ⟨ x ↦ τ ⟩∈ Γ) (ε₂ : ⟨ x ↦ ᴏ ⟩∈ A) → Γ , A ⊢ᴱ τ′ , A #[ x ↦ consume (𝒦 τ′) ])
      (mk-val : ∀ {N} {Γ : tcxt R N} {A : acxt N} (v : val τ) → Γ , A ⊢ᴱ τ′ , A)
      {N} {Γ : tcxt R N} {A : acxt N}
      (x : idx N)
      (v : val (Γ #[ x ]))
      {y : idx N}
      (ε₁ : ⟨ y ↦ τ ⟩∈ Γ)
      (ε₂ : ⟨ y ↦ ᴏ ⟩∈ A)
    → Γ ∸ⱽ x , A ∸ⱽ x ⊢ᴱ τ′ , A #[ y ↦ consume (𝒦 τ′) ] ∸ⱽ x
  s[∈][ τ , τ′ , mk-var , mk-val ]< x > = s[∈][ τ , τ′ , mk-var , mk-val ][ ᴢ , [] , [] ]< x >

  -- substitution for terms
  s[⊢]<_> : ∀ {N}
    (x : idx N)
    {τ} {Γ : tcxt R N} {A A′ : acxt N}
    (v : val (Γ #[ x ]))
    (e : Γ , A ⊢ᴱ τ , A′)
    → Γ ∸ⱽ x , A ∸ⱽ x ⊢ᴱ τ , A′ ∸ⱽ x
  s[⊢]< x > {τ} {Γ} v (Var ε₁ ε₂ ↯) =  s[∈][ τ , τ , (λ x y → Var x y ↯) , Val ]< x > v ε₁ ε₂ 
  s[⊢]< x > v (Val w) = Val w
  s[⊢]< x > v (Bit b r ℓ) = Bit b r ℓ
  s[⊢]< x > v (Flip ρ r) = Flip ρ r
  s[⊢]< x > v (CastS {r = r} ε₁ ε₂ ↯) = s[∈][ Flip r , Bit r S , (λ x y → CastS x y ↯) , CastS-V ]< x > v ε₁ ε₂
  s[⊢]< x > v (CastS-V w) = CastS-V w
  s[⊢]< x > v (CastP e) = CastP (s[⊢]< x > v e)
  s[⊢]< x > v (MuxBit e₁ e₂ e₃) = MuxBit (s[⊢]< x > v e₁) (s[⊢]< x > v e₂) (s[⊢]< x > v e₃)
  s[⊢]< x > v (MuxFlip e₁ e₂ e₃ ε₁ ε₂) = MuxFlip (s[⊢]< x > v e₁) (s[⊢]< x > v e₂) (s[⊢]< x > v e₃) ε₁ ε₂
  s[⊢]< x > v (If {A″₁ = A″₁} {A″₂ = A″₂} e₁ e₂ e₃ ↯) rewrite dist[∸/⊔]< A″₁ , A″₂ , x > = If (s[⊢]< x > v e₁) (s[⊢]< x > v e₂) (s[⊢]< x > v e₃) ↯
  s[⊢]< x > v (Pair e₁ e₂) = Pair (s[⊢]< x > v e₁) (s[⊢]< x > v e₂)
  s[⊢]< x > v (Let e₁ e₂) = Let (s[⊢]< x > v e₁) (s[⊢]< ꜱ x > v e₂)
  s[⊢]< x > v (LetPair e₁ e₂) = LetPair (s[⊢]< x > v e₁) (s[⊢]< ꜱ (ꜱ x) > v e₂)
  s[⊢]< x > v (Ref e) = Ref (s[⊢]< x > v e)
  s[⊢]< x > v (Read/Write e₁ e₂) = Read/Write (s[⊢]< x > v e₁) (s[⊢]< x > v e₂)
  
  -- cut: substitution for variable 0
  s[⊢] : ∀ {N}
    {τ τ′} {a a′} {Γ : tcxt R N} {A A′ : acxt N}
    (v : val τ′)
    (e : τ′ ∷ Γ , a ∷ A ⊢ᴱ τ , a′ ∷ A′)
    → Γ , A ⊢ᴱ τ , A′
  s[⊢] = s[⊢]< ᴢ >
