module _ {ℓ₁ ℓ₂ ℓᴿ₁ ℓᴿ₂} {A : Set ℓ₁} {B : Set ℓ₂} {{_ : has[<][ ℓᴿ₁ ] A}} {{_ : has[<][ ℓᴿ₂ ] B}} where

  data _<⸢∧⸣_ : (x y : A ∧ B) → Set (ℓ₁ ⩛ ℓᴿ₁ ⩛ ℓᴿ₂) where
    [<] : ∀ {x₁ x₂ y₁ y₂} → x₁ < y₁ → ⟨ x₁ , x₂ ⟩ <⸢∧⸣ ⟨ y₁ , y₂ ⟩
    [≡] : ∀ {x₁ x₂ y₁ y₂} → x₁ ≡ y₁ → x₂ < y₂ → ⟨ x₁ , x₂ ⟩ <⸢∧⸣ ⟨ y₁ , y₂ ⟩

  instance
    has[≤][∧] : has[<][ ℓ₁ ⩛ ℓᴿ₁ ⩛ ℓᴿ₂ ] (A ∧ B)
    has[≤][∧] = record { _<_ = _<⸢∧⸣_ }

  module _ {{_ : cor[<] A}} {{_ : cor[<] B}} where

    abstract
      ¬xRx[<⸢∧⸣] : ∀ {x : A ∧ B} → ¬ (x <⸢∧⸣ x)
      ¬xRx[<⸢∧⸣] ([<] ε) = ¬xRx ε
      ¬xRx[<⸢∧⸣] ([≡] ε₁ ε₂) = ¬xRx ε₂
  
      _¬◇[<⸢∧⸣]_ : ∀ {x y : A ∧ B} → x <⸢∧⸣ y → y <⸢∧⸣ x → 𝟘
      [<] ε₁ ¬◇[<⸢∧⸣] [<] ε₂ = ε₁ ¬◇ ε₂
      [<] ε₁ ¬◇[<⸢∧⸣] [≡] ε₂₁ ε₂₂ rewrite ε₂₁ = ¬xRx ε₁
      [≡] ε₁₁ ε₁₂ ¬◇[<⸢∧⸣] [<] ε₂ rewrite ε₁₁ = ¬xRx ε₂
      [≡] ε₁₁ ε₁₂ ¬◇[<⸢∧⸣] [≡] ε₂₁ ε₂₂ = ε₁₂ ¬◇ ε₂₂ 
  
      _⊚[<⸢∧⸣]_ : ∀ {x y z : A ∧ B} → y <⸢∧⸣ z → x <⸢∧⸣ y → x <⸢∧⸣ z
      [<] ε₁ ⊚[<⸢∧⸣] [<] ε₂ = [<] (ε₁ ⊚ ε₂)
      [≡] ↯ ε₁ ⊚[<⸢∧⸣] [<] ε₂ = [<] ε₂
      [<] ε₁ ⊚[<⸢∧⸣] [≡] ↯ ε₂ = [<] ε₁
      [≡] ↯ ε₁ ⊚[<⸢∧⸣] [≡] ↯ ε₂ = [≡] ↯ (ε₁ ⊚ ε₂)

    instance
      irreflexive[<⸢∧⸣] : irreflexive _<⸢∧⸣_
      irreflexive[<⸢∧⸣] = record { ¬xRx = ¬xRx[<⸢∧⸣] }

      asymmetric[<⸢∧⸣] : asymmetric _<⸢∧⸣_
      asymmetric[<⸢∧⸣] = record { _¬◇_ = _¬◇[<⸢∧⸣]_ }

      transitive[<⸢∧⸣] : transitive _<⸢∧⸣_
      transitive[<⸢∧⸣] = record { _⊚_ = _⊚[<⸢∧⸣]_ }

      cor[<][∧] : cor[<] (A ∧ B)
      cor[<][∧] = record {}
  
module _ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : Set ℓ₂} {{_ : has[⋚?] A}} {{_ : has[⋚?] B}} where

  _⋚?⸢∧⸣_ : (x y : A ∧ B) → ⋚!
  ⟨ x₁ , x₂ ⟩ ⋚?⸢∧⸣ ⟨ y₁ , y₂ ⟩ with x₁ ⋚? y₁
  … | [<] = [<]
  … | [≡] = x₂ ⋚? y₂
  … | [>] = [>]

  instance
    has[⋚?][∧] : has[⋚?] (A ∧ B)
    has[⋚?][∧] = record { _⋚?_ = _⋚?⸢∧⸣_ }
  
  module _ {ℓᴿ₁ ℓᴿ₂} {{_ : has[<][ ℓᴿ₁ ] A}} {{_ : cor[<] A}} {{_ : has[<][ ℓᴿ₂ ] B}} {{_ : cor[<] B}} {{_ : cor[⋚?] A}} {{_ : cor[⋚?] B}} where
    abstract
      _⋚⋆⸢∧⸣_ : (x y : A ∧ B) → x ⋚⋆! y
      ⟨ x₁ , x₂ ⟩ ⋚⋆⸢∧⸣ ⟨ y₁ , y₂ ⟩ with x₁ ⋚⋆ y₁ 
      … | [<] ε = [<] ([<] ε)
      … | [>] ε = [>] ([<] ε)
      … | [≡] ε with x₂ ⋚⋆ y₂
      … | [<] ε′ = [<] ([≡] ε ε′)
      … | [≡] ε′ = [≡] (change2[ (λ X Y → ⟨ X , Y ⟩ ≡ ⟨ y₁ , y₂ ⟩) ] ε ε′ ↯)
      … | [>] ε′ = [>] ([≡] (◇ ε) ε′)
  
      _⋚∿⸢∧⸣_ : ∀ x y → x ⋚∿! y [ x ⋚?⸢∧⸣ y , x ⋚⋆⸢∧⸣ y ]
      ⟨ x₁ , x₂ ⟩ ⋚∿⸢∧⸣ ⟨ y₁ , y₂ ⟩ with x₁ ⋚? y₁ | x₁ ⋚⋆ y₁ | x₁ ⋚∿ y₁
      … | [<] | [<] ε | [<] = [<]
      … | [>] | [>] ε | [>] = [>]
      … | [≡] | [≡] ε | [≡] with x₂ ⋚? y₂ | x₂ ⋚⋆ y₂ | x₂ ⋚∿ y₂
      … | [<] | [<] ε′ | [<] = [<]
      … | [≡] | [≡] ε′ | [≡] = [≡]
      … | [>] | [>] ε′ | [>] = [>]
  
    instance
      cor[⋚?][∧] : cor[⋚?] (A ∧ B)
      cor[⋚?][∧] = record { _⋚⋆_ = _⋚⋆⸢∧⸣_ ; _⋚∿_ = _⋚∿⸢∧⸣_ }

module _ {ℓ₁ ℓ₂ ℓᴿ₁ ℓᴿ₂} {A : Set ℓ₁} {B : Set ℓ₂} {{_ : has[<][ ℓᴿ₁ ] A}} {{_ : has[<][ ℓᴿ₂ ] B}} where

  module _ {{_ : cor[<] A}} {{_ : cor[<] B}} where

    _ = ⌁ AT irreflexive (_<⸢∧⸣_ AT (A ∧ B → A ∧ B → Set _))
    _ = ⌁ AT asymmetric  (_<⸢∧⸣_ AT (A ∧ B → A ∧ B → Set _))
    _ = ⌁ AT transitive  (_<⸢∧⸣_ AT (A ∧ B → A ∧ B → Set _))
    _ = ⌁ AT cor[<]      (A ∧ B)

  _ = ⌁ AT has[<][ _ ] (A ∧ B)

module _ {ℓ₁ ℓ₂} {A : Set ℓ₁} {B : Set ℓ₂} {{_ : has[⋚?] A}} {{_ : has[⋚?] B}} where
  module _ {ℓᴿ₁ ℓᴿ₂} {{_ : has[<][ ℓᴿ₁ ] A}} {{_ : cor[<] A}} {{_ : has[<][ ℓᴿ₂ ] B}} {{_ : cor[<] B}} {{_ : cor[⋚?] A}} {{_ : cor[⋚?] B}} where
    _ = ⌁ AT cor[⋚?] (A ∧ B)
  _ = ⌁ AT has[⋚?] (A ∧ B)
