module UVMVS.Core.Data.Bit.Base where

open import UVMVS.Core.Init

neg : ∀ (b : 𝔹) → 𝔹
neg ɪ = ᴏ
neg ᴏ = ɪ

_⊻_ : 𝔹 → 𝔹 → 𝔹
ɪ ⊻ y = neg y
ᴏ ⊻ y = y

cond : ∀ {ℓ} {A : Set ℓ} (b : 𝔹) (x y : A) → A
cond ɪ x _ = x
cond ᴏ _ y = y

abstract
  cond[•xx]<_,_> : ∀ {ℓ} {A : Set ℓ} (b : 𝔹) (x : A) → cond b x x ≡ x
  cond[•xx]< ɪ , x > = ↯
  cond[•xx]< ᴏ , x > = ↯
